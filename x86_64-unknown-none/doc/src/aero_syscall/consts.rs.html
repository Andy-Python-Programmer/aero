175        const IN        = 0x001;
176        /// There is an exceptional condition on the file descriptor.
177        const PRI       = 0x002;
178        const OUT       = 0x004;
179        const RDNORM    = 0x040;
180        const RDBAND    = 0x080;
181        const WRNORM    = 0x100;
182        const WRBAND    = 0x200;
183        const MSG       = 0x400;
184        const ERR       = 0x008;
185        const HUP       = 0x010;
186        const RDHUP     = 0x2000;
187        const EXCLUSIVE = 1 << 28;
188        const WAKEUP    = 1 << 29;
189        const ONESHOT   = 1 << 30;
190        const ET        = 1 << 31;
191    }
192}

194#[derive(Copy, Clone)]
195#[cfg_attr(target_arch = "x86_64", repr(packed))]

198    pub data: EPollData,
199}

201impl core::fmt::Debug for EPollEvent {
202    fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
203        let events = self.events;
204
205        f.debug_struct("EPollEvent")
206            .field("events", &events)
207            .finish()
208    }
209}

415#[repr(C)]
416pub struct IfReq {
417    /// interface name, e.g. "en0"
418    pub name: [u8; IF_NAME_SIZE],
419    pub data: IfrIfru,
420}
421
422impl IfReq {
423    /// Get the interface name, e.g. "en0". [`None`] is returned if UTF-8
424    /// validation failed.
425    pub fn name(&self) -> Option<&str> {
426        let null_index = self.name.iter().position(|&x| x == 0).unwrap_or_default();
427        let name = &self.name[..null_index];
428        core::str::from_utf8(name).ok()
429    }
