67#[derive(Debug, Default, Copy, Clone, PartialEq)]
68pub enum SignalHandler {
69    Ignore,
70    #[default]
71    Default,
72    Handle(extern "C" fn(usize)),
73}
74
75impl From<u64> for SignalHandler {
76    fn from(v: u64) -> Self {
77        let v = v as i64;
78        match v {
79            SIG_IGN => SignalHandler::Ignore,
80            SIG_DFL => SignalHandler::Default,
81            v => SignalHandler::Handle(unsafe {
82                core::mem::transmute::<u64, extern "C" fn(usize)>(v as u64)
83            }),
84        }
85    }
86}
87
88impl From<SignalHandler> for usize {
89    fn from(h: SignalHandler) -> Self {
90        match h {
91            SignalHandler::Ignore => SIG_IGN as usize,
92            SignalHandler::Default => SIG_DFL as usize,
93            SignalHandler::Handle(f) => f as usize,
94        }
95    }
96}
97
98impl From<SignalHandler> for u64 {
99    fn from(h: SignalHandler) -> Self {
100        match h {
101            SignalHandler::Ignore => SIG_IGN as u64,
102            SignalHandler::Default => SIG_DFL as u64,
103            #[allow(clippy::fn_to_numeric_cast)]
104            SignalHandler::Handle(f) => f as u64,
105        }
106    }
107}

109#[repr(C)]
110#[derive(Copy, Clone, Debug)]
111pub struct SigAction {
112    pub sa_handler: u64,
113    pub sa_flags: u64,
114    pub sa_sigaction: u64,
115    pub sa_mask: u64,
116}
117
118impl SigAction {
119    pub fn new(handler: SignalHandler, mask: u64, flags: SignalFlags) -> SigAction {
120        SigAction {
121            sa_handler: handler.into(),
122            sa_flags: flags.bits(),
123            sa_sigaction: 0,
124            sa_mask: mask,
125        }
126    }
127}

144#[repr(u64)]
145#[derive(Debug)]
146pub enum SigProcMask {
147    Block = SIG_BLOCK,
148    Unblock = SIG_UNBLOCK,
149    Set = SIG_SETMASK,
150}
151
152impl From<u64> for SigProcMask {
153    fn from(v: u64) -> Self {
154        match v {
155            SIG_BLOCK => SigProcMask::Block,
156            SIG_UNBLOCK => SigProcMask::Unblock,
157            SIG_SETMASK => SigProcMask::Set,
158            _ => panic!("invalid signal procmask {}", v),
159        }
160    }
161}
162
163impl From<SigProcMask> for usize {
164    fn from(s: SigProcMask) -> Self {
165        s as u64 as usize
166    }
