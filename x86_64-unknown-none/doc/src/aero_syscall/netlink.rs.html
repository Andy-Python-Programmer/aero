7/// Aligns `len` to the netlink message alignment.
8///
9/// **Note**: This function only rounds up.
10pub const fn nlmsg_align(len: u32) -> u32 {
11    (len + NLMSG_ALIGNTO - 1) & !(NLMSG_ALIGNTO - 1)
12}
13
14const RTA_ALIGNTO: u32 = 4;
15
16pub const fn rta_align(len: u32) -> u32 {
17    (len + RTA_ALIGNTO - 1) & !(RTA_ALIGNTO - 1)
18}

20pub const fn rta_length(len: u32) -> u32 {
21    rta_align(core::mem::size_of::<rtattr>() as u32) + len
22}

144bitflags::bitflags! {
145    #[repr(transparent)]
146    pub struct MessageFlags: u16 {
147        const REQUEST = 0x1; // It is a request message.
148        const MULTI = 0x2; // Multipart message, terminated by NLMSG_DONE.
149        const ACK = 0x4; // Reply with ack, with zero or error code.
150        const ECHO = 0x8; // Echo this request.
151        const DUMP_INTR = 0x10; // Dump was inconsistent due to sequence change.
152        const DUMP_FILTERED = 0x20; // Dump was filtered as requested.
153
154        // Modifers to GET request.
155        const ROOT = 0x100; // specify tree root.
156        const MATCH = 0x200; // return all matching.
157        const ATOMIC = 0x400; // atomic GET.
158        const DUMP = MessageFlags::ROOT.bits() | MessageFlags::MATCH.bits();
159    }
160}
