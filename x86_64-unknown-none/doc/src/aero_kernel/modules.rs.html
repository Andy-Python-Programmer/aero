18//! A kernel module is an object file that contains code that can extend
19//! the kernel functionality at runtime. When a kernel module is no longer needed,
20//! it can be unloaded. Most of the device drivers are used in the form of kernel modules.
21//!
22//! ## Example
23//!
24//! ```rust,no_run
25//! fn hello_init() {}
26//! fn hello_exit() {}
27//!
28//! aero_kernel::module_init!(hello_init);
29//! aero_kernel::module_exit!(hello_exit);
30//! ```
31
32use core::mem::size_of;
33
34use crate::{drivers, extern_sym, fs};
35
36/// Inner helper function to make sure the function provided to the [`module_init`] macro
37/// has a valid function signature. This function returns the passed module init function as
38/// a const void pointer.
39
40#[derive(Debug, Copy, Clone, PartialEq, PartialOrd, Eq, Ord)]
41#[repr(C)]
42pub enum ModuleType {
43    Block = 0,
44    Other = 1,
45}
46
47#[derive(Debug, Clone)]
48#[repr(C)]
49pub struct Module {
50    pub init: *const (),
51    pub ty: ModuleType,
52}
53
54unsafe impl Sync for Module {}
55
56#[macro_export]
57macro_rules! module_init {
58    ($init_function:expr, $ty:path) => {
59        use $crate::modules::ModuleType;
60
61        #[used]
62        #[link_section = ".kernel_modules.init"]
63        static __MODULE_INIT: $crate::modules::Module = $crate::modules::Module {
64            init: $init_function as *const (),
65            ty: $ty,
66        };
67    };
68}
69
70/// This function is responsible for initializing all of the kernel modules. Since currently
71/// we cannot read the ext2 root filesystem, we link all of the kernel modules into the kernel
72/// itself (this is temporary and modules will be loaded from the filesystem in the future).
73pub(crate) fn init() {
74    let modules_start = extern_sym!(__kernel_modules_start).cast::<Module>();
75    let modules_end = extern_sym!(__kernel_modules_end).cast::<Module>();
76
77    let size = (modules_end.addr() - modules_start.addr()) / size_of::<Module>();
78    let modules = unsafe { core::slice::from_raw_parts(modules_start, size) };
79
80    unsafe {
81        // TODO: refactor this out
82        let mut modules = modules.to_vec();
83        modules.sort_by(|e, a| e.ty.cmp(&a.ty));
84
85        let mut launched_fs = false;
86
87        for module in modules {
88            log::debug!("{module:?} {launched_fs}");
89
90            if module.ty != ModuleType::Block && !launched_fs {
91                let mut address_space = crate::mem::AddressSpace::this();
92                let mut offset_table = address_space.offset_page_table();
93
94                #[cfg(target_arch = "x86_64")]
95                drivers::pci::init(&mut offset_table);
96                log::info!("loaded PCI driver");
97
98                fs::block::launch().unwrap();
99                launched_fs = true;
100            }
101
102            let init = core::mem::transmute::<*const (), fn() -> ()>(module.init);
103            init();
104        }
105    }
106}
