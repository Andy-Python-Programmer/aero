18use core::mem;
19
20use spin::Once;
21
22use super::sdt::Sdt;
23
24pub(super) const SIGNATURE: &str = "MCFG";
25
26static MCFG: Once<&'static Mcfg> = Once::new();
27
28#[repr(C, packed)]
29pub struct DeviceConfig {
30    pub base_address: u64,
31    pub pci_seg_group: u16,
32    pub start_bus: u8,
33    pub end_bus: u8,
34    pub reserved: u32,
35}
36
37/// The ACPI MCFG table describes the location of the PCI Express configuration space.
38#[repr(C, packed)]
39pub struct Mcfg {
40    header: Sdt,
41    reserved: u64,
42}
43
44impl Mcfg {
45    pub(super) unsafe fn init(&'static self) {
46        MCFG.call_once(move || self);
47    }
48
49    pub fn entry_count(&self) -> usize {
50        (self.header.length as usize - mem::size_of::<Self>()) / mem::size_of::<DeviceConfig>()
51    }
52}
53
54/// Returns true if the ACPI table contains the MCFG entry.
55///
56/// ## Notes
57/// Returns false if called before the ACPI tables were initialized.
58pub fn is_available() -> bool {
59    MCFG.get().is_some()
60}
61
62/// Return a immutable reference to the MCFG table.
63pub fn get_mcfg_table() -> &'static Mcfg {
64    MCFG.get()
65        .expect("Attempted to get the MCFG table before it was initialized")
66}
