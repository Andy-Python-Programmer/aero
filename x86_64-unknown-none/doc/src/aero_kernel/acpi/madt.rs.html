18use core::mem;
19
20use alloc::vec::Vec;
21use spin::RwLock;
22
23use super::sdt::Sdt;
24
25pub(super) const SIGNATURE: &str = "APIC";
26
27pub static IO_APICS: RwLock<Vec<&'static IoApicHeader>> = RwLock::new(Vec::new());
28pub static ISOS: RwLock<Vec<&'static MadtIntSrcOverride>> = RwLock::new(Vec::new());
29
30#[repr(C, packed)]
31pub struct Madt {
32    header: Sdt,
33    local_apic_address: u32,
34    flags: u32,
35}
36
37impl Madt {
38    pub(super) fn init(&'static self) {
39        for entry in self.iter() {
40            match entry {
41                MadtEntry::IoApic(e) => IO_APICS.write().push(e),
42                MadtEntry::IntSrcOverride(e) => ISOS.write().push(e),
43                _ => {}
44            }
45        }
46    }
47
48    fn iter(&self) -> MadtIterator {
49        unsafe {
50            MadtIterator {
51                current: (self as *const _ as *const u8).add(mem::size_of::<Self>()),
52                limit: (self as *const _ as *const u8).offset(self.header.length as isize),
53            }
54        }
55    }
56}
57
58#[derive(Clone, Copy)]
59#[repr(C, packed)]
60pub struct EntryHeader {
61    entry_type: u8,
62    length: u8,
63}
64
65#[repr(C, packed)]
66struct MadtLocalApic {
67    header: EntryHeader,
68    processor_id: u8,
69    apic_id: u8,
70    flags: u32,
71}
72
73#[repr(C, packed)]
74pub struct MadtIntSrcOverride {
75    pub header: EntryHeader,
76    pub bus: u8,
77    pub irq: u8,
78    pub global_system_interrupt: u32,
79    pub flags: u16,
80}
81
82#[repr(C, packed)]
83pub struct IoApicHeader {
84    pub header: EntryHeader,
85    pub io_apic_id: u8,
86    pub reserved: u8,
87    pub io_apic_address: u32,
88    pub global_system_interrupt_base: u32,
89}
90
91enum MadtEntry {
92    LocalApic(&'static MadtLocalApic),
93    IoApic(&'static IoApicHeader),
94    IntSrcOverride(&'static MadtIntSrcOverride),
95}
96
97struct MadtIterator {
98    current: *const u8,
99    limit: *const u8,
100}
101
102impl Iterator for MadtIterator {
103    type Item = MadtEntry;
104
105    fn next(&mut self) -> Option<Self::Item> {
106        while self.current < self.limit {
107            unsafe {
108                let entry_pointer = self.current;
109                let header = *self.current.cast::<EntryHeader>();
110
111                self.current = self.current.offset(header.length as isize);
112
113                let item = match header.entry_type {
114                    0 => MadtEntry::LocalApic(&*entry_pointer.cast::<MadtLocalApic>()),
115                    1 => MadtEntry::IoApic(&*entry_pointer.cast::<IoApicHeader>()),
116                    2 => MadtEntry::IntSrcOverride(&*entry_pointer.cast::<MadtIntSrcOverride>()),
117
118                    0x10..=0x7f => continue,
119                    0x80..=0xff => continue,
120
121                    _ => {
122                        log::warn!("unknown MADT entry with id: {}", header.entry_type);
123
124                        return None;
125                    }
126                };
127
128                return Some(item);
129            }
130        }
131
132        None
133    }
134}
