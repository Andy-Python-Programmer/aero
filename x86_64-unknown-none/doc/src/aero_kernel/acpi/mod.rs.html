41enum AcpiHeader {
42    Rsdt(&'static rsdp::Rsdt<u32>),
43    Xsdt(&'static rsdp::Rsdt<u64>),
44}
45
46pub struct AcpiTable {
47    header: AcpiHeader,
48}
49
50impl AcpiTable {
51    fn new(rsdp_address: VirtAddr) -> Self {
52        match rsdp::find_rsdt_address(rsdp_address) {
53            rsdp::RsdtAddress::Xsdt(xsdt_addr) => {
54                let xsdt = rsdp::Rsdt::<u64>::new(xsdt_addr);
55                let header = AcpiHeader::Xsdt(xsdt);
56
57                log::debug!("found XSDT at {:#x}", xsdt_addr);
58
59                Self { header }
60            }
61
62            rsdp::RsdtAddress::Rsdt(rsdt_addr) => {
63                let rsdt = rsdp::Rsdt::<u32>::new(rsdt_addr);
64                let header = AcpiHeader::Rsdt(rsdt);
65
66                log::debug!("found RSDT at {:#x}", rsdt_addr);
67
68                Self { header }
69            }
70        }
71    }
72
73    /// Lookup ACPI table entry with the provided signature.
74    pub fn lookup_entry(&self, signature: &str, index: usize) -> Option<&'static Sdt> {
75        match self.header {
76            AcpiHeader::Rsdt(rsdt) => rsdt.lookup_entry(signature, index),
77            AcpiHeader::Xsdt(xsdt) => xsdt.lookup_entry(signature, index),
78        }
79    }
80
81    pub fn revision(&self) -> u8 {
82        match self.header {
83            AcpiHeader::Rsdt(rsdt) => rsdt.header.revision,
84            AcpiHeader::Xsdt(xsdt) => xsdt.header.revision,
85        }
86    }
87}

99static ACPI_TABLE: Once<Mutex<AcpiTable>> = Once::new();
100
101pub fn get_acpi_table() -> MutexGuard<'static, AcpiTable> {
102    ACPI_TABLE.get().unwrap().lock()
103}
104
105/// Initialize the ACPI tables.
106pub fn init(rsdp_address: VirtAddr) {
107    let acpi_table = AcpiTable::new(rsdp_address);
108
109    ACPI_TABLE.call_once(|| Mutex::new(acpi_table));
110
111    let acpi_table = get_acpi_table();
112
113    macro init_table($sig:path => $ty:ty) {
114        if let Some(table) = acpi_table.lookup_entry($sig, 0) {
115            <$ty>::new(table);
116        }
117    }
118
119    if let Some(header) = acpi_table.lookup_entry(mcfg::SIGNATURE, 0) {
120        unsafe {
121            let mcfg: &'static Mcfg = header.as_ref();
122            mcfg.init();
123        }
124    }
125
126    if let Some(header) = acpi_table.lookup_entry(madt::SIGNATURE, 0) {
127        unsafe {
128            // Not a valid MADT table without the local apic address and the flags.
129            if header.data_len() < 8 {
130                log::warn!(
131                    "assertion failed: header.data_len() < 8 => {}",
132                    header.data_len()
133                );
134            } else {
135                let madt: &'static Madt = header.as_ref();
136                madt.init();
137            }
138        }
139    }
140
141    init_table!(hpet::SIGNATURE => Hpet);
142}
