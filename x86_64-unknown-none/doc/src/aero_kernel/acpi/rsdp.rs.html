23pub(super) fn validate_rsdt_checksum<T: RsdtHeader>(header: &'static T) -> bool {
24    if header.signature() == b"RSD PTR " {
25        let hptr = header as *const _ as *const u8;
26        unsafe { validate_checksum(hptr, core::mem::size_of::<T>()) }
27    } else {
28        false
29    }
30}
31
32pub(super) unsafe fn validate_checksum(ptr: *const u8, size: usize) -> bool {
33    let mut sum: u8 = 0;
34
35    for i in 0..size {
36        sum = sum.wrapping_add(*(ptr.add(i)));
37    }
38
39    sum == 0
40}

87#[repr(C, packed)]
88pub(super) struct Rsdt<T: RsdtTyp + core::marker::Sized> {
89    pub header: Sdt,
90    _phantom: PhantomData<T>,
91}
92
93impl Rsdt<u32> {
94    pub fn new(address: VirtAddr) -> &'static Self {
95        let this = unsafe { &*address.as_ptr::<Self>() };
96
97        let valid_checksum = this.header.is_valid();
98        let valid_signature = this.header.signature() == b"RSDT";
99
100        assert!(valid_checksum, "rsdp: failed to validate RSDT checksum");
101        assert!(valid_signature, "rsdp: failed to validate RSDT signature");
102
103        this
104    }
105
106    pub fn entries_count(&self) -> usize {
107        (self.header.length as usize - core::mem::size_of::<Self>()) / core::mem::size_of::<u32>()
108    }
109
110    pub fn lookup_entry(&self, signature: &str, index: usize) -> Option<&'static Sdt> {
111        let mut current = 0;
112        let header_data_address = self.header.data_address() as *const u32;
113
114        for i in 0..self.entries_count() {
115            let item_addr_virt = unsafe {
116                let ptr = header_data_address.add(i);
117                PhysAddr::new(ptr.read_unaligned() as u64).as_hhdm_virt()
118            };
119            let item = unsafe { Sdt::from_address(item_addr_virt) };
120
121            if item.signature == signature.as_bytes() {
122                if current == index {
123                    return Some(item);
124                }
125
126                current += 1;
127            }
128        }
129
130        None
131    }
132}

176pub(super) enum RsdtAddress {
177    Xsdt(VirtAddr),
178    Rsdt(VirtAddr),
179}
180
181pub(super) fn find_rsdt_address(rsdp_address: VirtAddr) -> RsdtAddress {
182    // Look for RSDP v2 header, and if it does not exists, look for RSDP v1 header.
183    let v20 = unsafe { &*rsdp_address.as_ptr::<Rsdp20>() };
184    let is_v20 = v20.revision >= 2 && v20.xsdt_address != 0;
185
186    if is_v20 {
187        let valid = validate_rsdt_checksum(v20);
188        assert!(valid, "rsdp: failed to validate RSDP v20 checksum");
189
190        let xsdt_address = PhysAddr::new(v20.xsdt_address).as_hhdm_virt();
191        RsdtAddress::Xsdt(xsdt_address)
192    } else {
193        let v10 = unsafe { &*rsdp_address.as_ptr::<Rsdp10>() };
194        let valid = validate_rsdt_checksum(v10);
195
196        assert!(valid, "rsdp: failed to validate RSDP v10 checksum");
197
198        let rsdt_address = PhysAddr::new(v10.rsdt_address as u64).as_hhdm_virt();
199        RsdtAddress::Rsdt(rsdt_address)
200    }
