18//! The FADT ACPI table contains information about fixed register blocks pertaining to power
19//! management.
20//!
21//! **Notes**: <https://wiki.osdev.org/FADT>
22
23use super::sdt::Sdt;
24
25pub const SIGNATURE: &str = "FACP";
26
27#[repr(C, packed)]
28pub struct Fadt {
29    pub header: Sdt,
30    pub firmware_ctrl: u32,
31    pub dsdt: u32,
32
33    // Field used in ACPI 1.0; no longer in use, for compatibility only
34    reserved: u8,
35
36    pub preferred_power_management: u8,
37    pub sci_interrupt: u16,
38    pub smi_command_port: u32,
39    pub acpi_enable: u8,
40    pub acpi_disable: u8,
41    pub s4_bios_req: u8,
42    pub pstate_control: u8,
43    pub pm1a_event_block: u32,
44    pub pm1b_event_block: u32,
45    pub pm1a_control_block: u32,
46    pub pm1b_control_block: u32,
47    pub pm2_control_block: u32,
48    pub pm_timer_block: u32,
49    pub gpe0_block: u32,
50    pub gpe1_block: u32,
51    pub pm1_event_length: u8,
52    pub pm1_control_length: u8,
53    pub pm2_control_length: u8,
54    pub pm_timer_length: u8,
55    pub gpe0_ength: u8,
56    pub gpe1_length: u8,
57    pub gpe1_base: u8,
58    pub c_state_control: u8,
59    pub worst_c2_latency: u16,
60    pub worst_c3_latency: u16,
61    pub flush_size: u16,
62    pub flush_stride: u16,
63    pub duty_offset: u8,
64    pub duty_width: u8,
65    pub day_alarm: u8,
66    pub month_alarm: u8,
67    pub century: u8,
68
69    // Reserved in ACPI 1.0; used since ACPI 2.0+
70    pub boot_architecture_flags: u16,
71    reserved2: u8,
72
73    pub flags: u32,
74}
