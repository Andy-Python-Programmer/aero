21use aero_syscall::prelude::{EPollEvent, EPollEventFlags};
22use aero_syscall::SyscallError;

24use alloc::sync::Arc;
25use hashbrown::HashMap;

27use crate::userland::scheduler;

29use crate::utils::sync::Mutex;

31use super::inode::{INodeInterface, PollTable};
32use super::FileSystemError;

34pub struct EPoll {
35    events: Mutex<HashMap<usize, EPollEvent>>,
36}
37
38impl EPoll {
39    // FIXME: The bitflags does not support bitwise or operations in const.
40    const PRIVATE_BITS: EPollEventFlags = EPollEventFlags::from_bits_truncate(
41        EPollEventFlags::WAKEUP.bits()
42            | EPollEventFlags::ONESHOT.bits()
43            | EPollEventFlags::ET.bits()
44            | EPollEventFlags::EXCLUSIVE.bits(),
45    );
46
47    pub fn new() -> Arc<Self> {
48        Arc::new(Self {
49            events: Mutex::new(HashMap::new()),
50        })
51    }
52
53    /// Adds an event to the interest list.
54    ///
55    /// ## Errors
56    /// * `EEXIST`: The event already exists at `fd`.
57    pub fn add_event(&self, fd: usize, event: EPollEvent) -> Result<(), SyscallError> {
58        let mut events = self.events.lock_irq();
59
60        if events.get(&fd).is_some() {
61            return Err(SyscallError::EEXIST);
62        }
63
64        events.insert(fd, event);
65        Ok(())
66    }
67
68    /// Removes an event from the interest list.
69    ///
70    /// ## Errors
71    /// * `ENOENT`: The event does not exist at `fd`.
72    pub fn remove_event(&self, fd: usize) -> Result<(), SyscallError> {
73        let mut events = self.events.lock_irq();
74
75        if events.get(&fd).is_none() {
76            return Err(SyscallError::ENOENT);
77        }
78
79        events.remove(&fd);
80        Ok(())
81    }
82
83    /// Change the settings associated with file descriptor in the interest list to the
84    /// new settings specified in event.
85    ///
86    /// ## Errors
87    /// * `ENOENT`: The event does not exist at `fd`.
88    pub fn update_event(&self, fd: usize, event: EPollEvent) -> Result<(), SyscallError> {
89        let mut events = self.events.lock_irq();
90
91        if !events.contains_key(&fd) {
92            return Err(SyscallError::ENOENT);
93        }
94
95        events.insert(fd, event);
96        Ok(())
97    }
98
99    /// Retrieves ready events, and delivers them to the caller-supplied event buffer and
100    /// returns the number of ready events if the call was successful.
101    ///
102    /// ## Arguments
103    ///
104    /// * `events`: Used to return information from the ready list about file descriptors in the
105    ///   interest list that have some events available.
106    ///
107    /// * `max_events`: Maximum number of events.
108    ///
109    /// * `timeout`: specifies the minimum number of milliseconds that epoll wait will block.
110    ///   Specifying a timeout of `-1` will block indefinitely. While specifying a timeout of `0`
111    ///   will return immediately even if there are available no events.
112    ///
113    ///
114    /// ## Blocking
115    /// Blocks the current task until either:
116    ///
117    /// * A file descriptor delivers an event.
118    /// * The call is interrupted by a signal handler.
119    /// * The timeout expires.
120    pub fn wait(
121        &self,
122        ret_events: &mut [EPollEvent],
123        max_events: usize,
124        timeout: usize,
125    ) -> Result<usize, FileSystemError> {
126        let current_task = scheduler::get_scheduler().current_task();
127        let file_table = &current_task.file_table;
128
129        let mut table = self.events.lock();
130        let mut n = 0;
131
132        let mut fds = alloc::vec![];
133        let mut poll_table = PollTable::default();
134
135        // Iterate over all the registered events and check if they are ready.
136        for (fd, epoll_event) in table.iter_mut() {
137            if n == max_events {
138                break;
139            }
140
141            ret_events[n].events = EPollEventFlags::empty();
142            ret_events[n].data = unsafe { core::mem::zeroed() };
143
144            let fd = file_table
145                .get_handle(*fd)
146                .ok_or(FileSystemError::NotSupported)?; // EINVAL
147
148            let flags = epoll_event.events;
149            let ready: EPollEventFlags = fd.inode().poll(None)?.into();
150
151            // If the event mask does not contain any poll(2) events, the event
152            // descriptor is disabled.
153            if flags == Self::PRIVATE_BITS {
154                continue;
155            }
156
157            if !(ready & flags).is_empty() {
158                ret_events[n].events = ready & flags;
159                ret_events[n].data = epoll_event.data;
160
161                if flags.contains(EPollEventFlags::ONESHOT) {
162                    // The `EPOLLONESHOT` bit that disables the descriptor when an event is
163                    // received, until the next `EPOLL_CTL_MOD` will be issued.
164                    epoll_event.events = Self::PRIVATE_BITS;
165                }
166
167                n += 1;
168                continue;
169            }
170
171            // Not ready; add the event to the poll table.
172            fd.inode().poll(Some(&mut poll_table))?;
173            fds.push((fd, epoll_event, flags));
174        }
175
176        // If all events are ready, we can return now.
177        if n > 0 || fds.is_empty() {
178            debug_assert!(fds.is_empty());
179            return Ok(n);
180        }
181
182        // Start the timer if timeout specified, if not, we can block indefinitely.
183        // If the timeout is zero, then we have to return without blocking.
184        if timeout == 0 {
185            return Ok(0);
186        }
187
188        if timeout > 0 {
189            scheduler::get_scheduler().sleep(Some(timeout * 1_000_000))?;
190        } else {
191            scheduler::get_scheduler().sleep(None)?;
192        }
193
194        'search: loop {
195            if current_task.state() == TaskState::AwaitingIoDeadline
196                && current_task.load_sleep_duration() == 0
197                && timeout > 0
198            {
199                // Timeout has expired.
200                return Ok(0);
201            }
202
203            for (fd, event, flags) in fds.iter_mut() {
204                // If the event mask does not contain any poll(2) events, the event
205                // descriptor is disabled.
206                if *flags == Self::PRIVATE_BITS {
207                    continue;
208                }
209
210                let ready: EPollEventFlags = fd.inode().poll(None)?.into();
211
212                if !(ready & *flags).is_empty() {
213                    // The event is ready; break out of the search loop and set ready
214                    // events to 1.
215                    ret_events[n].events = ready & *flags;
216                    ret_events[n].data = event.data;
217
218                    if flags.contains(EPollEventFlags::ONESHOT) {
219                        // The `EPOLLONESHOT` bit that disables the descriptor when an event is
220                        // received, until the next `EPOLL_CTL_MOD` will be issued.
221                        event.events = Self::PRIVATE_BITS;
222                    }
223
224                    n = 1;
225                    break 'search;
226                }
227
228                scheduler::get_scheduler().await_io()?;
229            }
230        }
231
232        Ok(n)
233    }
234}
