18//! General implementation for file system caching. Stuff like inode needs to be cached
19//! to improve performance and in this case looking up inode data from an IO device such
20//! as a disk is very slow, so storing previously accessed inode data in memory makes file
21//! system access much quicker.
22//!
23//! ## Notes
24//! * <https://wiki.osdev.org/File_Systems>
25

47// NOTE: We require a custom wrapper around [`Arc`] and [`Weak`] since we need to be able
48// to move the cache item from the used list to the unused list when the cache item is dropped.
49// This would require us to implement a custom drop handler implementation.
50pub struct CacheArc<T: CacheDropper>(Arc<T>);
51
52impl<T: CacheDropper> CacheArc<T> {
53    /// Constructs a new `CacheArc<T>`.
54    #[inline]
55    pub fn new(data: T) -> Self {
56        Self(Arc::new(data))
57    }
58
59    pub fn downgrade(&self) -> CacheWeak<T> {
60        CacheWeak(Arc::downgrade(&self.0))
61    }
62}
63
64impl<T: CacheDropper> core::ops::Deref for CacheArc<T> {
65    type Target = Arc<T>;
66
67    fn deref(&self) -> &Self::Target {
68        &self.0
69    }
70}
71
72impl<T: CacheDropper> Clone for CacheArc<T> {
73    /// Makes a clone of the `CacheArc<T>` pointer.
74    ///
75    /// This creates another pointer to the same allocation, increasing the
76    /// strong reference count.
77    #[inline]
78    fn clone(&self) -> Self {
79        Self(self.0.clone())
80    }
81}
82
83impl<T: CacheDropper> Drop for CacheArc<T> {
84    /// Drops the `ArcCache`.
85    ///
86    /// This will decrement the strong reference count. If the strong reference
87    /// count reaches zero then the only other references (if any) are
88    /// [`CacheWeak`], so we `drop` the inner value.
89    fn drop(&mut self) {
90        let strong_count = Arc::strong_count(&self.0);
91
92        if strong_count == 1 {
93            self.drop_this(self.0.clone());
94        }
95    }
96}
97
98impl<T: CacheDropper> From<Arc<T>> for CacheArc<T> {
99    /// Converts an `Arc<T>` into a `CacheArc<T>`.
100    fn from(data: Arc<T>) -> Self {
101        Self(data)
102    }
103}

105pub struct CacheWeak<T: CacheDropper>(Weak<T>);
106
107impl<T: CacheDropper> CacheWeak<T> {
108    /// Constructs a new `Weak<T>`, without allocating any memory.
109    /// Calling [`upgrade`] on the return value always gives [`None`].
110    pub fn new() -> Self {
111        Self(Weak::new())
112    }
113
114    /// Attempts to upgrade the Weak pointer to an Arc, delaying dropping of the inner
115    /// value if successful.
116    ///
117    /// Returns [`None`] if the inner value has since been dropped.
118    pub fn upgrade(&self) -> Option<CacheArc<T>> {
119        Some(self.0.upgrade()?.into())
120    }
121}
122
123impl<T: CacheDropper> Clone for CacheWeak<T> {
124    /// Makes a clone of the `CacheWeak<T>` pointer.
125    #[inline]
126    fn clone(&self) -> Self {
127        Self(self.0.clone())
128    }
129}
130
131impl<T: CacheDropper> Default for CacheWeak<T> {
132    fn default() -> Self {
133        Self(Default::default())
134    }
135}

137pub trait CacheDropper {
138    fn drop_this(&self, this: Arc<Self>);
139}

141pub trait CacheKey: Hash + Ord + Borrow<Self> + Debug {}
142
143impl<T> CacheKey for T where T: Hash + Ord + Borrow<Self> + Debug {}
144
145pub trait Cacheable<K: CacheKey>: Sized {
146    fn cache_key(&self) -> K;
147}
148
149pub struct CacheItem<K: CacheKey, V: Cacheable<K>> {
150    cache: Weak<Cache<K, V>>,
151    value: V,
152    /// Whether the cache item has active strong references associated
153    /// with it.
154    used: AtomicBool,
155}
156
157impl<K: CacheKey, V: Cacheable<K>> CacheItem<K, V> {
158    pub fn new(cache: &Weak<Cache<K, V>>, value: V) -> CacheArc<Self> {
159        CacheArc::new(Self {
160            cache: cache.clone(),
161            value,
162            used: AtomicBool::new(false),
163        })
164    }
165
166    pub fn is_used(&self) -> bool {
167        self.used.load(Ordering::SeqCst)
168    }
169
170    pub fn set_used(&self, yes: bool) {
171        self.used.store(yes, Ordering::SeqCst);
172    }
173}
174
175impl<K: CacheKey, V: Cacheable<K>> ops::Deref for CacheItem<K, V> {
176    type Target = V;
177
178    fn deref(&self) -> &Self::Target {
179        &self.value
180    }
181}
182
183unsafe impl<K: CacheKey, V: Cacheable<K>> Sync for CacheItem<K, V> {}

185struct CacheIndex<K: CacheKey, V: Cacheable<K>> {
186    used: hashbrown::HashMap<K, Weak<CacheItem<K, V>>>,
187    /// Cache items that are longer have any active strong references associated
188    /// with them. These are stored in the cache index so, if the item is
189    /// accessed again, we can re-use it; reducing required memory allocation
190    /// and I/O (if applicable).
191    unused: lru::LruCache<K, Arc<CacheItem<K, V>>>,
192}
193
194pub struct Cache<K: CacheKey, V: Cacheable<K>> {
195    index: BMutex<CacheIndex<K, V>>,
196    self_ref: Weak<Cache<K, V>>,
197}
198
199impl<K: CacheKey, V: Cacheable<K>> Cache<K, V> {
200    pub fn new() -> Arc<Self> {
201        Arc::new_cyclic(|this| Cache::<K, V> {
202            index: BMutex::new(CacheIndex {
203                used: hashbrown::HashMap::new(),
204                unused: lru::LruCache::new(NonZeroUsize::new(4096).unwrap()),
205            }),
206            self_ref: this.clone(),
207        })
208    }
209
210    pub fn clear(&self) {
211        let mut index_mut = self.index.lock();
212
213        index_mut.unused.clear();
214        index_mut.used.clear();
215    }
216
217    pub fn make_item_cached(&self, value: V) -> CacheArc<CacheItem<K, V>> {
218        let item = CacheItem::<K, V>::new(&self.self_ref, value);
219
220        self.index
221            .lock()
222            .used
223            .insert(item.cache_key(), Arc::downgrade(&item));
224
225        item.set_used(true);
226        item
227    }
228
229    pub fn make_item_no_cache(&self, value: V) -> CacheArc<CacheItem<K, V>> {
230        CacheItem::<K, V>::new(&Weak::default(), value)
231    }
232
233    pub fn get(&self, key: K) -> Option<CacheArc<CacheItem<K, V>>> {
234        let mut index = self.index.lock();
235
236        if let Some(entry) = index.used.get(&key) {
237            let entry = entry.upgrade()?;
238            Some(CacheArc::from(entry))
239        } else if let Some(entry) = index.unused.pop(&key) {
240            entry.set_used(true);
241            index.used.insert(key, Arc::downgrade(&entry));
242
243            Some(entry.into())
244        } else {
245            None
246        }
247    }
248
249    pub fn rehash<F>(&self, item: CacheArc<CacheItem<K, V>>, update: F)
250    where
251        F: FnOnce(),
252    {
253        let mut index = self.index.lock();
254        let key = item.cache_key();
255
256        index.used.remove(&key).expect("cache: item is not used");
257        update();
258
259        let new_key = item.cache_key();
260        index.used.insert(new_key, Arc::downgrade(&item));
261    }
262
263    pub fn log(&self) {
264        let index = self.index.lock();
265
266        log::debug!("Cache:");
267
268        log::debug!("\t Used entries:    {}", index.used.len());
269        for (key, item) in index.used.iter() {
270            log::debug!("\t\t {:?} -> {:?}", key, item.strong_count());
271        }
272
273        log::debug!("\t Unused entries:  {}", index.unused.len());
274        for (key, item) in index.unused.iter() {
275            log::debug!("\t\t {:?} -> {:?}", key, Arc::strong_count(item))
276        }
277    }
278
279    /// Removes the item with the provided `key` from the cache.
280    pub fn remove(&self, key: &K) {
281        let mut index = self.index.lock();
282
283        if index.used.remove(key).is_none() {
284            let _ = index.unused.pop(key);
285        }
286    }
287
288    fn mark_item_unused(&self, item: CacheArc<CacheItem<K, V>>) {
289        item.set_used(false);
290
291        let mut index = self.index.lock();
292        let key = item.cache_key();
293
294        index.used.remove(&key);
295        index.unused.put(key, item.0.clone());
296    }
297}
298
299impl<K: CacheKey, T: Cacheable<K>> CacheDropper for CacheItem<K, T> {
300    fn drop_this(&self, this: Arc<Self>) {
301        if let Some(cache) = self.cache.upgrade() {
302            if self.is_used() {
303                cache.mark_item_unused(this.into());
304            }
305        }
306    }
307}

358impl Cacheable<DirCacheKey> for DirEntry {
359    fn cache_key(&self) -> DirCacheKey {
360        let this = self.data.lock();
361
362        if let Some(parent) = this.parent.as_ref() {
363            (parent.cache_marker, this.name.clone())
364        } else {
365            (0x00, this.name.clone())
366        }
367    }
368}

370pub trait DirCacheImpl {
371    fn absolute_path(&self) -> PathBuf;
372}
373
374impl DirCacheImpl for DirCacheItem {
375    fn absolute_path(&self) -> PathBuf {
376        let mut current_entry = Some(self.clone());
377        let mut path_nodes = Vec::new();
378        let mut result = PathBuf::new();
379
380        // We need to collect all of the path nodes, reverse them and then join them
381        // with the path separator.
382        while let Some(entry) = current_entry {
383            path_nodes.push(entry.name());
384            current_entry = entry.data.lock().parent.clone();
385        }
386
387        for node in path_nodes.iter().rev() {
388            result.push(node.as_str());
389            // result.push_str(node);
390
391            // // If we are not at the root node, we need to add the path separator.
392            // if node != "/" {
393            //     result.push('/');
394            // }
395        }
396
397        result
398    }
399}

401#[inline]
402pub fn clear_inode_cache() {
403    if let Some(cache) = INODE_CACHE.get() {

405    }
