50pub struct CachedPage {
51    owner: Weak<dyn CachedAccess>,
52    offset: usize,
53    page: PhysFrame,
54    dirty: AtomicBool,
55    dirty_mappings: Mutex<Vec<DirtyMapping>>,
56}
57
58impl CachedPage {
59    fn new(owner: Weak<dyn CachedAccess>, offset: usize) -> Self {
60        let k = Self {
61            owner,
62            offset,
63            page: FRAME_ALLOCATOR
64                .allocate_frame()
65                .expect("page_cache: out of memory"),
66            dirty: AtomicBool::new(false),
67            dirty_mappings: Mutex::new(Vec::new()),
68        };
69        // TODO: temporary hack. i mean this is fine but is there a cleaner way to do this. this is
70        // required since when the VM for the process umaps a page that contains a cached page, it
71        // will unmap this page which will decrease the refcnt to 0 and deallocate it.
72        get_vm_frames().unwrap()[k.page.start_address().as_u64() as usize / 4096usize]
73            .inc_ref_count();
74        k
75    }
76
77    fn data_mut(&self) -> &mut [MaybeUninit<u8>] {
78        let data_ptr = self
79            .page
80            .start_address()
81            .as_hhdm_virt()
82            .as_mut_ptr::<MaybeUninit<u8>>();
83
84        // SAFETY: It is safe to create a slice of MaybeUninit<T> because it has the same
85        // size and alignment as T.
86        unsafe { core::slice::from_raw_parts_mut(data_ptr, Size4KiB::SIZE as usize) }
87    }
88
89    pub fn data_addr(&self) -> PhysAddr {
90        self.page.start_address()
91    }
92
93    pub fn page(&self) -> PhysFrame {
94        self.page
95    }
96
97    fn make_key(device: &Weak<dyn CachedAccess>, offset: usize) -> PageCacheKey {
98        (device.as_ptr().addr(), offset)
99    }
100
101    /// Returns whether the page has been marked dirty.
102    fn is_dirty(&self) -> bool {
103        self.dirty.load(Ordering::SeqCst)
104    }
105
106    pub fn mark_dirty(&self) {
107        self.dirty.store(true, Ordering::SeqCst);
108    }
109
110    fn device(&self) -> Arc<dyn CachedAccess> {
111        self.owner.upgrade().unwrap()
112    }
113
114    fn sync(&self) {
115        if !self.is_dirty() {
116            return;
117        }
118
119        // Commit the changes made to the cache to the owner.
120        let owner = self.device();
121        let offset_bytes = self.offset * Size4KiB::SIZE as usize;
122        owner.write_direct(offset_bytes, self.page);
123
124        for mut mapping in self.dirty_mappings.lock_irq().drain(..) {
125            let mut offset_table = mapping.addr_space.offset_page_table();
126            offset_table
127                .unmap(Page::<Size4KiB>::containing_address(mapping.addr))
128                .unwrap()
129                .1
130                .flush();
131        }
132
133        self.dirty.store(false, Ordering::SeqCst);
134    }
135}
136
137impl Drop for CachedPage {
138    fn drop(&mut self) {
139        self.sync()
140    }
141}
142
143impl Cacheable<PageCacheKey> for CachedPage {
144    fn cache_key(&self) -> PageCacheKey {
145        Self::make_key(&self.owner, self.offset)
146    }
147}

150    pub(in crate::fs) static ref PAGE_CACHE: Arc<Cache<PageCacheKey, CachedPage>> = Cache::new();
151}

153impl Cache<PageCacheKey, CachedPage> {
154    /// Returns the cached page at the given offset, if not present, it will be allocated,
155    /// initialized with the data on the disk and placed in the page cache.
156    ///
157    /// ## Arguments
158    ///
159    /// * `device` - The device to get the page from.
160    /// * `offset` - The offset in bytes to the data. This will be rounded down to the nearest page
161    ///   boundary.
162    pub fn get_page(&self, device: &Weak<dyn CachedAccess>, offset: usize) -> PageCacheItem {
163        let cache_offset = offset / Size4KiB::SIZE as usize;
164        let cache_key = CachedPage::make_key(device, cache_offset);
165
166        if let Some(page) = PAGE_CACHE.get(cache_key) {
167            return page;
168        }
169
170        let page = CachedPage::new(device.clone(), cache_offset);
171        let device = device.upgrade().expect("page_cache: device dropped");
172
173        let aligned_offset = align_down(offset as u64, Size4KiB::SIZE) as usize;
174        device
175            .read_direct(aligned_offset, page.page())
176            .expect("page_cache: failed to read block");
177
178        PAGE_CACHE.make_item_cached(page)
179    }
180}

184pub struct DirtyRef<T: Sized> {
185    cache: PageCacheItem,
186    ptr: *mut T,
187}
188
189impl<T> DirtyRef<T> {
190    pub fn new(device: &Weak<dyn CachedAccess>, offset: usize) -> Self {
191        let cache = PAGE_CACHE.get_page(device, offset);
192
193        let ptr_offset = offset % Size4KiB::SIZE as usize;
194        let ptr = &cache.data_mut()[ptr_offset..ptr_offset + core::mem::size_of::<T>()];
195
196        Self {
197            ptr: ptr.as_ptr() as *mut T,
198            cache,
199        }
200    }
201}
202
203impl<T> Deref for DirtyRef<T> {
204    type Target = T;
205
206    fn deref(&self) -> &Self::Target {
207        unsafe { &*self.ptr }
208    }
209}
210
211impl<T> DerefMut for DirtyRef<T> {
212    fn deref_mut(&mut self) -> &mut Self::Target {
213        self.cache.mark_dirty();
214        unsafe { &mut *self.ptr }
215    }
216}

221pub trait BlockDeviceInterface: Send + Sync {
222    fn block_size(&self) -> usize;
223
224    fn read_dma(&self, sector: usize, start: PhysAddr, size: usize) -> Option<usize>;
225    fn write_dma(&self, sector: usize, start: PhysAddr, size: usize) -> Option<usize>;
226
227    fn read_block(&self, sector: usize, dest: &mut [MaybeUninit<u8>]) -> Option<usize>;
228    fn write_block(&self, sector: usize, buf: &[u8]) -> Option<usize>;
229}
230
231pub trait CachedAccess: Send + Sync {
232    fn sref(&self) -> Weak<dyn CachedAccess>;
233
234    fn read_direct(&self, offset: usize, dest: PhysFrame) -> Option<usize>;
235    fn write_direct(&self, offset: usize, src: PhysFrame) -> Option<usize>;
236
237    fn read(&self, mut offset: usize, dest: &mut [MaybeUninit<u8>]) -> Option<usize> {
238        let mut loc = 0;
239
240        while loc < dest.len() {
241            let page = PAGE_CACHE.get_page(&self.sref(), offset);
242
243            let page_offset = offset % Size4KiB::SIZE as usize;
244            let size = core::cmp::min(Size4KiB::SIZE as usize - page_offset, dest.len() - loc);
245
246            let data = &page.data_mut()[page_offset..page_offset + size];
247            dest[loc..loc + size].copy_from_slice(data);
248
249            core::mem::forget(page);
250
251            loc += size;
252            offset = align_down(offset as u64 + Size4KiB::SIZE, Size4KiB::SIZE) as usize;
253        }
254
255        Some(loc)
256    }
257
258    /// Writes the given data to the device at the given offset and returns the
259    /// number of bytes written.
260    ///
261    /// ## Notes
262    ///
263    /// * This function does **not** sync the written data to the disk.
264    fn write(&self, mut offset: usize, buffer: &[u8]) -> Option<usize> {
265        let mut loc = 0;
266
267        while loc < buffer.len() {
268            // TODO: If it is not found in the page cache, then, when the write perfectly falls on
269            // page size boundaries, the page is not even read from disk, but allocated and
270            // immediately marked dirty.
271            let page = PAGE_CACHE.get_page(&self.sref(), offset);
272
273            let page_offset = offset % Size4KiB::SIZE as usize;
274            let size = core::cmp::min(Size4KiB::SIZE as usize - page_offset, buffer.len() - loc);
275
276            MaybeUninit::copy_from_slice(
277                &mut page.data_mut()[page_offset..page_offset + size],
278                &buffer[loc..loc + size],
279            );
280
281            page.mark_dirty();
282            page.sync();
283
284            loc += size;
285            offset = align_down(offset as u64 + Size4KiB::SIZE, Size4KiB::SIZE) as usize;
286        }
287
288        Some(loc)
289    }
290}

294/// Installs the provided block `device` into the filesyetm.
295pub fn install_block_device(dev: Arc<BlockDevice>) -> Result<()> {
296    let mut devs = BLOCK_DEVS.lock();
297    install_device(dev.clone())?;
298
299    log::debug!("block: installed block device {}", dev.name());
300    devs.insert(dev.id, dev);
301
302    Ok(())
303}

345        self.dev.write_block(sector, buf)

367        )
368    }

387struct PartitionBlockDevice {
388    offset: usize, // offset in sectors
389    size: usize,   // capacity in sectors
390    device: Arc<dyn BlockDeviceInterface>,
391}
392
393impl PartitionBlockDevice {
394    fn new(offset: usize, size: usize, device: Arc<dyn BlockDeviceInterface>) -> Arc<Self> {
395        Arc::new(Self {
396            offset,
397            size,
398            device,
399        })
400    }
401}
402
403impl BlockDeviceInterface for PartitionBlockDevice {
404    fn read_block(&self, sector: usize, dest: &mut [MaybeUninit<u8>]) -> Option<usize> {
405        if sector >= self.size {
406            return None;
407        }
408
409        self.device.read_block(self.offset + sector, dest)
410    }
411
412    fn write_block(&self, sector: usize, buf: &[u8]) -> Option<usize> {
413        if sector >= self.size {
414            return None;
415        }
416
417        self.device.write_block(self.offset + sector, buf)
418    }
419
420    fn write_dma(&self, sector: usize, start: PhysAddr, size: usize) -> Option<usize> {
421        if sector >= self.size {
422            return None;
423        }
424
425        self.device.write_dma(self.offset + sector, start, size)
426    }
427
428    fn read_dma(&self, sector: usize, start: PhysAddr, size: usize) -> Option<usize> {
429        if sector >= self.size {
430            return None;
431        }
432
433        self.device.read_dma(self.offset + sector, start, size)
434    }
435
436    fn block_size(&self) -> usize {
437        self.device.block_size()
438    }
439}

441pub fn launch() -> Result<()> {
442    let mut blocks_copy = Vec::<Arc<BlockDevice>>::new();
443
444    for (_, device) in BLOCK_DEVS.lock().iter() {
445        blocks_copy.push(device.clone());
446    }
447
448    for block in blocks_copy {
449        if let Some(gpt) = Gpt::new(&block) {
450            log::info!("block: found GPT on {}!", block.name());
451
452            for (i, entry) in gpt
453                .entries()
454                .iter()
455                .enumerate()
456                .filter(|(_, e)| e.is_used())
457            {
458                let start = entry.start_lba() as usize;
459                let size = entry.size() as usize;
460
461                log::info!(
462                    "gpt: found partition (name=`{}`, start={:#x}, size{:#x})!",
463                    entry.partition_name(),
464                    start,
465                    size
466                );
467
468                let name = alloc::format!("{}p{}", block.name(), i);
469                let partition_device = PartitionBlockDevice::new(start, size, block.clone());
470                let device = BlockDevice::new(name, partition_device);
471
472                install_block_device(device.clone())?;
473
474                // Check what filesystem is on this partition and mount it.
475                if let Some(ext2) = Ext2::new(device.clone()) {
476                    log::info!("gpt: found ext2 filesystem on {}!", device.name());
477
478                    super::ROOT_FS.call_once(|| ext2.clone());
479                    super::ROOT_DIR.call_once(|| ext2.root_dir());
480                }
481            }
482        }
483    }
484
485    super::devfs::init()?;
486    log::info!("installed devfs");
487
488    super::procfs::init()?;
489    log::info!("installed procfs");
490
491    Ok(())
