28pub struct GroupDescriptors {
29    descriptors: RwLock<Box<[disk::GroupDescriptor]>>,
30    ext2: Weak<Ext2>,
31}
32
33impl GroupDescriptors {
34    /// Reads the block group descriptors from the disk.
35    ///
36    /// ## Arguments
37    ///
38    /// * `ext2` - Weak pointer to the EXT2 filesystem. This function does not call `upgrade` on the
39    ///   pointer so, it is valid to pass a semi-constructed `Arc<T>` pointer (e.g through
40    ///   `Arc::new_cyclic`) though invoking other functions on the group descriptors require the
41    ///   pointer to be fully constructed).
42    ///
43    /// * `superblock` - Reference to the EXT2 superblock.
44    ///
45    /// * `device` - Block device to read the group descriptors from.
46    pub fn new(
47        ext2: Weak<Ext2>,
48        device: &BlockDevice,
49        superblock: &disk::SuperBlock,
50    ) -> Option<Self> {
51        let bgdt_len = superblock.bgdt_len();
52        let mut bgdt = Box::<[disk::GroupDescriptor]>::new_uninit_slice(bgdt_len);
53
54        device.read(superblock.bgdt_block(), bgdt.as_bytes_mut())?;
55
56        // SAFETY: We have initialized the BGD (Block Group Descriptor Table) above.
57        let bgdt = unsafe { bgdt.assume_init() };
58        Some(Self {
59            descriptors: RwLock::new(bgdt),
60            ext2,
61        })
62    }
63
64    // XXX: The free inodes are managed by bitmaps. An EXT2 filesystem contains
65    // several block groups where each group contains a bitmap for blocks and
66    // a bitmap for inodes and their free counts. The group descriptors are located
67    // after the super block.
68
69    /// Returns the index of the block group which has free block(s)
70    /// available.
71    pub fn find_free_block(&self) -> Option<usize> {
72        let (index, _) = self
73            .descriptors
74            .read()
75            .iter()
76            .enumerate()
77            .find(|(_, e)| e.free_blocks_count >= 1)?;
78
79        Some(index)
80    }
81
82    /// Returns the index of the block group which has free inode(s)
83    /// available.
84    pub fn find_free_inode(&self) -> Option<usize> {
85        let (index, _) = self
86            .descriptors
87            .read()
88            .iter()
89            .enumerate()
90            .find(|(_, e)| e.free_inodes_count >= 1)?;
91
92        Some(index)
93    }
94
95    pub fn find_inode(&self, id: usize) -> Option<Box<disk::INode>> {
96        let fs = self.ext2.upgrade()?;
97        let this = self.descriptors.read();
98        let superblock = &fs.superblock;
99
100        // There is one inode table per block group and can be located by
101        // the `inode_table` offset in the group descriptor. Also there are
102        // `inodes_per_group` inodes per table.
103        let ino_per_group = superblock.inodes_per_group as usize;
104
105        let ino_block_group = (id - 1) / ino_per_group;
106        let ino_table_index = (id - 1) % ino_per_group;
107
108        let group_descriptor = this[ino_block_group];
109        let table_offset = group_descriptor.inode_table as usize * superblock.block_size();
110
111        let mut inode = Box::<disk::INode>::new_uninit();
112
113        fs.block.read(
114            table_offset + (ino_table_index * core::mem::size_of::<disk::INode>()),
115            inode.as_bytes_mut(),
116        )?;
117
118        // SAFETY: We have initialized the inode above.
119        let inode = unsafe { inode.assume_init() };
120        Some(inode)
121    }
122
123    /// Allocates a block pointer using the first fit allocation strategy.
124    pub fn alloc_block_ptr(&self) -> Option<usize> {
125        let fs = self.ext2.upgrade()?;
126        let blocks_per_group = fs.superblock.blocks_per_group as usize;
127
128        if let Some(block_group_idx) = self.find_free_block() {
129            let mut descriptors = self.descriptors.write();
130            let block_group = &mut descriptors[block_group_idx];
131
132            let mut bitmap = Bitmap::new(&fs, block_group.block_bitmap as usize)?;
133            let block_id = block_group_idx * blocks_per_group + bitmap.alloc()?;
134
135            block_group.free_blocks_count -= 1;
136            drop(descriptors);
137
138            // TODO: decrement the number of free blocks in the superblock.
139            return Some(block_id);
140        }
141
142        None
143    }
144
145    /// Allocates a new inode using the first fit allocation strategy.
146    pub fn alloc_inode(&self) -> Option<usize> {
147        let fs = self.ext2.upgrade()?;
148        let ino_per_group = fs.superblock.inodes_per_group as usize;
149
150        if let Some(block_group_idx) = self.find_free_inode() {
151            let mut descriptors = self.descriptors.write();
152            let block_group = &mut descriptors[block_group_idx];
153
154            let mut bitmap = Bitmap::new(&fs, block_group.inode_bitmap as usize)?;
155            // Since inode numbers start from 1 rather than 0, the first bit in the first block
156            // group's inode bitmap represent inode number 1. Thus, we add 1 to the allocated
157            // inode number.
158            let inode_id = block_group_idx * ino_per_group + bitmap.alloc()? + 1;
159
160            block_group.free_inodes_count -= 1;
161            drop(descriptors); // release the lock
162
163            return Some(inode_id);
164        }
165
166        None
167    }
168}
169
170struct Bitmap {
171    bitmap: Box<[u8]>,
172    fs: Weak<Ext2>,
173    offset: usize,
174}
175
176impl Bitmap {
177    /// Reads the bitmap at `block` from the disk. The bitmap is required have
178    /// a size of `block_size` bytes.
179    ///
180    /// **Note**: Any changes to the bitmap will be written back to the disk when the
181    /// bitmap has been dropped.
182    fn new(fs: &Arc<Ext2>, block: usize) -> Option<Self> {
183        let block_size = fs.superblock.block_size();
184        let offset = block * block_size;
185
186        let mut bitmap = Box::<[u8]>::new_uninit_slice(block_size);
187
188        fs.block.read(offset, &mut bitmap)?;
189
190        // SAFETY: We have initialized the bitmap above.
191        let bitmap = unsafe { bitmap.assume_init() };
192
193        Some(Self {
194            bitmap,
195            offset,
196            fs: Arc::downgrade(fs),
197        })
198    }
199
200    /// Allocates a free bit in the bitmap and returns its index.
201    pub fn alloc(&mut self) -> Option<usize> {
202        for (i, byte) in self.bitmap.iter_mut().enumerate() {
203            if *byte == u8::MAX {
204                continue;
205            }
206
207            for bit in 0..8 {
208                if !byte.get_bit(bit) {
209                    byte.set_bit(bit, true);
210
211                    return Some(i * 8 + bit);
212                }
213            }
214        }
215
216        None
217    }
218}
219
220impl Drop for Bitmap {
221    fn drop(&mut self) {
222        let fs = self
223            .fs
224            .upgrade()
225            .expect("ext2: filesystem has been dropped");
226
227        fs.block.write(self.offset, &self.bitmap);
228    }
229}
