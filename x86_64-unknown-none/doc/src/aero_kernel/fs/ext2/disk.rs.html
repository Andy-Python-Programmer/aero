92impl SuperBlock {
93    pub const MAGIC: u16 = 0xef53;
94
95    /// Returns the number of entries per block.
96    pub fn entries_per_block(&self) -> usize {
97        self.block_size() / core::mem::size_of::<u32>()
98    }
99
100    pub fn revision(&self) -> Revision {
101        match self.rev_level {
102            0 => Revision::Revision0,
103            1 => Revision::Revision1,
104            revison => unreachable!("ext2: invalid revison {revison}"),
105        }
106    }
107
108    /// Returns the size of a block in bytes.
109    pub fn block_size(&self) -> usize {
110        1024usize << self.log_block_size
111    }
112
113    /// Returns the length of the BGDT.
114    pub fn bgdt_len(&self) -> usize {
115        self.blocks_count.div_ceil(self.blocks_per_group) as usize
116    }
117
118    pub fn bgdt_block(&self) -> usize {
119        // XXX: The block group descriptors are always located in the block immediately
120        // following the superblock.
121        let block_size = self.block_size();
122
123        if block_size >= 2048 {
124            block_size
125        } else {
126            block_size * 2
127        }
128    }
129}

146#[derive(Debug)]
147#[repr(C)]
148pub struct DirEntry {
149    pub inode: u32,
150    pub entry_size: u16,
151    pub name_size: u8,
152    pub file_type: u8,
153    name: IncompleteArrayField<u8>,
154}
155
156impl DirEntry {
157    pub fn set_name(&mut self, name: &str) {
158        assert!(name.len() < u8::MAX as usize);
159
160        self.name_size = name.len() as u8;
161
162        let name_bytes = unsafe { self.name.as_mut_slice(name.len()) };
163        name_bytes.copy_from_slice(name.as_bytes());
164    }
165
166    pub fn name(&self) -> &str {
167        unsafe { core::str::from_utf8_unchecked(self.name.as_slice(self.name_size as usize)) }
168    }
169
170    #[inline]
171    pub fn is_used(&self) -> bool {
172        // value of 0 indicates that the entry is not used
173        self.inode != 0
174    }
175}

177#[repr(u8)]
178#[derive(PartialEq, Copy, Clone)]
179pub enum FileType {
180    Unknown = 0,
181    Fifo = 1,
182    CharDev = 2,
183    Directory = 4,
184    BlockDev = 6,
185    File = 8,
186    Symlink = 10,
187    Socket = 12,
188}
189
190impl FileType {
191    pub fn bits(&self) -> u16 {
192        let val = *self as u8;
193        (val as u16) << 12
194    }
195}
196
197impl From<FileType> for inode::FileType {
198    fn from(ty: FileType) -> Self {
199        match ty {
200            FileType::Symlink => Self::Symlink,
201            FileType::Directory => Self::Directory,
202            FileType::BlockDev | FileType::CharDev => Self::Device,
203
204            _ => Self::File,
205        }
206    }
207}

211pub struct INode {
212    type_and_perm: u16,
213    pub user_id: u16,
214    size_lower: u32,
215    last_access: u32,
216    pub creation_time: u32,
217    last_modification: u32,
218    pub deletion_time: u32,
219    pub group_id: u16,
220    pub hl_count: u16,
221    pub block_count: u32,
222    pub flags: u32,
223    pub os_specific: u32,
224    pub data_ptr: [u32; 15],
225    pub gen_number: u32,
226    pub ext_attr_block: u32,
227    pub size_or_acl: u32,
228    pub fragment_address: u32,
229    pub os_specific2: [u8; 12],
230}
231
232impl INode {
233    pub fn set_file_type(&mut self, file_type: FileType) {
234        // The last 4 bits are used to store the filetype.
235        let mask = 0b0000_1111_1111_1111u16;
236        self.type_and_perm = file_type.bits() | (self.type_and_perm & mask);
237    }
238
239    pub fn set_size(&mut self, size: usize) {
240        self.size_lower = size as u32;
241        self.size_or_acl = (size >> 32) as u32;
242    }
243
244    pub fn size(&self) -> usize {
245        self.size_lower as usize | ((self.size_or_acl as usize) << 32)
246    }
247
248    pub fn set_permissions(&mut self, permissions: u16) {
249        let mut val = self.type_and_perm;
250        val.set_bits(..13, permissions);
251        self.type_and_perm = val;
252    }
253
254    pub fn file_type(&self) -> FileType {
255        let ty = self.type_and_perm >> 12;
256
257        match ty {
258            0x1 => FileType::Fifo,
259            0x2 => FileType::CharDev,
260            0x4 => FileType::Directory,
261            0x6 => FileType::BlockDev,
262            0x8 => FileType::File,
263            0xa => FileType::Symlink,
264            0xc => FileType::Socket,
265            _ => FileType::Unknown,
266        }
267    }
268
269    #[inline]
270    pub fn last_access(&self) -> Duration {
271        Duration::from_secs(self.last_access as u64)
272    }
273
274    #[inline]
275    pub fn last_modification(&self) -> Duration {
276        Duration::from_secs(self.last_modification as u64)
277    }
278
279    #[inline]
280    pub fn creation_time(&self) -> Duration {
281        Duration::from_secs(self.creation_time as u64)
282    }
283}
