341#[derive(Clone, Copy, Debug)]
342#[repr(C)]
343pub struct CompletionEntry {
344    pub result: u32, // Used by admin commands to return data.
345    pub reserved: u32,
346    pub sq_head: u16,    // Portion of the queue that may be reclaimed
347    pub sq_id: u16,      // Submission Queue that generated this entry.
348    pub command_id: u16, // Command ID of the command that was completed.
349    pub status: u16,     // Reason why the command failed, if it did.
