12const fn calculate_doorbell_offset(queue_id: u16, multiplier: usize, dstrd: usize) -> usize {
13    0x1000 + ((((queue_id as usize) * 2) + multiplier) * (4 << dstrd))
14}

36#[repr(C)]
37struct DoorBell(VolatileCell<u32>);
38
39unsafe impl Send for DoorBell {}
40unsafe impl Sync for DoorBell {}
41
42pub(super) struct Queue<'bell, T: QueueType> {
43    doorbell: &'bell DoorBell,
44    index: usize,
45    queue: Dma<[UnsafeCell<T::Type>]>,
46    phase: bool,
47}
48
49impl<'bell, T: QueueType> Queue<'bell, T> {
50    pub fn new(registers: &Registers, size: usize, queue_id: u16) -> Result<Self, Error> {
51        let dstrd = registers.capability.get_doorbell_stride() as usize;
52        let doorbell_offset = calculate_doorbell_offset(queue_id, T::DOORBELL_OFFSET, dstrd);
53
54        let base_addr = registers as *const _ as usize;
55        // SAFETY: The address is valid and aligned so, its safe to read from it.
56        let doorbell = unsafe { &*((base_addr + doorbell_offset) as *const DoorBell) };
57
58        Ok(Self {
59            doorbell,
60            queue: unsafe { Dma::new_zeroed_slice(size).assume_init() },
61            index: 0,
62            phase: true,
63        })
64    }
65
66    pub fn addr(&self) -> PhysAddr {
67        self.queue.addr()
68    }
69}
70
71impl Queue<'_, Completion> {
72    pub fn next_cmd_result(&mut self) -> Option<CompletionEntry> {
73        let queue_len = self.queue.len();
74        let cmd = &mut self.queue[self.index];
75
76        while (cmd.get_mut().status & 0x1) != self.phase as u16 {
77            core::hint::spin_loop();
78        }
79
80        let cmd = cmd.get_mut();
81        let status = cmd.status >> 1;
82
83        if status != 0 {
84            log::error!("nvme: command error {status:#x}");
85            return None;
86        }
87
88        self.index = (self.index + 1) % queue_len;
89
90        // invert the phase bit since we are in the next cycle/phase.
91        if self.index == 0 {
92            self.phase = !self.phase;
93        }
94
95        self.doorbell.0.set(self.index as u32);
96
97        Some(*cmd)
98    }
99}
100
101impl Queue<'_, Submission> {
102    pub fn submit_command(&mut self, command: Command) {
103        self.queue[self.index] = UnsafeCell::new(command);
104
105        self.index = (self.index + 1) % self.queue.len();
106        self.doorbell.0.set(self.index as u32); // ring ring!
107    }
108}

110static QUEUE_PAIR_ID: AtomicU16 = AtomicU16::new(0);
111
112pub(super) struct QueuePair<'a> {
113    id: u16,
114    size: usize,
115
116    cid: u16,
117
118    submission: Queue<'a, Submission>,
119    completion: Queue<'a, Completion>,
120}
121
122impl<'a> QueuePair<'a> {
123    pub fn new(registers: &Registers, size: usize) -> Result<Self, Error> {
124        let queue_id = QUEUE_PAIR_ID.fetch_add(1, Ordering::SeqCst);
125
126        Ok(Self {
127            size,
128            id: queue_id,
129
130            cid: 0,
131
132            submission: Queue::new(registers, size, queue_id)?,
133            completion: Queue::new(registers, size, queue_id)?,
134        })
135    }
136
137    pub fn submit_command<T: Into<Command>>(&mut self, command: T) {
138        let mut command = command.into();
139
140        unsafe {
141            // SAFETY: The offset of the `command_id` field is the same, regardless of the command
142            // type.
143            *ptr::addr_of_mut!(command).cast::<u16>().add(1) = self.cid;
144        }
145
146        self.cid += 1;
147
148        self.submission.submit_command(command);
149        self.completion.next_cmd_result().unwrap();
150    }
151
152    /// Returns the physical address of the submission queue.
153    pub fn submission_addr(&self) -> PhysAddr {
154        self.submission.addr()
155    }
156
157    /// Returns the physical address of the completion queue.
158    pub fn completion_addr(&self) -> PhysAddr {
159        self.completion.addr()
160    }
161
162    /// Returns the unique ID of this queue pair.
163    pub fn id(&self) -> u16 {
164        self.id
165    }
166
167    /// Returns the number of entries in the queue.
168    pub fn len(&self) -> usize {
169        self.size
170    }
171}
