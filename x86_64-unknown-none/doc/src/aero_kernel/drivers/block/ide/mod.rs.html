93pub struct IdeDevice {
94    ide_devs: [Option<Arc<IdeDrive>>; 4],
95    channels: [Option<Arc<IdeChannel>>; 2],
96}
97
98impl IdeDevice {
99    pub fn new() -> IdeDevice {
100        const EMPTY_DRIVE: Option<Arc<IdeDrive>> = None;
101        const EMPTY_CHANNELS: Option<Arc<IdeChannel>> = None;
102
103        IdeDevice {
104            ide_devs: [EMPTY_DRIVE; 4],
105            channels: [EMPTY_CHANNELS; 2],
106        }
107    }
108}
109
110impl IdeDevice {
111    fn launch(&mut self, header: &PciHeader) {
112        log::trace!("ide: starting ide");
113
114        let program_interface = header.program_interface();
115
116        if program_interface.contains(ProgramInterface::DMA_CAPABLE) {
117            log::warn!("ide: dma not supported");
118            return;
119        }
120
121        if header.get_header_type() != 0 {
122            log::debug!("ide: header type != 0");
123            return;
124        }
125
126        let bmid_1 = header.base_address4() & 0xFFFF_FFFC;
127        let bmid_2 = bmid_1 + 8;
128
129        let (io1, io2) = {
130            (
131                if header.base_address0() != 0 {
132                    header.base_address0() & 0xFFFF_FFFC
133                } else {
134                    0x1F0
135                },
136                if header.base_address1() != 0 {
137                    header.base_address1() & 0xFFFF_FFFC
138                } else {
139                    0x3F6
140                },
141            )
142        };
143
144        let (io3, io4) = {
145            (
146                if header.base_address2() != 0 {
147                    header.base_address2() & 0xFFFF_FFFC
148                } else {
149                    0x170
150                },
151                if header.base_address3() != 0 {
152                    header.base_address3() & 0xFFFF_FFFC
153                } else {
154                    0x376
155                },
156            )
157        };
158
159        let c1 = IdeChannel::new(io1 as u16, io2 as u16, bmid_1 as u16, 14);
160        let c2 = IdeChannel::new(io3 as u16, io4 as u16, bmid_2 as u16, 15);
161
162        let mut idx = 0;
163        for (ci, c) in [c1, c2].iter().enumerate() {
164            for &s in [false, true].iter() {
165                if c.detect(s) {
166                    self.ide_devs[idx] = Some(IdeDrive::new(s, c.clone()));
167                    idx += 1;
168
169                    if self.channels[ci].is_none() {
170                        self.channels[ci] = Some(c.clone());
171                    }
172                }
173            }
174        }
175
176        if idx > 0 {
177            header.enable_bus_mastering();
178
179            for channel in self.channels.iter_mut().filter_map(|a| a.as_mut()) {
180                channel.init();
181            }
182
183            for (i, drive) in self.ide_devs.iter().filter_map(|e| e.as_ref()).enumerate() {
184                let name = alloc::format!("blck{}", i);
185                let block_device = BlockDevice::new(name, drive.clone());
186
187                block::install_block_device(block_device).unwrap();
188            }
189        }
190    }
191}

193struct Ide {
194    device: Mutex<IdeDevice>,
195}
196
197impl PciDeviceHandle for Ide {
198    fn handles(&self, vendor_id: Vendor, device_id: DeviceType) -> bool {
199        matches!(
200            (vendor_id, device_id),
201            (Vendor::Intel, DeviceType::IdeController)
202        )
203    }
204
205    fn start(&self, header: &PciHeader, _offset_table: &mut OffsetPageTable) {
206        self.device.lock_irq().launch(header);
207    }
208}
209
210fn get_device() -> &'static Arc<Ide> {
211    DRIVER
212        .get()
213        .expect("ide: attempted to get the ide driver instance before it was initialized")
214}
215
216fn init() {
217    DRIVER.call_once(|| {
218        Arc::new(Ide {
219            device: Mutex::new(IdeDevice::new()),
220        })
221    });
222
223    register_device_driver(get_device().clone());
224}
