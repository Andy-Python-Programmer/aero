37/// Represents modset objects visible to userspace; this includes connectors,
38/// CRTCs, encoders, frambuffers and planes.
39#[downcastable]
40trait ModeObject: Send + Sync {
41    /// Returns the mode object's ID.
42    fn id(&self) -> u32;
43    fn object(&self) -> Arc<dyn ModeObject>;
44
45    // Conversion methods:
46
47    /// Converts this mode object into a connector.
48    fn as_connector(&self) -> Option<Arc<Connector>> {
49        self.object().downcast_arc::<Connector>()
50    }
51
52    /// Converts this mode object into an encoder.
53    fn as_encoder(&self) -> Option<Arc<Encoder>> {
54        self.object().downcast_arc::<Encoder>()
55    }
56
57    /// Converts this mode object into a CRTC.
58    fn as_crtc(&self) -> Option<Arc<Crtc>> {
59        self.object().downcast_arc::<Crtc>()
60    }
61
62    /// Converts this mode object into a framebuffer.
63    fn as_framebuffer(&self) -> Option<Arc<Framebuffer>> {
64        self.object().downcast_arc::<Framebuffer>()
65    }
66}

104// ## Notes:
105//
106// Plane: Image source
107//      - one or more framebuffers.
108//      - contains cropped/resized version of the framebuffer.
109//
110// CRTCs:
111//      - takes the plane and performs the composition.
112//      - has the display mode and params.
113//
114// Encoders:
115//      - takes the raw data from the CRTCs and converts it into a specific format.
116//
117// Connectors:
118//      - outputs the encoded data to an external display.
119//      - handles hotplug events.
120//      - reads EDIDs.
121//
122// Plane -> CRTCs -> Encoder -> Connector
123//                |============ LCD connector

274fn copy_field<T>(buffer: *mut T, buffer_size: &mut usize, value: &[T]) {
275    // do not overflow the user buffer.
276    let mut copy_len = value.len();
277
278    if *buffer_size > value.len() {
279        copy_len = *buffer_size;
280    }
281
282    // let userspace know exact length of driver value (which could be
283    // larger than the userspace-supplied buffer).
284    *buffer_size = value.len();
285
286    // finally, try filling in the user buffer.
287    if copy_len != 0 && !buffer.is_null() {
288        unsafe {
289            core::ptr::copy_nonoverlapping(value.as_ptr(), buffer, copy_len);
290        }
291    }
292}

308/// The direct rendering manager (DRM) exposes the GPUs through the device filesystem. Each
309/// GPU detected by the DRM is referred to as a DRM device and a device file (`/dev/dri/cardX`)
310/// is created to interface with it; where X is a sequential number.
311struct Drm {
312    sref: Weak<Self>,
313
314    inode: usize,
315    card_id: usize,
316    device: Arc<dyn DrmDevice>,
317
318    id_alloc: IdAllocator,
319    mapping_alloc: IdAllocator,
320    buffer_alloc: IdAllocator,
321
322    buffers: Mutex<HashMap<u32, BufferObject>>,
323    mode_objs: Mutex<HashMap<u32, Arc<dyn ModeObject>>>,
324
325    // All of the mode objects:
326    crtcs: Mutex<Vec<Arc<Crtc>>>,
327    encoders: Mutex<Vec<Arc<Encoder>>>,
328    connectors: Mutex<Vec<Arc<Connector>>>,
329    framebuffers: Mutex<Vec<Arc<Framebuffer>>>,
330}
331
332impl Drm {
333    pub fn new(device: Arc<dyn DrmDevice>) -> Arc<Self> {
334        Arc::new_cyclic(|sref| Self {
335            sref: sref.clone(),
336
337            inode: devfs::alloc_device_marker(),
338            card_id: DRM_CARD_ID.fetch_add(1, Ordering::SeqCst),
339            device,
340
341            buffer_alloc: IdAllocator::new(),
342            id_alloc: IdAllocator::new(),
343            mapping_alloc: IdAllocator::new(),
344
345            buffers: Mutex::new(HashMap::new()),
346            mode_objs: Mutex::new(HashMap::new()),
347
348            crtcs: Mutex::new(alloc::vec![]),
349            encoders: Mutex::new(alloc::vec![]),
350            connectors: Mutex::new(alloc::vec![]),
351            framebuffers: Mutex::new(alloc::vec![]),
352        })
353    }
354
355    /// Installs and initializes the CRTC identifier.
356    pub fn install_crtc(&self, crtc: Arc<Crtc>) {
357        self.crtcs.lock().push(crtc.clone());
358        self.install_object(crtc)
359    }
360
361    /// Installs and initializes the connector identifier.
362    pub fn install_connector(&self, connector: Arc<Connector>) {
363        self.connectors.lock().push(connector.clone());
364        self.install_object(connector)
365    }
366
367    /// Installs and initializes the encoder identifier.
368    pub fn install_encoder(&self, encoder: Arc<Encoder>) {
369        self.encoders.lock().push(encoder.clone());
370        self.install_object(encoder)
371    }
372
373    /// Installs and initializes the framebuffer identifier.
374    pub fn install_framebuffer(&self, fb: Arc<Framebuffer>) {
375        self.framebuffers.lock().push(fb.clone());
376        self.install_object(fb)
377    }
378
379    pub fn allocate_object_id(&self) -> u32 {
380        self.id_alloc.alloc() as _
381    }
382
383    fn install_object(&self, object: Arc<dyn ModeObject>) {
384        self.mode_objs.lock().insert(object.id(), object.clone());
385    }
386
387    fn find_object(&self, id: u32) -> Option<Arc<dyn ModeObject>> {
388        self.mode_objs.lock().get(&id).cloned()
389    }
390
391    fn find_handle(&self, handle: u32) -> Option<BufferObject> {
392        self.buffers.lock().get(&handle).cloned()
393    }
394
395    fn create_handle(&self, buffer: BufferObject) -> u32 {
396        let handle = self.buffer_alloc.alloc() as u32;
397
398        self.buffers.lock().insert(handle, buffer);
399        handle
400    }
401}

403impl INodeInterface for Drm {
404    // The DRM is accessed using IOCTLs on a device representing a graphics
405    // card.
406    fn ioctl(&self, command: usize, arg: usize) -> fs::Result<usize> {
407        match command {
408            DRM_IOCTL_VERSION => {
409                let mut struc = unsafe { UserRef::<DrmVersion>::new(VirtAddr::new(arg as u64)) };
410
411                let (major, minor, patch_level) = self.device.driver_version();
412                let (name, desc, date) = self.device.driver_info();
413
414                struc.version_major = major as _;
415                struc.version_minor = minor as _;
416                struc.version_patch_level = patch_level as _;
417
418                copy_field::<u8>(struc.name, &mut struc.name_len, name.as_bytes());
419                copy_field::<u8>(struc.desc, &mut struc.desc_len, desc.as_bytes());
420                copy_field::<u8>(struc.date, &mut struc.date_len, date.as_bytes());
421
422                Ok(0)
423            }
424
425            DRM_IOCTL_GET_CAP => {
426                let mut struc = unsafe { UserRef::<DrmGetCap>::new(VirtAddr::new(arg as u64)) };
427
428                // NOTE: The user is responsible for zeroing out the structure.
429                match struc.capability {
430                    DRM_CAP_DUMB_BUFFER => {
431                        if self.device.can_dumb_create() {
432                            struc.value = 1;
433                        }
434                    }
435
436                    cap => {
437                        log::warn!("drm: unknown capability (`{cap}`)");
438                        return Err(FileSystemError::NotSupported);
439                    }
440                }
441
442                Ok(0)
443            }
444
445            DRM_IOCTL_MODE_GETRESOURCES => {
446                let mut struc =
447                    unsafe { UserRef::<DrmModeCardRes>::new(VirtAddr::new(arg as u64)) };
448
449                /// Copies the mode object IDs into the user provided buffer. For safety, checkout
450                /// the [`copy_field`] function.
451                fn copy_mode_obj_id<T: ModeObject>(
452                    obj: &Mutex<Vec<Arc<T>>>,
453                    buffer: *mut u32,
454                    buffer_size: &mut u32,
455                ) {
456                    let objs = obj.lock();
457                    let mut count_objs = 0;
458
459                    copy_field::<u32>(
460                        buffer,
461                        &mut count_objs,
462                        objs.iter().map(|e| e.id()).collect::<Vec<_>>().as_slice(),
463                    );
464
465                    *buffer_size = count_objs as _;
466                }
467
468                let crtc_id_ptr = struc.crtc_id_ptr as *mut u32;
469                let encoder_id_ptr = struc.encoder_id_ptr as *mut u32;
470                let con_id_ptr = struc.connector_id_ptr as *mut u32;
471                let fb_id_ptr = struc.fb_id_ptr as *mut u32;
472
473                copy_mode_obj_id(&self.crtcs, crtc_id_ptr, &mut struc.count_crtcs);
474                copy_mode_obj_id(&self.encoders, encoder_id_ptr, &mut struc.count_encoders);
475                copy_mode_obj_id(&self.connectors, con_id_ptr, &mut struc.count_connectors);
476                copy_mode_obj_id(&self.framebuffers, fb_id_ptr, &mut struc.count_fbs);
477
478                let (xmin, ymin) = self.device.min_dim();
479
480                struc.min_width = xmin as _;
481                struc.min_height = ymin as _;
482
483                let (xmax, ymax) = self.device.max_dim();
484
485                struc.max_width = xmax as _;
486                struc.max_height = ymax as _;
487
488                Ok(0)
489            }
490
491            DRM_IOCTL_GET_CRTC => {
492                let struc = unsafe { UserRef::<DrmModeCrtc>::new(VirtAddr::new(arg as u64)) };
493                let _object = self.find_object(struc.crtc_id).unwrap().as_crtc().unwrap();
494
495                log::warn!("drm::get_crtc: is a stub!");
496                Ok(0)
497            }
498
499            DRM_IOCTL_SET_CRTC => {
500                let struc = unsafe { UserRef::<DrmModeCrtc>::new(VirtAddr::new(arg as u64)) };
501                let _object = self.find_object(struc.crtc_id).unwrap().as_crtc().unwrap();
502
503                let object = self
504                    .find_object(struc.fb_id)
505                    .unwrap()
506                    .as_framebuffer()
507                    .unwrap();
508
509                self.device.commit(&object.buffer_obj);
510                log::warn!("drm::set_crtc: is a stub!");
511
512                Ok(0)
513            }
514
515            DRM_IOCTL_GET_ENCODER => {
516                let mut struc =
517                    unsafe { UserRef::<DrmModeGetEncoder>::new(VirtAddr::new(arg as u64)) };
518
519                let object = self
520                    .find_object(struc.encoder_id)
521                    .unwrap()
522                    .as_encoder()
523                    .unwrap();
524
525                struc.crtc_id = object.current_crtc.id();
526
527                let mut crtc_mask = 0;
528                for crtc in object.possible_crtcs.iter() {
529                    crtc_mask = 1 << crtc.index;
530                }
531
532                struc.possible_crtcs = crtc_mask;
533
534                let mut clone_mask = 0;
535                for clone in object.possible_clones.iter() {
536                    clone_mask = 1 << clone.index;
537                }
538
539                struc.possible_clones = clone_mask;
540                struc.encoder_typ = 0; // todo: fill in the encoder typ.
541
542                Ok(0)
543            }
544
545            DRM_IOCTL_GET_CONNECTOR => {
546                let mut struc =
547                    unsafe { UserRef::<DrmModeGetConnector>::new(VirtAddr::new(arg as u64)) };
548
549                let object = self
550                    .find_object(struc.connector_id)
551                    .unwrap()
552                    .as_connector()
553                    .unwrap();
554
555                // Fill in the array containing all of the possible encoders and its length.
556                let encoder_ids_ptr = struc.encoders_ptr as *mut u32;
557                let mut encoder_count = 0;
558
559                copy_field::<u32>(
560                    encoder_ids_ptr,
561                    &mut encoder_count,
562                    object
563                        .possible_encoders
564                        .iter()
565                        .map(|e| e.id())
566                        .collect::<Vec<_>>()
567                        .as_slice(),
568                );
569
570                struc.count_encoders = encoder_count as _;
571
572                struc.encoder_id = object.current_encoder.id();
573                struc.connector_type = object.connector_typ;
574                struc.connector_type_id = 0; // todo
575                struc.connection = object.status as _;
576
577                // NOTE: The physical size will come from the EDID.
578                struc.mm_width = 0; // todo
579                struc.mm_height = 0; // todo
580                struc.subpixel = 0; // todo
581
582                // Fill in the array containing all of the possible modes and its length.
583                let modes_ptr = struc.modes_ptr as *mut DrmModeInfo;
584                let mut modes_count = 0;
585
586                copy_field::<DrmModeInfo>(modes_ptr, &mut modes_count, object.modes.as_slice());
587                struc.count_modes = modes_count as _;
588
589                Ok(0)
590            }
591
592            DRM_IOCTL_MODE_CREATE_DUMB => {
593                let mut struc =
594                    unsafe { UserRef::<DrmModeCreateDumb>::new(VirtAddr::new(arg as u64)) };
595
596                let (mut buffer, pitch) =
597                    self.device
598                        .dumb_create(struc.width, struc.height, struc.bpp);
599
600                assert!(buffer.size < (1usize << 32));
601
602                buffer.mapping = self.mapping_alloc.alloc() << 32;
603
604                struc.pitch = pitch;
605                struc.size = buffer.size as _;
606                struc.handle = self.create_handle(buffer);
607
608                Ok(0)
609            }
610
611            DRM_IOCTL_MODE_ADDFB => {
612                let mut struc = unsafe { UserRef::<DrmModeFbCmd>::new(VirtAddr::new(arg as u64)) };
613
614                let handle = self.find_handle(struc.handle).unwrap();
615                self.device
616                    .framebuffer_create(&handle, struc.width, struc.height, struc.pitch);
617
618                let fb = Framebuffer::new(self.allocate_object_id(), handle);
619                self.install_framebuffer(fb.clone());
620
621                struc.fb_id = fb.id();
622                Ok(0)
623            }
624
625            DRM_IOCTL_MODE_MAP_DUMB => {
626                let mut struc =
627                    unsafe { UserRef::<DrmModeMapDumb>::new(VirtAddr::new(arg as u64)) };
628
629                let handle = self.find_handle(struc.handle).unwrap();
630                struc.offset = handle.mapping as _;
631                Ok(0)
632            }
633
634            _ => {
635                // command[8..16] is the ASCII character supposedly unique to each driver.
636                if command.get_bits(8..16) == DRM_IOCTL_BASE {
637                    // command[0..8] is the function number.
638                    let function = command.get_bits(0..8);
639                    unimplemented!("drm: function (`{function:#x}`) not supported")
640                }
641
642                log::warn!("drm: unknown ioctl command (`{command}`)");
643                Err(FileSystemError::NotSupported)
644            }
645        }
646    }
647
648    fn mmap(
649        &self,
650        offset: usize,
651        _size: usize,
652        _flags: aero_syscall::MMapFlags,
653    ) -> fs::Result<PhysFrame> {
654        let buffers = self.buffers.lock();
655        let (_, handle) = buffers
656            .iter()
657            .find(|(_, h)| offset <= h.mapping + h.size && offset >= h.mapping)
658            .unwrap();
659
660        let index = (offset - handle.mapping) / Size4KiB::SIZE as usize;
661        Ok(handle.memory[index])
662    }

680    name: &str,
681    typ: u32,
682    clock: u32,
683    hdisplay: u16,
684    hsync_start: u16,
685    hsync_end: u16,
686    htotal: u16,
687    hskew: u16,
688    vdisplay: u16,
689    vsync_start: u16,
690    vsync_end: u16,
691    vtotal: u16,
692    vscan: u16,
693    flags: u32,
694) -> DrmModeInfo {
695    let mut this = DrmModeInfo {
696        clock,
697        hdisplay,
698        hsync_start,
699        hsync_end,
700        htotal,
701        hskew,
702        vdisplay,
703        vsync_start,
704        vsync_end,
705        vtotal,
706        vscan,
707        vrefresh: 0,
708        flags,
709        typ,
710        name: [0; DRM_DISPLAY_MODE_LEN],
711    };
712
713    for (i, byte) in name.as_bytes().iter().enumerate() {
714        this.name[i] = *byte as i8;
715    }
716
717    this
718}
