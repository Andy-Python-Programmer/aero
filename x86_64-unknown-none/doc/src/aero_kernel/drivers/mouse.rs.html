28bitflags::bitflags! {
29    /// Represents the flags currently set for the mouse.
30    #[derive(Default, Debug, Copy, Clone)]
31    pub struct MouseFlags: u8 {
32        const LEFT_BUTTON = 0b0000_0001;
33        const RIGHT_BUTTON = 0b0000_0010;
34        const MIDDLE_BUTTON = 0b0000_0100;
35        const ALWAYS_ONE = 0b0000_1000;
36        const X_SIGN = 0b0001_0000;
37        const Y_SIGN = 0b0010_0000;
38        const X_OVERFLOW = 0b0100_0000;
39        const Y_OVERFLOW = 0b1000_0000;
40    }
41}
42
43const DATA_PORT: u16 = 0x60;
44const CMD_PORT: u16 = 0x64;
45
46lazy_static::lazy_static! {
47    static ref MOUSE: Arc<Mouse> = Arc::new(Mouse::new());
48}
49
50static PACKETS: Mutex<Vec<Packet>> = Mutex::new(Vec::new());
51
52#[derive(Default, Debug, Copy, Clone)]
53#[repr(C)]
54struct Packet {
55    x: i16,
56    y: i16,
57
58    flags: MouseFlags,
59}
60
61struct Mouse {
62    packet: Mutex<(Packet, usize)>,
63    wq: WaitQueue,
64    marker: usize,
65}
66
67impl Mouse {
68    fn new() -> Mouse {
69        Self {
70            packet: Mutex::new((Packet::default(), 0)),
71            wq: WaitQueue::new(),
72            marker: devfs::alloc_device_marker(),
73        }
74    }
75
76    fn process_packet(&self, packet: u8) {
77        let sign_extend = |v: u8| ((v as u16) | 0xFF00) as i16;
78        let mut inner = self.packet.lock_irq();
79
80        match inner.1 {
81            0 => {
82                // The first packet contains the mouse flags.
83                let flags = MouseFlags::from_bits_truncate(packet);
84                let this = &mut inner.0;
85
86                this.flags = flags;
87            }
88
89            1 => {
90                let this = &mut inner.0;
91
92                // The second byte contains the "delta X".
93                if !this.flags.contains(MouseFlags::X_OVERFLOW) {
94                    if this.flags.contains(MouseFlags::X_SIGN) {
95                        this.x = sign_extend(packet);
96                    } else {
97                        this.x = packet as i16;
98                    }
99                }
100            }
101
102            2 => {
103                let this = &mut inner.0;
104
105                // The third packet contains the "delta Y".
106                if !this.flags.contains(MouseFlags::Y_OVERFLOW) {
107                    if this.flags.contains(MouseFlags::Y_SIGN) {
108                        this.y = sign_extend(packet);
109                    } else {
110                        this.y = packet as i16;
111                    }
112
113                    PACKETS.lock_irq().push(*this);
114                    self.wq.notify_all();
115                }
116            }
117
118            _ => unreachable!(),
119        }
120
121        inner.1 = (inner.1 + 1) % 3;
122    }
123}
124
125impl Device for Mouse {
126    fn device_marker(&self) -> usize {
127        self.marker
128    }
129
130    fn device_name(&self) -> String {
131        String::from("mouse0")
132    }
133
134    fn inode(&self) -> Arc<dyn INodeInterface> {
135        MOUSE.clone()
136    }
137}
138
139impl INodeInterface for Mouse {
140    fn read_at(&self, _offset: usize, buffer: &mut [u8]) -> fs::Result<usize> {
141        let size = core::mem::size_of::<Packet>();
142
143        if buffer.len() < size {
144            return Err(fs::FileSystemError::NotSupported);
145        }
146
147        let packet = PACKETS
148            .lock_irq()
149            .pop()
150            .ok_or(fs::FileSystemError::WouldBlock)?;
151
152        unsafe {
153            *(buffer.as_mut_ptr().cast::<Packet>()) = packet;
154        }
155
156        Ok(size)
157    }
158
159    fn poll(&self, table: Option<&mut PollTable>) -> fs::Result<PollFlags> {
160        if let Some(e) = table {
161            e.insert(&MOUSE.wq)
162        }
163
164        if !PACKETS.lock_irq().is_empty() {
165            Ok(PollFlags::IN)
166        } else {
167            Ok(PollFlags::empty())
168        }
169    }
170}
171
172fn irq_handler(_stack: &mut InterruptStack) {
173    let data = unsafe { io::inb(0x60) };
174    MOUSE.process_packet(data);
175}
176
177pub fn ps2_mouse_init() {
178    let irq_vector = interrupts::allocate_vector();
179    interrupts::register_handler(irq_vector, irq_handler);
180
181    unsafe {
182        io::outb(CMD_PORT, 0xd4);
183        io::outb(DATA_PORT, 0xF6);
184        while io::inb(DATA_PORT) != 0xfa {}
185        io::outb(CMD_PORT, 0xd4);
186        io::outb(DATA_PORT, 0xf4);
187        while io::inb(DATA_PORT) != 0xfa {}
188        io::outb(CMD_PORT, 0xd4);
189        io::outb(DATA_PORT, 0xE7);
190        while io::inb(DATA_PORT) != 0xfa {}
191    }
192
193    apic::io_apic_setup_legacy_irq(12, irq_vector, 1);
194
195    devfs::install_device(MOUSE.clone()).unwrap();
196    log::trace!("ps2: initialized mouse");
197}
