37struct Ps2KeyboardState {
38    special: bool,
39    released: bool,
40}
41
42impl Ps2KeyboardState {
43    #[inline]
44    const fn new() -> Self {
45        Self {
46            special: false,
47            released: false,
48        }
49    }
50
51    fn flush(&self) {
52        unsafe {
53            while io::inb(0x64) & 1 == 1 {
54                let _ = io::inb(0x60);
55            }
56        }
57    }

186struct KeyboardDevice {
187    marker: usize,
188    buffer: Mutex<Vec<u8>>,
189    sref: Weak<Self>,
190    wq: WaitQueue,
191}
192
193impl KeyboardDevice {
194    fn new() -> Arc<Self> {
195        Arc::new_cyclic(|this| Self {
196            marker: devfs::alloc_device_marker(),
197            buffer: Mutex::new(Vec::new()),
198            sref: this.clone(),
199            wq: WaitQueue::new(),
200        })
201    }
202}
203
204impl Device for KeyboardDevice {
205    fn device_marker(&self) -> usize {
206        self.marker
207    }
208
209    fn device_name(&self) -> String {
210        String::from("kbd0")
211    }
212
213    fn inode(&self) -> Arc<dyn INodeInterface> {
214        self.sref.upgrade().unwrap()
215    }
216}
217
218impl KeyboardListener for KeyboardDevice {
219    fn on_key(&self, keycode: KeyCode, released: bool) {
220        if released {
221            self.buffer.lock_irq().push(0x80 | keycode as u8);
222        } else {
223            self.buffer.lock_irq().push(keycode as u8);
224        }
225
226        self.wq.notify_all()
227    }
228}
229
230impl INodeInterface for KeyboardDevice {
231    fn read_at(&self, _offset: usize, buffer: &mut [u8]) -> fs::Result<usize> {
232        if self.buffer.lock_irq().is_empty() {
233            return Ok(0);
234        }
235
236        // TODO: block using wq
237        let mut sbuf = self.buffer.lock_irq();
238        let drainage = core::cmp::min(buffer.len(), sbuf.len());
239
240        for (i, byte) in sbuf.drain(..drainage).enumerate() {
241            buffer[i] = byte;
242        }
243
244        Ok(drainage)
245    }
246
247    fn poll(&self, table: Option<&mut fs::inode::PollTable>) -> fs::Result<PollFlags> {
248        if let Some(q) = table {
249            q.insert(&self.wq)
250        }
251
252        if !self.buffer.lock_irq().is_empty() {
253            Ok(PollFlags::IN)
254        } else {
255            Ok(PollFlags::empty())
256        }
257    }
258}

260/// This function is responsible for initializing PS2 keyboard driver.
261pub fn ps2_keyboard_init() {
262    let lock = PS2_KEYBOARD_STATE.lock_irq();
263
264    unsafe {
265        io::outb(0x60, 0xF5); // command: disable scanning
266
267        if io::inb(0x60) != 0xFA {
268            log::warn!("ps2: disable scanning failed, no ACK");
269        }
270
271        io::outb(0x60, 0xF4); // command: enable reporting
272        lock.flush();
273
274        if io::inb(0x60) != 0xFA {
275            log::warn!("ps2: failed to enable error reporting, no ACK");
276        }
277
278        io::outb(0x64, 0x20); // command: read
279        lock.flush();
280
281        let mut config = ConfigFlags::from_bits_truncate(io::inb(0x60));
282
283        config.remove(ConfigFlags::FIRST_DISABLED | ConfigFlags::SECOND_DISABLED);
284        config.remove(ConfigFlags::FIRST_TRANSLATE); // Use scancode set 2
285        config.insert(ConfigFlags::FIRST_INTERRUPT | ConfigFlags::SECOND_INTERRUPT);
286
287        io::outb(0x64, 0x60); // command: write config
288        io::outb(0x60, config.bits());
289
290        lock.flush();
291    }
292
293    let keyboard_vector = interrupts::allocate_vector();
294    interrupts::register_handler(keyboard_vector, keyboard_irq_handler);
295
296    apic::io_apic_setup_legacy_irq(1, keyboard_vector, 1);
297
298    super::mouse::ps2_mouse_init();
299
300    // TODO: Move this into /dev/input instead
301    // TODO: Add support for multiple keyboards
302    register_keyboard_listener(KEYBOARD.clone());
303    devfs::install_device(KEYBOARD.clone()).expect("failed to install keyboard device");
304}

441            }
442        }
443    }
