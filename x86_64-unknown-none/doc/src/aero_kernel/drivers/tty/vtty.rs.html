156struct StdinBuffer {
157    back_buffer: Vec<u8>,
158    front_buffer: Vec<u8>, // more like a queue
159
160    cursor: usize,
161}
162
163impl StdinBuffer {
164    fn new() -> Self {
165        Self {
166            back_buffer: Vec::new(),
167            front_buffer: Vec::new(),
168
169            cursor: 0,
170        }
171    }
172
173    fn swap_buffer(&mut self) {
174        for c in self.back_buffer.drain(..) {
175            self.front_buffer.push(c);
176        }
177    }
178
179    fn is_complete(&self) -> bool {
180        !self.back_buffer.is_empty() || !self.front_buffer.is_empty()
181    }
182
183    fn advance_cursor(&mut self) {
184        self.cursor += 1;
185    }
186}

230impl INodeInterface for Tty {
231    fn open(
232        &self,
233        _handle: Arc<fs::file_table::FileHandle>,
234    ) -> fs::Result<Option<fs::cache::DirCacheItem>> {
235        let connected = self.connected.fetch_add(1, Ordering::SeqCst);
236        if connected == 0 {
237            crate::drivers::keyboard::register_keyboard_listener(TTY.clone());
238
239            let current_task = scheduler::get_scheduler().current_task();
240            current_task.attach(self.sref.upgrade().unwrap());
241        }
242
243        Ok(None)
244    }
245
246    fn close(&self, _flags: aero_syscall::OpenFlags) {
247        self.connected.fetch_sub(1, Ordering::SeqCst);
248    }
249
250    fn read_at(&self, flags: OpenFlags, _offset: usize, buffer: &mut [u8]) -> fs::Result<usize> {
251        let mut stdin = self.block_queue.wait(
252            WaitQueueFlags::from(flags) | WaitQueueFlags::DISABLE_IRQ,
253            &self.stdin,
254            |future| future.is_complete(),
255        )?;
256
257        // record the back buffer size before swapping
258        stdin.swap_buffer();
259        let back_len = stdin.front_buffer.len();
260
261        if buffer.len() > stdin.front_buffer.len() {
262            for (i, c) in stdin.front_buffer.drain(..).enumerate() {
263                buffer[i] = c;
264            }
265        } else {
266            for (i, c) in stdin.front_buffer.drain(..buffer.len()).enumerate() {
267                buffer[i] = c;
268            }
269        }
270
271        Ok(core::cmp::min(buffer.len(), back_len))
272    }
273
274    fn write_at(&self, _offset: usize, buffer: &[u8]) -> fs::Result<usize> {
275        let string = core::str::from_utf8(buffer).map_err(|_| FileSystemError::NotSupported)?;
276
277        crate::rendy::print!("{}", string);
278
279        log::debug!("TTY::write_at(): {}", unsafe {
280            core::str::from_utf8_unchecked(buffer)
281        });
282
283        Ok(buffer.len())
284    }
285
286    fn poll(&self, table: Option<&mut PollTable>) -> fs::Result<PollFlags> {
287        if let Some(e) = table {
288            e.insert(&self.block_queue)
289        }
290        let mut events = PollFlags::empty();
291
292        if self.stdin.lock_irq().is_complete() {
293            events.insert(PollFlags::IN);
294        }
295
296        Ok(events)
297    }
298
299    fn ioctl(&self, command: usize, arg: usize) -> fs::Result<usize> {
300        match command {
301            aero_syscall::TIOCGWINSZ => {
302                let winsize = VirtAddr::new(arg as u64);
303                let winsize = unsafe { &mut *(winsize.as_mut_ptr::<aero_syscall::WinSize>()) };
304
305                let (rows, cols) = rendy::get_rows_cols();
306
307                winsize.ws_row = rows as u16;
308                winsize.ws_col = cols as u16;
309
310                let (xpixel, ypixel) = rendy::get_resolution();
311
312                winsize.ws_xpixel = xpixel as u16;
313                winsize.ws_ypixel = ypixel as u16;
314
315                Ok(0x00)
316            }
317
318            aero_syscall::TCGETS => {
319                let termios = VirtAddr::new(arg as u64);
320                let termios = unsafe { &mut *(termios.as_mut_ptr::<aero_syscall::Termios>()) };
321
322                let lock = TERMIOS.lock_irq();
323                let this = &*lock;
324
325                *termios = this.clone();
326                Ok(0x00)
327            }
328
329            aero_syscall::TCSETSF => {
330                // Allow the output buffer to drain, discard pending input.
331                let mut stdin = self.stdin.lock_irq();
332                stdin.back_buffer.clear();
333                stdin.cursor = 0;
334                core::mem::drop(stdin);
335
336                let termios = VirtAddr::new(arg as u64);
337                let termios = unsafe { &*(termios.as_mut_ptr::<aero_syscall::Termios>()) };
338
339                let mut lock = TERMIOS.lock_irq();
340                let this = &mut *lock;
341
342                *this = termios.clone();
343                Ok(0x00)
344            }
345
346            _ => Err(fs::FileSystemError::NotSupported),
347        }
348    }
349}

375#[cfg(target_arch = "x86_64")]
376impl KeyboardListener for Tty {
377    fn on_key(&self, key: KeyCode, released: bool) {
378        let mut state = self.state.lock();
379        let termios = TERMIOS.lock_irq();
380
381        let push_str = |k: &str| {
382            // TODO: decckm
383            let mut stdin = self.stdin.lock_irq();
384
385            for each in k.bytes() {
386                stdin.back_buffer.push(each);
387            }
388        };
389
390        let lchar = || {
391            let mut shift = state.lshift || state.rshift;
392            let ctrl = state.lctrl || state.rctrl;
393
394            if state.caps {
395                shift = !shift;
396            }
397
398            let map = match (shift, ctrl, state.lalt, state.altgr) {
399                (false, false, false, false) => PLAIN_MAP,
400                (true, false, false, false) => SHIFT_MAP,
401                (false, true, false, false) => CTRL_MAP,
402                (false, false, true, false) => ALT_MAP,
403                (false, false, false, true) => ALTGR_MAP,
404                (true, true, false, false) => SHIFT_CTRL_MAP,
405                (false, true, true, false) => CTRL_ALT_MAP,
406                _ => PLAIN_MAP,
407            };
408
409            let character =
410                unsafe { core::char::from_u32_unchecked((map[key as usize] & 0xff) as _) };
411
412            // Check if the character is actually printable printable.
413            if !(0x20..0x7e).contains(&(character as u32)) {
414                return;
415            }
416
417            {
418                let mut stdin = self.stdin.lock_irq();
419
420                stdin.back_buffer.push(character as u8);
421                stdin.advance_cursor();
422            }
423
424            if termios.c_lflag.contains(aero_syscall::TermiosLFlag::ECHO) {
425                rendy::print!("{}", character);
426            }
427        };
428
429        let backspace = || {
430            if termios.c_lflag.contains(aero_syscall::TermiosLFlag::ICANON) {
431                let mut stdin = self.stdin.lock_irq();
432
433                if stdin.back_buffer.pop().is_some()
434                    && termios.c_lflag.contains(aero_syscall::TermiosLFlag::ECHO)
435                {
436                    rendy::backspace();
437                    stdin.cursor -= 1;
438                }
439            } else {
440                push_str("\x08");
441            }
442        };
443
444        if !termios.c_lflag.contains(aero_syscall::TermiosLFlag::ICANON) && !released {
445            match key {
446                KeyCode::KEY_BACKSPACE if !released => backspace(),
447                KeyCode::KEY_CAPSLOCK if !released => state.caps = !state.caps,
448
449                KeyCode::KEY_LEFTSHIFT => state.lshift = !released,
450                KeyCode::KEY_RIGHTSHIFT => state.rshift = !released,
451
452                KeyCode::KEY_LEFTCTRL => state.lctrl = !released,
453                KeyCode::KEY_RIGHTCTRL => state.rctrl = !released,
454
455                KeyCode::KEY_LEFTALT => state.lalt = !released,
456                KeyCode::KEY_RIGHTALT => state.altgr = !released,
457                KeyCode::KEY_ENTER => push_str("\n"),
458
459                KeyCode::KEY_UP => push_str("\x1b[A"),
460                KeyCode::KEY_LEFT => push_str("\x1b[D"),
461                KeyCode::KEY_DOWN => push_str("\x1b[B"),
462                KeyCode::KEY_RIGHT => push_str("\x1b[C"),
463
464                _ if !released => lchar(),
465                _ => {}
466            }
467
468            self.stdin.lock_irq().cursor = 0;
469            self.block_queue.notify_all();
470            return;
471        }
472
473        match key {
474            KeyCode::KEY_CAPSLOCK if !released => state.caps = !state.caps,
475            KeyCode::KEY_ENTER | KeyCode::KEY_KPENTER if !released => {
476                let mut stdin = self.stdin.lock_irq();
477
478                stdin.back_buffer.push(b'\n');
479                stdin.cursor = 0;
480
481                if termios.c_lflag.contains(aero_syscall::TermiosLFlag::ECHO) {
482                    rendy::print!("\n");
483                }
484
485                self.block_queue.notify_all();
486            }
487
488            KeyCode::KEY_BACKSPACE if !released => backspace(),
489
490            KeyCode::KEY_LEFTSHIFT => state.lshift = !released,
491            KeyCode::KEY_RIGHTSHIFT => state.rshift = !released,
492
493            KeyCode::KEY_LEFTCTRL => state.lctrl = !released,
494            KeyCode::KEY_RIGHTCTRL => state.rctrl = !released,
495
496            KeyCode::KEY_LEFTALT => state.lalt = !released,
497            KeyCode::KEY_RIGHTALT => state.altgr = !released,
498
499            KeyCode::KEY_LEFT if !released => {
500                let mut stdin = self.stdin.lock_irq();
501
502                // We are at the start of the input so, we cannot shift
503                // the cursor to the left anymore.
504                if stdin.cursor == 0 {
505                    return;
506                }
507
508                let (x, y) = rendy::get_cursor_position();
509                rendy::set_cursor_position(x - 1, y);
510
511                stdin.cursor -= 1;
512            }
513
514            KeyCode::KEY_RIGHT if !released => {
515                let mut stdin = self.stdin.lock_irq();
516
517                // We are at the end of the input so, we cannot shift
518                // the cursor to the right anymore.
519                if stdin.cursor == stdin.back_buffer.len() {
520                    return;
521                }
522
523                let (x, y) = rendy::get_cursor_position();
524                rendy::set_cursor_position(x + 1, y);
525
526                stdin.advance_cursor();
527            }
528
529            _ if !released => lchar(),
530
531            _ => {}
532        }
533    }
534}

537    devfs::install_device(TTY.clone())?;
538    Ok(())
