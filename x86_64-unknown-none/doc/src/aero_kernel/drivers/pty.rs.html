51#[derive(Debug, Ioctl)]
52pub enum TermiosCmd {
53    /// Get window size.
54    #[command(libc::TIOCGWINSZ)]
55    GetWinSize(UserRef<WinSize>),
56
57    /// Set window size.
58    #[command(libc::TIOCSWINSZ)]
59    SetWinSize(UserRef<WinSize>),
60
61    /// Get the current serial port settings.
62    ///
63    /// Equivalent to `tcgetattr(fd, argp)`.
64    #[command(libc::TCGETS)]
65    TcGets(UserRef<Termios>),
66
67    /// Allow the output buffer to drain, discard pending input, and set the current serial
68    /// port settings.
69    ///
70    /// Equivalent to `tcsetattr(fd, TCSAFLUSH, argp)`.
71    #[command(libc::TCSETSF)]
72    TcSetsf(UserRef<Termios>),
73
74    /// Allow the output buffer to drain, and set the current serial port settings.
75    ///
76    /// Equivalent to `tcsetattr(fd, TCSADRAIN, argp)`.
77    #[command(libc::TCSETSW)]
78    TcSetsw(UserRef<Termios>),
79
80    /// Make the given terminal the controlling terminal of the calling process. The calling
81    /// process must be a session leader and not have a controlling terminal already. For this
82    /// case, arg should be specified as zero.
83    ///
84    /// If this terminal is already the controlling terminal of a different session group, then the
85    /// ioctl fails with EPERM, unless the caller has the CAP_SYS_ADMIN capability and arg equals
86    /// 1, in which case the terminal is stolen, and all processes that had it as controlling
87    /// terminal lose it.
88    // FIXME: argument usize
89    #[command(libc::TIOCSCTTY)]
90    SetCtrlTerm,
91
92    /// Get the process group ID of the foreground process group on this terminal.
93    ///
94    /// When successful, equivalent to `*argp = tcgetpgrp(fd)`.
95    // FIXME: argument usize
96    #[command(libc::TIOCGPGRP)]
97    GetProcGroupId,
98}

100struct Master {
101    id: u32,
102    wq: WaitQueue,
103    window_size: Mutex<WinSize>,
104    buffer: Mutex<Vec<u8>>,
105    discipline: LineDiscipline,
106}
107
108impl Master {
109    pub fn new() -> Self {
110        Self {
111            id: PTY_ID.fetch_add(1, Ordering::SeqCst),
112            wq: WaitQueue::new(),
113            window_size: Mutex::new(WinSize::default()),
114            buffer: Mutex::new(Vec::new()),
115            discipline: LineDiscipline::new(),
116        }
117    }
118
119    #[inline]
120    fn set_window_size(&self, size: WinSize) {
121        *self.window_size.lock_irq() = size;
122    }
123
124    #[inline]
125    fn get_window_size(&self) -> WinSize {
126        *self.window_size.lock_irq()
127    }
128}
129
130impl INodeInterface for Master {
131    fn read_at(&self, _offset: usize, buffer: &mut [u8]) -> fs::Result<usize> {
132        let mut pty_buffer = self.buffer.lock_irq();
133
134        if pty_buffer.is_empty() {
135            return Err(FileSystemError::WouldBlock);
136        }
137
138        let size = core::cmp::min(pty_buffer.len(), buffer.len());
139        buffer[..size].copy_from_slice(&pty_buffer.drain(..size).collect::<Vec<_>>());
140        Ok(size)
141    }
142
143    fn write_at(&self, _offset: usize, buffer: &[u8]) -> fs::Result<usize> {
144        self.discipline.write(buffer, |ctrl| match ctrl {
145            LineControl::Echo(c) => self.buffer.lock_irq().push(c),
146        });
147        self.wq.notify_all();
148        Ok(buffer.len())
149    }
150
151    fn poll(&self, table: Option<&mut fs::inode::PollTable>) -> fs::Result<fs::inode::PollFlags> {
152        if let Some(e) = table {
153            e.insert(&self.wq)
154        }
155        let mut flags = fs::inode::PollFlags::OUT;
156
157        if !self.buffer.lock_irq().is_empty() {
158            flags |= fs::inode::PollFlags::IN;
159        }
160
161        Ok(flags)
162    }
163
164    fn ioctl(&self, command: usize, arg: usize) -> fs::Result<usize> {
165        match command {
166            TIOCGPTN => {
167                let id = VirtAddr::new(arg as u64).read_mut::<u32>()?;
168                *id = self.id;
169            }
170
171            aero_syscall::TIOCSWINSZ => {
172                let winsize = VirtAddr::new(arg as u64).read_mut::<WinSize>()?;
173                *self.window_size.lock_irq() = *winsize;
174            }
175
176            _ => {
177                panic!("ptmx: unknown ioctl (command={command:#x})")
178            }
179        }
180
181        Ok(0)
182    }
183}

185impl TerminalDevice for Slave {
186    fn attach(&self, task: Arc<Task>) {
187        assert!(task.is_session_leader());
188        self.master.discipline.set_foreground(&task);
189    }
190
191    fn detach(&self, task: Arc<Task>) {
192        use aero_syscall::signal::SIGINT;
193        use aero_syscall::VINTR;
194
195        if !self.master.discipline.termios.lock().is_cooked() {
196            return;
197        }
198
199        if let ExitStatus::Signal(signo) = task.exit_status() {
200            let mut buffer = self.master.buffer.lock_irq();
201            let termios = self.master.discipline.termios.lock();
202
203            // converts `X` into `^X` and pushes the result into the master PTY buffer.
204            let mut ctrl = |c| {
205                buffer.extend_from_slice(&[b'^', c + 0x40]);
206            };
207
208            if *signo == SIGINT {
209                ctrl(termios.c_cc[VINTR])
210            }
211        }
212    }
213}

215struct Slave {
216    sref: Weak<Self>,
217    master: Arc<Master>,
218}
219
220impl Slave {
221    pub fn new(master: Arc<Master>) -> Arc<Self> {
222        Arc::new_cyclic(|sref| Self {
223            sref: sref.clone(),
224            master,
225        })
226    }
227
228    fn sref(&self) -> Arc<Self> {
229        self.sref.upgrade().unwrap()
230    }
231}
232
233impl INodeInterface for Slave {
234    fn metadata(&self) -> fs::Result<fs::inode::Metadata> {
235        Ok(fs::inode::Metadata::with_file_type(FileType::Device))
236    }
237
238    fn stat(&self) -> fs::Result<aero_syscall::Stat> {
239        Ok(aero_syscall::Stat::default())
240    }
241
242    fn ioctl(&self, command: usize, arg: usize) -> fs::Result<usize> {
243        match TermiosCmd::from_command_arg(command, arg) {
244            TermiosCmd::GetWinSize(mut size) => *size = self.master.get_window_size(),
245            TermiosCmd::SetWinSize(size) => self.master.set_window_size(*size),
246            TermiosCmd::TcGets(mut termios) => *termios = self.master.discipline.termios(),
247            TermiosCmd::TcSetsf(termios) => self.master.discipline.set_termios(termios.clone()),
248            TermiosCmd::TcSetsw(termios) => {
249                // TODO: Allow the output buffer to drain and then set the current serial port
250                // settings.
251                self.master.discipline.set_termios(termios.clone())
252            }
253
254            TermiosCmd::SetCtrlTerm => {
255                let current_task = scheduler::get_scheduler().current_task();
256                assert!(current_task.is_session_leader());
257
258                current_task.attach(self.sref());
259            }
260
261            // FIXME: the following ioctls are not implemented.
262            TermiosCmd::GetProcGroupId => return Err(FileSystemError::NotSupported),
263        }
264
265        Ok(0)
266    }
267
268    fn poll(&self, table: Option<&mut fs::inode::PollTable>) -> fs::Result<PollFlags> {
269        if let Some(table) = table {
270            table.insert(&self.master.wq);
271            table.insert(self.master.discipline.wait_queue());
272        }
273
274        let mut flags = PollFlags::OUT;
275
276        if !self.master.discipline.is_empty() {
277            flags |= PollFlags::IN;
278        }
279
280        Ok(flags)
281    }
282
283    fn read_at(&self, _offset: usize, buffer: &mut [u8]) -> fs::Result<usize> {
284        Ok(self.master.discipline.read(buffer)?)
285    }
286
287    fn write_at(&self, _offset: usize, buffer: &[u8]) -> fs::Result<usize> {
288        if self
289            .master
290            .discipline
291            .termios()
292            .c_oflag
293            .contains(aero_syscall::TermiosOFlag::ONLCR)
294        {
295            let mut master = self.master.buffer.lock_irq();
296
297            for b in buffer.iter() {
298                if *b == b'\n' {
299                    // ONLCR: Convert NL to CR + NL
300                    master.extend_from_slice(b"\r\n");
301                    continue;
302                }
303
304                master.push(*b);
305            }
306        } else {
307            let mut pty_buffer = self.master.buffer.lock_irq();
308            pty_buffer.extend_from_slice(buffer);
309        }
310
311        self.master.wq.notify_all();
312        Ok(buffer.len())
313    }
314}

342impl INodeInterface for Ptmx {
343    fn open(&self, _handle: Arc<fs::file_table::FileHandle>) -> fs::Result<Option<DirCacheItem>> {
344        let master = Arc::new(Master::new());
345        let slave = Slave::new(master.clone());
346        let inode = DirEntry::from_inode(master, String::from("<pty>"));

348        PTS_FS.get().unwrap().insert_slave(slave);
349        Ok(Some(inode))
350    }

353#[derive(Default)]
354struct PtsINode {
355    inode: Once<INodeCacheItem>,
356    fs: Once<Weak<PtsFs>>,
357    slaves: RwLock<BTreeMap<u32, INodeCacheItem>>,
358}
359
360impl INodeInterface for PtsINode {
361    fn metadata(&self) -> fs::Result<fs::inode::Metadata> {
362        Ok(fs::inode::Metadata {
363            id: 0,
364            file_type: FileType::Directory,
365            children_len: self.slaves.read().len(),
366            size: 0,
367        })
368    }
369
370    fn stat(&self) -> fs::Result<aero_syscall::Stat> {
371        Ok(aero_syscall::Stat::default())
372    }
373
374    fn dirent(&self, parent: DirCacheItem, index: usize) -> fs::Result<Option<DirCacheItem>> {
375        Ok(match index {
376            0x00 => Some(DirEntry::new(
377                parent,
378                self.inode.get().unwrap().clone(),
379                String::from("."),
380            )),
381
382            0x01 => Some(DirEntry::new(
383                parent,
384                self.inode.get().unwrap().clone(),
385                String::from(".."),
386            )),
387
388            _ => {
389                let a = self
390                    .slaves
391                    .read()
392                    .iter()
393                    .nth(index - 2)
394                    .map(|(id, inode)| DirEntry::new(parent, inode.clone(), id.to_string()));
395                log::debug!("{}", a.is_some());
396                a
397            }
398        })
399    }
400
401    fn lookup(&self, dir: DirCacheItem, name: &str) -> fs::Result<DirCacheItem> {
402        let id = name.parse::<u32>().unwrap();
403        let slaves = self.slaves.read();
404
405        let (_, inode) = slaves
406            .iter()
407            .find(|(&e, _)| e == id)
408            .ok_or(FileSystemError::EntryNotFound)?;
409
410        Ok(DirEntry::new(dir, inode.clone(), String::from(name)))
411    }
412
413    fn weak_filesystem(&self) -> Option<Weak<dyn FileSystem>> {
414        Some(self.fs.get()?.clone())
415    }
416}
417
418struct PtsFs {
419    root_dir: DirCacheItem,
420}
421
422impl PtsFs {
423    fn new() -> Arc<Self> {
424        let icache = cache::icache();
425        let root_inode = icache.make_item_no_cache(CachedINode::new(Arc::new(PtsINode::default())));
426
427        let root_dir = DirEntry::new_root(root_inode.clone(), String::from("/"));
428        let pts_root = root_dir.inode().downcast_arc::<PtsINode>().unwrap();
429
430        let this = Arc::new(Self { root_dir });
431
432        // Initialize the PTS root inode.
433        pts_root.fs.call_once(|| Arc::downgrade(&this));
434        pts_root.inode.call_once(|| root_inode.clone());
435
436        this
437    }
438
439    fn insert_slave(&self, slave: Arc<Slave>) {
440        let icache = cache::icache();
441
442        let pts_root = self.root_dir.inode().downcast_arc::<PtsINode>().unwrap();
443        pts_root.slaves.write().insert(
444            slave.master.id,
445            icache.make_item_no_cache(CachedINode::new(slave)),
446        );
447    }
448}
449
450impl fs::FileSystem for PtsFs {
451    fn root_dir(&self) -> DirCacheItem {
452        self.root_dir.clone()
453    }
454}
455
456fn pty_init() {
457    devfs::install_device(PTMX.clone()).unwrap();
458
459    let fs = PTS_FS.call_once(PtsFs::new);
460
461    let root = DEV_FILESYSTEM.root_dir().inode();
462    root.mkdir("pts").unwrap();
463
464    let pts_dir = fs::lookup_path(Path::new("/dev/pts")).unwrap();
465    MOUNT_MANAGER.mount(pts_dir, fs.clone()).unwrap();
466}
