32enum Status {
33    Resolved,
34    Pending(Vec<RawPacket>),
35}
36
37struct Entry {
38    mac: MacAddr,
39    status: Status,
40}
41
42impl Entry {
43    fn new(mac: MacAddr, status: Status) -> Self {
44        Self { mac, status }
45    }
46}
47
48struct Cache(BTreeMap<Ipv4Addr, Entry>);
49
50impl Cache {
51    fn new() -> Self {
52        Self(BTreeMap::new())
53    }
54
55    fn insert(&mut self, ip: Ipv4Addr, mac: MacAddr) {
56        if let Some(entry) = self.0.get_mut(&ip) {
57            let status = core::mem::replace(&mut entry.status, Status::Resolved);
58
59            if let Status::Pending(queue) = status {
60                entry.mac = mac;
61                entry.status = Status::Resolved;
62
63                for mut packet in queue {
64                    log::trace!("[ ARP ] (!!) Sending queued packed to {ip:?} {mac:?}");
65
66                    // FIXME: make this cleaner
67                    let eth = unsafe { &mut *packet.as_mut_ptr().cast::<Eth>() };
68                    eth.dest_mac = mac;
69
70                    super::default_device().send(packet);
71                }
72            }
73        } else {
74            self.0.insert(ip, Entry::new(mac, Status::Resolved));
75        }
76    }
77
78    fn request(&mut self, ip: Ipv4Addr, packet: RawPacket) {
79        assert!(ip != Ipv4Addr::LOOPBACK);
80
81        if self.0.get_mut(&ip).is_some() {
82            todo!()
83        } else {
84            let queue = alloc::vec![packet];
85            let entry = Entry::new(MacAddr::NULL, Status::Pending(queue));
86
87            self.0.insert(ip, entry);
88        }
89    }
90
91    fn get(&self, ip: Ipv4Addr) -> Option<MacAddr> {
92        if let Some(entry) = self.0.get(&ip) {
93            return Some(entry.mac);
94        }
95
96        None
97    }
98}
99
100static CACHE: Once<RwLock<Cache>> = Once::new();
101
102pub fn get(ip: Ipv4Addr) -> Option<MacAddr> {
103    CACHE
104        .get()
105        .as_ref()
106        .expect("arp: cache not initialized")
107        .read()
108        .get(ip)
109}
110
111pub fn init() {
112    CACHE.call_once(|| {
113        let mut cache = Cache::new();
114        cache.insert(Ipv4Addr::BROADCAST, MacAddr::BROADCAST);
115
116        RwLock::new(cache)
117    });
118}

175pub fn do_recv(arp: &Arp) {
176    CACHE
177        .get()
178        .as_ref()
179        .expect("arp: cache not initialized")
180        .write()
181        .insert(arp.src_ip(), arp.src_mac());
182
183    let device = default_device();
184
185    if arp.opcode() == ArpOpcode::Request && arp.dest_ip() == device.ip() {
186        let addr = ArpAddress::new(arp.src_mac(), arp.src_ip());
187        let reply_arp = make_arp(ArpOpcode::Reply, addr);
188
189        reply_arp.send();
190    }
191}
192
193pub fn request_ip(target: Ipv4Addr, to: RawPacket) {
194    let arp = make_arp(ArpOpcode::Request, ArpAddress::new(MacAddr::NULL, target));
195
196    log::debug!("[ ARP ] (!!) Sending request for {target:?}");
197
198    CACHE
199        .get()
200        .as_ref()
201        .expect("arp: cache not initialized")
202        .write()
203        .request(target, to);
204
205    arp.send();
206}
207
208fn make_arp(opcode: ArpOpcode, dest_addr: ArpAddress) -> Arp {
209    let device = default_device();
210    let src_addr = ArpAddress::new(device.mac(), device.ip());
211
212    Arp::new(
213        ArpHardwareType::Ethernet,
214        EthType::Ip,
215        src_addr,
216        dest_addr,
217        opcode,
218    )
219}
