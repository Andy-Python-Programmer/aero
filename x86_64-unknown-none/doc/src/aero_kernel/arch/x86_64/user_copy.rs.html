29/// Copy to/from a block of data from user space. Returns whether the copy was successful.
30///
31/// # Safety
32/// The caller must ensure that:
33/// * If copying to userspace, the `dest` pointer must be within the userland address space.
34/// * If copying from userspace, the `dest` pointer must be valid for `size` bytes.
35///
36/// ## Concurrent Accesses
37/// Concurrent access, *including potential data races with userspace memory*, are permitted since
38/// other userspace threads or processes may modify the memory concurrently. This behavior is
39/// similar to how [`std::io`] permits data races with file contents on disk.
40///
41/// [`std::io`]: https://doc.rust-lang.org/std/io/index.html
42#[naked]
43unsafe extern "C" fn copy_to_from_user(
44    dest: *mut u8,
45    src: *const u8,
46    size: usize,
47
48    fault_resume: *const u8,
49) -> bool {
50    // Registers used:
51    //
52    // %rdi = argument 1, `dest`
53    // %rsi = argument 2, `src`
54    // %rdx = argument 3, `size`
55    // %rcx = argument 4, `fault_resume` (copied to %r10)
56    naked_asm!(
57        // Copy `fault_resume` out of %rcx because it will be utilized by `rep movsb` latter.
58        "mov r10, rcx",
59        // Setup the page fault resume.
60        "lea rax, 1f",
61        "mov [r10], rax",
62        // XXX: From this point until the fault return is reset, no function calls or stack
63        // manipulations should be performed. We must ensure the ability to restore all callee
64        // registers without any knowledge of the exact location within this code where a fault may
65        // occur.
66        //
67        // Copy 8 bytes at a time and then one byte at a time for the remainder.
68        "mov rcx, rdx",
69        "shr rcx, 3",
70        "rep movsq",
71        "and edx, 7",
72        "je 2f",
73        "mov ecx, edx",
74        "rep movsb",
75        // Set return value to `true`.
76        "2:",
77        "mov eax, 1",
78        // Reset the `fault_resume` pointer and return.
79        "3:",
80        "mov qword ptr [r10], 0",
81        "ret",
82        // `fault_resume` handler - set return value to `false` and return.
83        "1:",
84        "xor eax, eax",
85        "jmp 3b",
86    )
87}

89/// Copy a structure from userspace memory. Returns whether the copy was successful.
90#[must_use]
91fn copy_from_user<T>(dest: &mut MaybeUninit<T>, src: *const T) -> bool {
92    let fault_resume = unsafe { PF_RESUME.addr() }.as_ptr();
93    let size = core::mem::size_of::<T>();
94
95    user_access_ok(src);
96
97    // SAFETY: We have verified that the `src` pointer is within the userland address space.
98    unsafe { copy_to_from_user(dest.as_mut_ptr().cast(), src.cast(), size, fault_resume) }
99}

101/// Copy a structure from userspace memory. Returns whether the copy was successful.
102#[must_use]
103fn copy_to_user<T>(dest: *mut T, src: &T) -> bool {
104    let fault_resume = unsafe { PF_RESUME.addr() }.as_ptr();
105    let size = core::mem::size_of::<T>();
106    let src_ptr = src as *const T;
107
108    user_access_ok(dest);
109
110    // SAFETY: We have verified that the `dest` pointer is within the userland address space.
111    unsafe { copy_to_from_user(dest.cast(), src_ptr.cast(), size, fault_resume) }
112}

114/// A reference to a structure in userspace memory, which can be either read-only or read-write.
115///
116/// Concurrent access, *including data races to/from userspace memory*, are permitted. See the
117/// documentation of [`copy_to_from_user`] for more information.
118pub struct UserRef<T> {
119    ptr: *mut T,
120    val: T,
121}
122
123impl<T> UserRef<T> {
124    pub unsafe fn new(address: VirtAddr) -> Self {
125        let mut val = MaybeUninit::<T>::uninit();
126
127        // FIXME: Return an error if the copy fails.
128        assert!(copy_from_user(&mut val, address.as_ptr()));
129
130        Self {
131            ptr: address.as_mut_ptr(),
132            // SAFETY: We have initialized the value via `copy_from_user` above.
133            val: unsafe { val.assume_init() },
134        }
135    }
136
137    pub fn take(self) -> T
138    where
139        T: Clone,
140    {
141        self.val.clone()
142    }
143}
144
145impl<T> Deref for UserRef<T> {
146    type Target = T;
147
148    fn deref(&self) -> &Self::Target {
149        &self.val
150    }
151}
152
153impl<T> DerefMut for UserRef<T> {
154    fn deref_mut(&mut self) -> &mut Self::Target {
155        &mut self.val
156    }
157}
158
159impl<T> Drop for UserRef<T> {
160    fn drop(&mut self) {
161        assert!(copy_to_user(self.ptr, &self.val));
162    }
163}
164
165impl<T: Debug> Display for UserRef<T> {
166    fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
167        self.val.fmt(f)
168    }
169}
170
171impl<T: Debug> Debug for UserRef<T> {
172    fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
173        write!(f, "UserRef({:?})", self.val)
174    }
175}

177impl<T: Debug> SysArg for UserRef<T> {
178    fn from_usize(value: usize) -> Self {
179        // TODO: SAFETY

181    }
