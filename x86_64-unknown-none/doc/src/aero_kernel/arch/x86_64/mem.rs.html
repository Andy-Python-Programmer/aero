18use core::arch::naked_asm;
19
20fn should_store_by_byte() -> bool {
21    let cpuid = raw_cpuid::CpuId::new();
22    if let Some(features) = cpuid.get_extended_feature_info() {
23        // Check if "Enhanced" or "Fast Short" optimizations are available.
24        features.has_rep_movsb_stosb()
25    } else {
26        false
27    }
28}

30#[naked]
31unsafe extern "C" fn memcpy_movsq(dest: *mut u8, src: *const u8, len: usize) -> *mut u8 {
32    // Registers used:
33    //
34    // %rdi = argument 1, `dest`
35    // %rsi = argument 2, `src`
36    // %rdx = argument 3, `len`
37    naked_asm!(
38        // Save the return value.
39        "mov rax, rdi",
40        // Copy in 8 byte chunks.
41        "mov rcx, rdx",
42        "shr rcx, 3",
43        "rep movsq",
44        // Copy the rest.
45        "mov rcx, rdx",
46        "and rcx, 0x7",
47        "rep movsb",
48        "ret",
49    );
50}

52#[naked]
53unsafe extern "C" fn memcpy_movsb(dest: *mut u8, src: *const u8, len: usize) -> *mut u8 {
54    // Registers used:
55    //
56    // %rdi = argument 1, `dest`
57    // %rsi = argument 2, `src`
58    // %rdx = argument 3, `len`
59    naked_asm!(
60        // Save the return value.
61        "mov rax, rdi",
62        // Copy!
63        "mov rcx, rdx",
64        "rep movsb",
65        "ret",
66    )
67}
68
69#[indirect]
70extern "C" fn memcpy() -> fn(dest: *mut u8, src: *const u8, len: usize) {
71    if should_store_by_byte() {
72        memcpy_movsb
73    } else {
74        memcpy_movsq
75    }
76}

78#[naked]
79unsafe extern "C" fn memset_stosq(dest: *mut u8, byte: i32, len: usize) -> *mut u8 {
80    // Registers used:
81    //
82    // %rdi = argument 1, `dest`
83    // %rsi = argument 2, `byte`
84    // %rdx = argument 3, `len`
85    naked_asm!(
86        // Save the return value.
87        "mov r11, rdi",
88        // Create an 8-byte copy of the pattern.
89        "mov rcx, rdx",
90        "movzx rax, sil",
91        "mov r10, 0x0101010101010101",
92        "mul r10",
93        "mov rdx, rcx",
94        // Copy in 8 byte chunks.
95        "shr rcx, 3",
96        "rep stosq",
97        // Copy the rest.
98        "mov rcx, rdx",
99        "and rcx, 0x7",
100        "rep stosb",
101        // Restore the return value.
102        "mov rax, r11",
103        "ret",
104    )
105}

107#[naked]
108unsafe extern "C" fn memset_stosb(dest: *mut u8, byte: i32, len: usize) -> *mut u8 {
109    // Registers used:
110    //
111    // %rdi = argument 1, `dest`
112    // %rsi = argument 2, `byte`
113    // %rdx = argument 3, `len`
114    naked_asm!(
115        // Save the return value.
116        "mov r11, rdi",
117        "mov al, sil",
118        "mov rcx, rdx",
119        "rep stosb",
120        "mov rax, r11",
121        "ret",
122    )
123}
124
125#[indirect]
126extern "C" fn memset() -> fn(dest: *mut u8, byte: i32, len: usize) {
127    if should_store_by_byte() {
128        memset_stosb
129    } else {
130        memset_stosq
131    }
132}

134#[no_mangle]
135#[naked]
136unsafe extern "C" fn memmove_erms(dest: *mut u8, src: *const u8, len: usize) -> *mut u8 {
137    // Registers used:
138    //
139    // %rdi = argument 1, `dest`
140    // %rsi = argument 2, `src`
141    // %rdx = argument 3, `len`
142    naked_asm!(
143        "mov rax, rdi",
144        // Skip zero length.
145        "test rdx, rdx",
146        "jz 2f",
147        // Copying forwards:
148        "mov rcx, rdx",
149        "cmp rdi, rsi",
150        "jb 1f",
151        // `src` == `dest`
152        "je 2f",
153        "lea rdx, [rsi + rcx]",
154        "cmp rdi, rdx",
155        "jb 3f",
156        "1:",
157        "rep movsb",
158        "2:",
159        "ret",
160        // Copying backwards:
161        "3:",
162        "lea rdi, [rdi + rcx - 1]",
163        "lea rsi, [rsi + rcx - 1]",
164        "std",
165        "rep movsb",
166        "cld",
167        "ret",
168    )
169}
170
171#[no_mangle]
172extern "C" fn memmove(dest: *mut u8, src: *const u8, len: usize) -> *mut u8 {
173    unsafe { memmove_erms(dest, src, len) }
