20/// 64-bit SYSCALL instruction entry point.
21///
22/// The instruction supports to to 6 arguments in registers.
23///
24/// Registers state on entry:
25/// * `RAX` - system call number
26/// * `RCX` - return address
27/// * `R11` - saved flags (note: R11 is callee-clobbered register in C ABI)
28/// * `RDI` - argument 1
29/// * `RSI` - argument 2
30/// * `RDX` - argument 3
31/// * `R10` - argument 4 (needs to be moved to RCX to conform to C ABI)
32/// * `R8`  - argument 5
33/// * `R9`  - argument 6
34///
35/// (note: `R12`..`R15`, `RBP`, `RBX` are callee-preserved in C ABI)
36///
37/// The instruction saves the `RIP` to `RCX`, clears `RFLAGS.RF` then saves `RFLAGS` to `R11`.
38/// Followed by, it loads the new `SS`, `CS`, and `RIP` from previously programmed MSRs.
39///
40/// The instruction also does not save anything on the stack and does *not* change the `RSP`.
41#[naked]
42unsafe extern "C" fn x86_64_syscall_handler() {
43    naked_asm!(
44        // make the GS base point to the kernel TLS
45        "swapgs",
46        // save the user stack pointer
47        "mov qword ptr gs:{tss_temp_ustack_off}, rsp",
48        // restore kernel stack
49        "mov rsp, qword ptr gs:{tss_rsp0_off}",
50        "push {userland_ss}",
51        // push userspace stack ptr
52        "push qword ptr gs:{tss_temp_ustack_off}",
53        "push r11",
54        "push {userland_cs}",
55        "push rcx",
56
57        "push rax",
58        asm_macros::push_scratch!(),
59        asm_macros::push_preserved!(),
60
61        // push a fake error code to match with the layout of `InterruptErrorStack`
62        "push 0",
63
64        "mov rdi, rsp",
65
66        "cld",
67        "call {x86_64_do_syscall}",
68        "cli",
69
70        // pop the fake error code
71        "add rsp, 8",
72
73        asm_macros::pop_preserved!(),
74        asm_macros::pop_scratch!(),
75
76        // cook the sysret frame
77        "pop rcx",
78        "add rsp, 8",
79        "pop r11",
80        "pop rsp",
81
82        // restore user GS
83        "swapgs",
84        "sysretq",
85
86        // constants:
87        userland_cs = const USER_CS.bits(),
88        userland_ss = const USER_SS.bits(),
89        // XXX: add 8 bytes to skip the x86_64 cpu local self ptr
90        tss_temp_ustack_off = const offset_of!(Tss, reserved2) + core::mem::size_of::<usize>(),
91        tss_rsp0_off = const offset_of!(Tss, rsp) + core::mem::size_of::<usize>(),
92        x86_64_do_syscall = sym x86_64_do_syscall,
93    )
94}

96/// 64-bit SYSENTER instruction entry point.
97///
98/// The SYSENTER mechanism performs a fast transition to the kernel.
99///
100/// The new `CS` is loaded from the `IA32_SYSENTER_CS` MSR, and the new instruction and stack
101/// pointers are loaded from `IA32_SYSENTER_EIP` and `IA32_SYSENTER_ESP`, respectively. `RFLAGS.IF`
102/// is cleared, but other flags are unchanged.
103///
104/// As the instruction does not save *any* state, the user is required to provide the return `RIP`
105/// and `RSP` in the `RCX` and `R11` registers, respectively. These addresses must be canonical.
106///
107/// The instruction expects the call number and arguments in the same registers as for SYSCALL.
108#[naked]
109unsafe extern "C" fn x86_64_sysenter_handler() {
110    naked_asm!(
111        "swapgs",
112        // Build the interrupt frame expected by the kernel.
113        "push {userland_ss}",
114        "push r11",
115        "pushfq",
116        "push {userland_cs}",
117        "push rcx",
118        // Mask the same flags as for SYSCALL.
119        // XXX: Up to this point the code can be single-stepped if the user sets TF.
120        "pushfq",
121        "and dword ptr [rsp], 0x300",
122        "popfq",
123        "push rax",
124        asm_macros::push_scratch!(),
125        asm_macros::push_preserved!(),
126        "push 0",
127        // Store the stack pointer (interrupt frame ptr) in `RBP` for safe keeping, and align the
128        // stack as specified by the SysV calling convention.
129        "mov rbp, rsp",
130        "and rsp, ~0xf",
131        "mov rdi, rbp",
132        "call {x86_64_check_sysenter}",
133        "mov rdi, rbp",
134        "call {x86_64_do_syscall}",
135        // Reload the stack pointer, skipping the error code.
136        "lea rsp, [rbp + 8]",
137        asm_macros::pop_preserved!(),
138        asm_macros::pop_scratch!(),
139        // Pop the `IRET` frame into the registers expected by `SYSEXIT`.
140        "pop rdx", // return `RIP` in `RDX`
141        "add rsp, 8",
142        "popfq",   // restore saved `RFLAGS`
143        "pop rcx", // return `RSP` in `RCX`
144        // SAFETY: The above call to `x86_64_check_sysenter` is guarantees that we execute
145        // `sysexit` with canonical addresses in RCX and RDX. Otherwise we would fault in the
146        // kernel having already swapped back to the user's GS.
147        "swapgs",
148        // SYSEXIT does *not* restore `IF` to re-enable interrupts.
149        // This is done here, rather then when restoring `RFLAGS` above, since `STI` will keep
150        "sti",
151        // interrupts inhibited until after the *following* instruction executes.
152        "sysexitq",
153        // constants:
154        userland_cs = const USER_CS.bits(),
155        userland_ss = const USER_SS.bits(),
156        x86_64_check_sysenter = sym x86_64_check_sysenter,
157        x86_64_do_syscall = sym x86_64_do_syscall,
158    )
159}

161fn arch_prctl(command: usize, address: usize) -> Result<usize, SyscallError> {
162    match command {
163        ARCH_SET_FS => unsafe {
164            let _guard = IrqGuard::new();
165
166            scheduler::get_scheduler()
167                .current_task()
168                .arch_task_mut()
169                .set_fs_base(VirtAddr::new(address as u64));
170
171            Ok(0x00)
172        },
173
174        ARCH_GET_FS => Ok(scheduler::get_scheduler()
175            .current_task()
176            .arch_task()
177            .get_fs_base()
178            .as_u64() as usize),
179
180        ARCH_SET_GS => unsafe {
181            let _guard = IrqGuard::new();
182
183            scheduler::get_scheduler()
184                .current_task()
185                .arch_task_mut()
186                .set_gs_base(VirtAddr::new(address as u64));
187
188            Ok(0x00)
189        },
190
191        ARCH_GET_GS => Ok(scheduler::get_scheduler()
192            .current_task()
193            .arch_task()
194            .get_gs_base()
195            .as_u64() as usize),
196
197        _ => Err(SyscallError::EINVAL),
198    }
199}

201/// Check the user-provided return addresses for system calls via SYSENTER
202///
203/// We cannot execute `sysexit` on return with non-canonical return addresses, or we
204/// will take a fault in the kernel with the user's GS base already swapped back.
205pub(super) extern "sysv64" fn x86_64_check_sysenter(stack: &mut InterruptErrorStack) {
206    let rip = stack.stack.iret.rip;
207    let rsp = stack.stack.iret.rsp;
208    let max_user_addr = super::task::userland_last_address().as_u64();
209
210    if rip > max_user_addr || rsp > max_user_addr {
211        log::error!("bad sysenter: rip={:#018x},rsp={:#018x}", rip, rsp);
212
213        // Forcibly kill the process, we have nowhere to return to.
214        scheduler::get_scheduler().exit(ExitStatus::Normal(-1));
215    }
216}

218pub(super) extern "C" fn x86_64_do_syscall(stack: &mut InterruptErrorStack) {
219    let stack = &mut stack.stack;
220
221    let syscall_number = stack.scratch.rax as usize; // syscall number
222    let a = stack.scratch.rdi as usize; // argument 1
223    let b = stack.scratch.rsi as usize; // argument 2
224    let c = stack.scratch.rdx as usize; // argument 3
225    let d = stack.scratch.r10 as usize; // argument 4
226    let e = stack.scratch.r8 as usize; // argument 5
227    let f = stack.scratch.r9 as usize; // argument 6
228
229    match syscall_number {
230        // handle arch-specific syscalls (`sigreturn` and `arch_prctl`):
231        aero_syscall::prelude::SYS_SIGRETURN => {
232            super::signals::sigreturn(stack);
233            return;
234        }
235
236        aero_syscall::prelude::SYS_ARCH_PRCTL => {
237            let result = self::arch_prctl(a, b);
238            let result_usize = aero_syscall::syscall_result_as_usize(result);
239
240            stack.scratch.rax = result_usize as _;
241            return;
242        }
243
244        aero_syscall::prelude::SYS_EXIT => {}
245        _ => unsafe { super::interrupts::enable_interrupts() },
246    }
247
248    let result_usize = crate::syscall::generic_do_syscall(syscall_number, a, b, c, d, e, f);
249
250    super::signals::syscall_check_signals(result_usize as isize, stack);
251    stack.scratch.rax = result_usize as _;
252}

254/// Initializes support for the `syscall` and `sysret` instructions for the
255/// current CPU.
256pub(super) fn init() {
257    let cpuid = CpuId::new();
258
259    // Check if syscall is supported as it is a required CPU feature for aero to run.
260    let has_syscall = cpuid
261        .get_extended_processor_and_feature_identifiers()
262        .map_or(false, |i| i.has_syscall_sysret());
263
264    assert!(has_syscall);
265
266    unsafe {
267        // Enable support for `syscall` and `sysret` instructions if the current
268        // CPU supports them and the target pointer width is 64.
269        let syscall_base = GdtEntryIndex::KERNEL_CODE << 3;
270        let sysret_base = (GdtEntryIndex::KERNEL_TLS << 3) | 3;
271
272        let star_hi = syscall_base as u32 | ((sysret_base as u32) << 16);
273
274        io::wrmsr(io::IA32_STAR, (star_hi as u64) << 32);
275        io::wrmsr(io::IA32_LSTAR, x86_64_syscall_handler as u64);
276
277        // Clear the trap flag and enable interrupts.
278        io::wrmsr(io::IA32_FMASK, 0x300);
279
280        // Set the EFER.SCE bit to enable the syscall feature
281        let efer = io::rdmsr(io::IA32_EFER);
282        io::wrmsr(io::IA32_EFER, efer | 1);
283    }
284
285    // Enable support for the `sysenter` instruction for system calls.
286    //
287    // This instruction is only supported on AMD processors in Legacy mode,
288    // not in Long mode (Compatibility or 64-bit modes), so still report support
289    // for it via `cpuid`. In this case the #UD exception is caught to handle the
290    // system call.
291    let has_sysenter = cpuid
292        .get_feature_info()
293        .map_or(false, |i| i.has_sysenter_sysexit());
294
295    if has_sysenter {
296        log::info!("enabling support for sysenter");
297        unsafe {
298            io::wrmsr(
299                io::IA32_SYSENTER_CS,
300                (GdtEntryIndex::KERNEL_CODE << 3) as u64,
301            );
302            io::wrmsr(io::IA32_SYSENTER_EIP, x86_64_sysenter_handler as u64);
303        }
304    }
305}
