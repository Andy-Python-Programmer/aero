18use core::alloc::Layout;
19use core::ops::{Deref, DerefMut};
20
21use crate::extern_sym;
22use crate::mem::paging::VirtAddr;
23
24use super::io;
25
26#[repr(C)]
27pub struct CpuLocal<T>(T);
28
29impl<T> CpuLocal<T> {
30    pub const fn new(val: T) -> Self {
31        Self(val)
32    }
33
34    pub fn addr(&self) -> VirtAddr {
35        let val: u64;
36
37        unsafe {
38            // gs:[0] -> SELF_PTR
39            asm!(
40                "mov {}, qword ptr gs:[0]",
41                lateout(reg) val,
42                options(nostack, preserves_flags, pure, readonly),
43            );
44        }
45
46        let self_addr = VirtAddr::new(self as *const _ as u64);
47        let section_addr = VirtAddr::new(extern_sym!(__cpu_local_start) as u64);
48
49        let offset = self_addr - section_addr;
50        VirtAddr::new(val) + offset
51    }
52}
53
54impl<T> Deref for CpuLocal<T> {
55    type Target = T;
56
57    fn deref(&self) -> &Self::Target {
58        unsafe { &*self.addr().as_ptr() }
59    }
60}
61
62impl<T> DerefMut for CpuLocal<T> {
63    fn deref_mut(&mut self) -> &mut Self::Target {
64        unsafe { &mut *self.addr().as_mut_ptr() }
65    }
66}
67
68/// The GS register holds a pointer to CPU-local data at a fixed offset of 0. While this approach
69/// requires an additional memory lookup for accessing the data, it enables better code optimization
70/// since the subsequent access is just a normal memory access. Considering the size of the
71/// CPU-local data, this optimization is beneficial.
72#[cpu_local(subsection = "self_ptr")]
73static SELF_PTR: u64 = 0;
74
75#[cpu_local]
76static mut CPUID: usize = 0;
77
78pub fn init(cpu_id: usize) {
79    let start = VirtAddr::new(extern_sym!(__cpu_local_start).addr() as u64);
80    let end = VirtAddr::new(extern_sym!(__cpu_local_end).addr() as u64);
81
82    unsafe {
83        let size = end - start;
84
85        let layout = Layout::from_size_align_unchecked(size as _, 64);
86        let data = alloc::alloc::alloc_zeroed(layout);
87
88        core::ptr::copy_nonoverlapping::<u8>(start.as_ptr(), data, size as usize);
89        *data.cast::<u64>() = data as u64;
90
91        io::wrmsr(io::IA32_GS_BASE, data as u64);
92        *CPUID = cpu_id;
93    }
94}
