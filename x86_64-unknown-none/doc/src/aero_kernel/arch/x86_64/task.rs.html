18//! ## How does `x86_64` context switching work?
19//!
20//! The [`arch_task_spinup`] function is responsible for switching the current
21//! task to the next one. This function works by updating the TSS's RSP0 field to point
22//! to the per-task kernel stack and switches the page table for the next process.
23//!
24//! After a task is born, it directly context switches to it's specific trampoline. The
25//! trampoline is responsible for jumping to its appropriate context. At the point when
26//! the trampoline is called, the stack should have the switch frame pushed.
27//!
28//! The switch frame is not saved or restored at context switch time. Instead its
29//! stored when you enter and leave the kernel since then the context switch function
30//! does not have to worry about clobbering the user mode register values since
31//! they are safely stored on the kernel stack.
32

81/// Returns the first address outside the user range.
82///
83/// ## Notes
84/// * On Intel CPUs, if a SYSCALL instruction is at the highest canonical address, then that syscall
85///   will enter the kernel with a non-canonical return address, and SYSRET will explode
86///   dangerously. We avoid this particular problem by preventing anything from being mapped at the
87///   maximum canonical address.
88///
89/// * On AMD CPUs in the Ryzen family, there's a nasty bug in which the CPUs malfunction if they
90///   execute code from the highest canonical page. They'll speculate right off the end of the
91///   canonical space and bad things happen. This is worked around in the same way as the Intel
92///   problem.
93pub fn userland_last_address() -> VirtAddr {
94    // Reference: https://elixir.bootlin.com/linux/latest/source/arch/x86/include/asm/page_64.h#L61
95    static CACHED: spin::Once<VirtAddr> = spin::Once::new();
96
97    *CACHED.call_once(|| {
98        let la57 = crate::mem::paging::level_5_paging_enabled();
99        let virtual_mask_shift: u64 = if la57 { 56 } else { 47 };
100
101        VirtAddr::new((1u64 << virtual_mask_shift) - Size4KiB::SIZE)
102    })
103}

105/// Returns whether the given pointer is within the userland address space.
106pub fn user_access_ok<T>(ptr: *const T) -> bool {
107    let size = core::mem::size_of::<T>();
108    VirtAddr::new(ptr as u64 + size as u64) <= super::task::userland_last_address()
109}

114const USERLAND_STACK_TOP: VirtAddr = VirtAddr::new(0x7fffffffe000);
115const USERLAND_STACK_BOTTOM: VirtAddr = USERLAND_STACK_TOP.const_sub_u64(USERLAND_STACK_SIZE);

117#[naked]
118unsafe extern "C" fn jump_userland_exec(stack: VirtAddr, rip: VirtAddr, rflags: u64) {
119    #[rustfmt::skip]
120    naked_asm!(
121        "push rdi", // stack
122        "push rsi", // rip
123        "push rdx", // rflags
124        "cli",
125        "pop r11",
126        "pop rcx",
127        "pop rsp",
128        "swapgs",
129        "sysretq"
130    );
131}

133#[naked]
134unsafe extern "C" fn task_spinup(prev: &mut Unique<Context>, next: &Context) {
135    naked_asm!(
136        // save callee-saved registers
137        "push rbp",
138        "push rbx",
139        "push r12",
140        "push r13",
141        "push r14",
142        "push r15",
143        // save CR3
144        "mov rax, cr3",
145        "push rax",
146        // update old context
147        "mov [rdi], rsp",
148        // switch to new stack
149        "mov rsp, rsi",
150        // restore CR3
151        "pop rax",
152        "mov cr3, rax",
153        // restore callee-saved registers
154        "pop r15",
155        "pop r14",
156        "pop r13",
157        "pop r12",
158        "pop rbx",
159        "pop rbp",
160        // resume the next thread
161        "ret",
162    );
163}

221    pub fn new_kernel(entry_point: VirtAddr, enable_interrupts: bool) -> Self {
222        let switch_stack = Self::alloc_switch_stack().unwrap().as_mut_ptr::<u8>();
223
224        let task_stack = unsafe {
225            let layout = Layout::from_size_align_unchecked(4096 * 16, 0x1000);
226            alloc_zeroed(layout).add(layout.size())
227        };
228
229        let address_space = AddressSpace::this();
230
231        let mut stack_ptr = switch_stack as u64;
232        let mut stack = StackHelper::new(&mut stack_ptr);
233
234        let kframe = unsafe { stack.offset::<InterruptErrorStack>() };
235
236        kframe.stack.iret.ss = 0x10; // kernel stack segment
237        kframe.stack.iret.cs = 0x08; // kernel code segment
238        kframe.stack.iret.rip = entry_point.as_u64();
239        kframe.stack.iret.rsp = task_stack as u64;
240        kframe.stack.iret.rflags = if enable_interrupts { 0x200 } else { 0x00 };
241
242        let context = unsafe { stack.offset::<Context>() };
243
244        *context = Context::default();
245        context.rip = iretq_init as u64;
246        context.cr3 = controlregs::read_cr3_raw();
247
248        Self {
249            context: unsafe { Unique::new_unchecked(context) },
250            address_space,
251            context_switch_rsp: VirtAddr::new(switch_stack as u64),
252            user: false,
253
254            fs_base: VirtAddr::zero(),
255            gs_base: VirtAddr::zero(),
256
257            fpu_storage: None,
258        }
259    }

261    pub fn clone_process(
262        &self,
263        entry: usize,
264        usr_stack: usize,
265    ) -> Result<Self, MapToError<Size4KiB>> {
266        log::trace!("ArchTask::clone_process(entry={entry:#x}, stack={usr_stack:#x})");
267
268        assert!(self.user, "cannot clone a kernel task");
269
270        let address_space = AddressSpace::this();
271        let switch_stack = Self::alloc_switch_stack()?.as_mut_ptr::<u8>();
272
273        let mut new_stack_ptr = switch_stack as u64;
274        let mut new_stack = StackHelper::new(&mut new_stack_ptr);
275
276        let mut old_stack_ptr = self.context_switch_rsp.as_u64();
277        let mut old_stack = StackHelper::new(&mut old_stack_ptr);
278
279        let old_registers_frame = unsafe { old_stack.offset::<InterruptErrorStack>() };
280
281        let registers_frame = unsafe { new_stack.offset::<InterruptErrorStack>() };
282        *registers_frame = *old_registers_frame;
283
284        // registers_frame.stack.iret.cs = old_registers_frame.stack.iret.cs;
285        // registers_frame.stack.iret.ss = old_registers_frame.stack.iret.ss;
286        // registers_frame.stack.iret.rflags = old_registers_frame.stack.iret.rflags;
287
288        registers_frame.stack.iret.rip = entry as _;
289        registers_frame.stack.iret.rsp = usr_stack as _;
290        // registers_frame.stack.iret.rflags = 0x200;
291
292        let context = unsafe { new_stack.offset::<Context>() };
293
294        *context = Context::default();
295        context.rip = fork_init as _;
296        context.cr3 = address_space.cr3().start_address().as_u64();
297
298        let mut fpu_storage = self.fpu_storage.unwrap().clone();
299
300        Ok(Self {
301            context: unsafe { Unique::new_unchecked(context) },
302            context_switch_rsp: VirtAddr::new(switch_stack as u64),
303            address_space,
304            user: true,
305
306            // The FS and GS bases are inherited from the parent process.
307            fs_base: VirtAddr::new(1),
308            gs_base: self.gs_base,
309
310            fpu_storage: Some(fpu_storage),
311        })
312    }

314    pub fn fork(&self, address_space: AddressSpace) -> Result<Self, MapToError<Size4KiB>> {
315        assert!(self.user, "cannot fork a kernel task");
316
317        // Since the fork function marks all of the userspace entries in both the forked
318        // and the parent address spaces as read only, we will flush the page table of the
319        // current process to trigger COW.
320        unsafe {
321            asm!("mov cr3, {}", in(reg) controlregs::read_cr3_raw(), options(nostack));
322        }
323
324        let switch_stack = Self::alloc_switch_stack()?.as_mut_ptr::<u8>();
325
326        let mut old_stack_ptr = self.context_switch_rsp.as_u64();
327        let mut old_stack = StackHelper::new(&mut old_stack_ptr);
328
329        let mut new_stack_ptr = switch_stack as u64;
330        let mut new_stack = StackHelper::new(&mut new_stack_ptr);
331
332        unsafe {
333            let registers_frame = new_stack.offset::<InterruptErrorStack>();
334            let old_registers_frame = old_stack.offset::<InterruptErrorStack>();
335
336            *registers_frame = *old_registers_frame;
337            registers_frame.stack.scratch.rax = 0x00; // Set the syscall result to 0
338        }
339
340        // Prepare the trampoline...
341        let context = unsafe { new_stack.offset::<Context>() };
342
343        *context = Context::default();
344        context.rip = fork_init as u64;
345        context.cr3 = address_space.cr3().start_address().as_u64();
346
347        let fpu_storage = self.fpu_storage.unwrap().clone();
348
349        Ok(Self {
350            context: unsafe { Unique::new_unchecked(context) },
351            context_switch_rsp: VirtAddr::new(switch_stack as u64),
352            address_space,
353            user: true,
354
355            // The FS and GS bases are inherited from the parent process.
356            fs_base: self.fs_base,
357            gs_base: self.gs_base,
358
359            fpu_storage: Some(fpu_storage),
360        })
361    }

363    pub fn exec(
364        &mut self,
365        vm: &Vm,
366        executable: &DirCacheItem,
367
368        argv: Option<ExecArgs>,
369        envv: Option<ExecArgs>,
370    ) -> Result<(), MapToError<Size4KiB>> {
371        let address_space = if self.user {
372            self.unref_pt();
373            AddressSpace::new()?
374        } else {
375            AddressSpace::new()?
376        };
377
378        let loaded_binary = vm
379            .load_bin(executable, argv, envv)
380            .expect("exec: failed to load ELF");
381
382        // a kernel task can only execute a user executable
383        self.user = true;
384
385        // mmap the userland stack...
386        vm.mmap(
387            USERLAND_STACK_BOTTOM,
388            USERLAND_STACK_SIZE as usize,
389            MMapProt::PROT_WRITE | MMapProt::PROT_READ,
390            MMapFlags::MAP_FIXED | MMapFlags::MAP_PRIVATE | MMapFlags::MAP_ANONYOMUS,
391            0,
392            None,
393        );
394
395        address_space.switch(); // Perform the address space switch
396
397        self.context = Unique::dangling();
398        self.address_space = address_space; // Update the address space reference
399
400        self.fs_base = VirtAddr::zero();
401        self.gs_base = VirtAddr::zero();
402
403        let mut fpu_storage = FpuState::default();
404
405        // unsafe {
406        //     xrstor(&fpu_storage);
407
408        //     // The x87 FPU control word is set to 0x37f (default), which masks all
409        //     // floating-point exceptions, sets rounding to nearest, and sets the x87
410        //     // FPU precision to 64 bits (as documented in Intel SDM volume 1 section
411        //     // 8.1.5).
412        //     const DEFAULT_FPU_CWORD: u16 = 0x37f;
413        //     asm!("fldcw [{}]", in(reg) &DEFAULT_FPU_CWORD, options(nomem));
414
415        //     // Set the default MXCSR value at reset as documented in Intel SDM volume 2A.
416        //     controlregs::write_mxcsr(
417        //         MxCsr::INVALID_OPERATION_MASK
418        //             | MxCsr::DENORMAL_MASK
419        //             | MxCsr::DIVIDE_BY_ZERO_MASK
420        //             | MxCsr::OVERFLOW_MASK
421        //             | MxCsr::UNDERFLOW_MASK
422        //             | MxCsr::PRECISION_MASK,
423        //     );
424
425        //     xsave(&mut fpu_storage);
426        // }
427
428        self.fpu_storage = Some(fpu_storage);
429
430        let mut stack_addr = USERLAND_STACK_TOP.as_u64();
431        let mut stack = StackHelper::new(&mut stack_addr);
432
433        let mut envp = Vec::new();
434        let mut argp = Vec::new();
435
436        if let Some(envv) = loaded_binary.envv {
437            envp = envv.push_into_stack(&mut stack);
438        }
439
440        if let Some(argv) = loaded_binary.argv {
441            argp = argv.push_into_stack(&mut stack);
442        }
443
444        stack.align_down();
445
446        let size = envp.len() + 1 + argp.len() + 1 + 1;
447
448        if size % 2 == 1 {
449            unsafe {
450                stack.write(0u64);
451            }
452        }
453
454        let p2_header = loaded_binary.elf.header.pt2;
455
456        unsafe {
457            let hdr: [(AuxvType, usize); 5] = [
458                (
459                    AuxvType::Phdr,
460                    (p2_header.ph_offset() + loaded_binary.base_addr.as_u64()) as usize,
461                ),
462                (AuxvType::PhEnt, p2_header.ph_entry_size() as usize),
463                (AuxvType::PhNum, p2_header.ph_count() as usize),
464                (AuxvType::Entry, p2_header.entry_point() as usize),
465                (AuxvType::Secure, 0),
466            ];
467
468            stack.write(0usize); // Make it 16 bytes aligned
469            stack.write(AuxvType::Null);
470            stack.write(hdr);
471        }
472
473        // struct ExecStackData {
474        //     argc: isize,
475        //     argv: *const *const u8,
476        //     envv: *const *const u8,
477        // }
478        unsafe {
479            stack.write(0u64);
480            stack.write_slice(envp.as_slice());
481            stack.write(0u64);
482            stack.write_slice(argp.as_slice());
483            stack.write(argp.len());
484        }
485
486        core::mem::drop(envp);
487        core::mem::drop(argp);
488
489        assert_eq!(stack.top() % 16, 0);
490
491        unsafe {
492            jump_userland_exec(VirtAddr::new(stack.top()), loaded_binary.entry_point, 0x200);
493        }
494
495        Ok(())
496    }

498    /// Allocates a new context switch stack for the process and returns the stack
499    /// top address. See the module level documentation for more information.
500    fn alloc_switch_stack() -> Result<VirtAddr, MapToError<Size4KiB>> {
501        let frame = FRAME_ALLOCATOR.alloc_zeroed(4096 * 4).unwrap();
502
503        Ok(frame.as_hhdm_virt() + (4096u64 * 4))
504    }

506    fn unref_pt(&mut self) {
507        assert!(self.user);
508
509        self.address_space
510            .offset_page_table()
511            .page_table()
512            .for_each_entry(
513                PageTableFlags::PRESENT | PageTableFlags::USER_ACCESSIBLE,
514                |entry| {
515                    entry.unref_vm_frame();
516                    entry.set_unused();
517
518                    Ok(())
519                },
520            )
521            .expect("dealloc: failed to unref the page table");
522    }

524    /// Deallocates the architecture-specific task resources. This function is called
525    /// when the process is turned into a zombie.
526    pub fn dealloc(&mut self) {
527        if self.user {
528            self.unref_pt();
529        }
530
531        // deallocate the switch stack
532        {
533            let frame: PhysFrame<Size4KiB> = PhysFrame::containing_address(
534                (self.context_switch_rsp - Size4KiB::SIZE).as_hhdm_phys(),
535            );
536
537            FRAME_ALLOCATOR.deallocate_frame(frame);
538        }
539    }

584#[derive(Debug, Copy, Clone)]
585#[repr(C, align(16))]
586pub struct FpuState {
587    /// x87 FPU Control Word (16 bits). See Figure 8-6 in the Intel 64 and IA-32 Architectures
588    /// Software Developers Manual Volume 1, for the layout of the x87 FPU control word.
589    pub fcw: u16,
590    /// x87 FPU Status Word (16 bits).
591    pub fsw: u16,
592    /// x87 FPU Tag Word (8 bits) + reserved (8 bits).
593    pub ftw: u16,
594    /// x87 FPU Opcode (16 bits).
595    pub fop: u16,
596    /// x87 FPU Instruction Pointer Offset ([31:0]). The contents of this field differ depending on
597    /// the current addressing mode (32-bit, 16-bit, or 64-bit) of the processor when the
598    /// FXSAVE instruction was executed: 32-bit mode  32-bit IP offset. 16-bit mode  low 16
599    /// bits are IP offset; high 16 bits are reserved. 64-bit mode with REX.W  64-bit IP
600    /// offset. 64-bit mode without REX.W  32-bit IP offset.
601    pub fip: u32,
602    /// x87 FPU Instruction Pointer Selector (16 bits) + reserved (16 bits).
603    pub fcs: u32,
604    /// x87 FPU Instruction Operand (Data) Pointer Offset ([31:0]). The contents of this field
605    /// differ depending on the current addressing mode (32-bit, 16-bit, or 64-bit) of the
606    /// processor when the FXSAVE instruction was executed: 32-bit mode  32-bit DP offset.
607    /// 16-bit mode  low 16 bits are DP offset; high 16 bits are reserved. 64-bit mode
608    /// with REX.W  64-bit DP offset. 64-bit mode without REX.W  32-bit DP offset.
609    pub fdp: u32,
610    /// x87 FPU Instruction Operand (Data) Pointer Selector (16 bits) + reserved.
611    pub fds: u32,
612    /// MXCSR Register State (32 bits).
613    pub mxcsr: u32,
614    /// This mask can be used to adjust values written to the MXCSR register, ensuring that
615    /// reserved bits are set to 0. Set the mask bits and flags in MXCSR to the mode of
616    /// operation desired for SSE and SSE2 SIMD floating-point instructions.
617    pub mxcsr_mask: u32,
618    /// x87 FPU or MMX technology registers. Layout: [12 .. 9 | 9 ... 0] LHS = reserved; RHS = mm.
619    pub mm: [u128; 8],
620    /// XMM registers (128 bits per field).
621    pub xmm: [u128; 16],
622    /// reserved.
623    pub _pad: [u64; 12],
624}
625
626impl Default for FpuState {
627    fn default() -> Self {
628        Self {
629            mxcsr: 0x1f80,
630            mxcsr_mask: 0x037f,
631            // rest are zeroed
632            fcw: 0,
633            fsw: 0,
634            ftw: 0,
635            fop: 0,
636            fip: 0,
637            fcs: 0,
638            fdp: 0,
639            fds: 0,
640            mm: [0; 8],
641            xmm: [u128::MAX; 16],
642            _pad: [0; 12],
643        }
644    }
645}

647fn xsave(fpu: &mut FpuState) {
648    // The implicit EDX:EAX register pair specifies a 64-bit instruction mask. The specific state
649    // components saved correspond to the bits set in the requested-feature bitmap (RFBM), which is
650    // the logical-AND of EDX:EAX and XCR0.
651    // unsafe {
652    //     asm!("xsave64 [{}]", in(reg) fpu.as_ptr(), in("eax") u32::MAX, in("edx") u32::MAX,
653    // options(nomem, nostack)) }
654
655    use core::arch::x86_64::_fxsave64;
656
657    unsafe { _fxsave64((fpu as *mut FpuState).cast()) }
658}

660fn xrstor(fpu: &FpuState) {
661    // unsafe {
662    //     asm!("xrstor [{}]", in(reg) fpu.as_ptr(), in("eax") u32::MAX, in("edx") u32::MAX,
663    // options(nomem, nostack)); }
664    use core::arch::x86_64::_fxrstor64;
665
666    unsafe { _fxrstor64((fpu as *const FpuState).cast()) }
667}

669/// Check out the module level documentation for more information.
670pub fn arch_task_spinup(from: &mut ArchTask, to: &ArchTask) {
671    unsafe {
672        if let Some(fpu) = from.fpu_storage.as_mut() {
673            xsave(fpu);
674        }
675
676        if let Some(fpu) = to.fpu_storage.as_ref() {
677            xrstor(fpu);
678        }
679
680        // Load the new thread's kernel stack pointer everywhere it's needed.
681        let kstackp = to.context_switch_rsp.as_u64();
682        super::gdt::TSS.rsp[0] = kstackp;
683        io::wrmsr(io::IA32_SYSENTER_ESP, kstackp);
684
685        // Preserve and restore the %fs, %gs bases.
686        from.fs_base = io::get_fsbase();
687        from.gs_base = io::get_inactive_gsbase();
688
689        io::set_fsbase(to.fs_base);
690        io::set_inactive_gsbase(to.gs_base);
691
692        task_spinup(&mut from.context, to.context.as_ref());
693    }
694}
