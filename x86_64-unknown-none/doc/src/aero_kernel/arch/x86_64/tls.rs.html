18//! Thread Local Storage (TLS) are per-thread global variables. On 64-bit each CPU core's
19//! `fs` GDT segment points to the thread local memory area where the thread local static's
20//! live. TLS statics are simply accessed through an offset from `fs`.
21//!
22//! ## Notes
23//! * <https://wiki.osdev.org/Thread_Local_Storage>
24//! * <https://doc.rust-lang.org/std/thread/struct.LocalKey.html>

95static CPU_INFO: Mutex<Vec<CpuInfo>> = Mutex::new(Vec::new());
96
97pub struct CpuInfo {
98    pub cpuid: usize,
99
100    pub fpu: bool,
101    pub vendor: Option<String>,
102    pub brand: Option<String>,
103    pub features: Vec<&'static &'static str>,
104}
105
106pub fn get_cpuid() -> usize {
107    0
108}
109
110pub fn init() {
111    let cpuid = raw_cpuid::CpuId::new();
112
113    let features = cpuid
114        .get_feature_info()
115        .map(|cpu_features| {
116            CPU_FEATURES
117                .iter()
118                .filter(|(_, check_fn)| (check_fn)(&cpu_features))
119                .map(|(name, _)| name)
120                .collect::<Vec<_>>()
121        })
122        .unwrap_or_default();
123
124    CPU_INFO.lock().push(CpuInfo {
125        cpuid: 0,
126
127        fpu: cpuid
128            .get_feature_info()
129            .map(|e| e.has_fpu())
130            .unwrap_or(false),
131
132        features,
133        vendor: cpuid.get_vendor_info().map(|e| String::from(e.as_str())),
134        brand: cpuid
135            .get_processor_brand_string()
136            .map(|e| String::from(e.as_str())),
137    })
138}
139
140pub fn for_cpu_info_cached<F>(mut f: F)
141where
142    F: FnMut(&CpuInfo),
143{
144    let lock = CPU_INFO.lock();
145
146    for info in lock.iter() {
147        f(info);
148    }
149}
