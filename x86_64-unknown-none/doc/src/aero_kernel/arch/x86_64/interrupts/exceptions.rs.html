31macro interrupt_exception(fn $name:ident() => $message:expr) {
32    pub fn $name(stack: &mut InterruptErrorStack) {
33        unwind::prepare_panic();
34
35        log::error!("EXCEPTION: {}", $message);
36        log::error!("FS={:#x}", unsafe { io::rdmsr(io::IA32_FS_BASE) },);
37        log::error!("GS={:#x}", unsafe { io::rdmsr(io::IA32_GS_BASE) });
38        log::error!("Stack: {:#x?}", stack);
39        dbg!(
40            scheduler::get_scheduler()
41                .current_task()
42                .arch_task()
43                .fpu_storage
44        );
45
46        if stack.stack.iret.rip != 0 {
47            unsafe {
48                log::error!(
49                    "RIP={:?}",
50                    core::slice::from_raw_parts(stack.stack.iret.rip as *const u8, 512)
51                );
52            }
53        }
54
55        unwind::unwind_stack_trace();
56
57        unsafe {
58            loop {
59                super::halt();
60            }
61        }
62    }
63}

82pub fn simd(stack: &mut InterruptErrorStack) {

98pub fn invalid_opcode(stack: &mut InterruptErrorStack) {
99    // Catch SYSENTER on AMD CPUs.
100    //
101    // The RIP on the stack for #UD points to the instruction which generated the exception.
102    // The return RIP and RSP need to be changed to the user-provided values in RCX and R11.
103    const SYSENTER_OPCODE: [u8; 2] = [0x0f, 0x34];
104
105    let opcode = unsafe { *(stack.stack.iret.rip as *const [u8; 2]) };
106    if opcode == SYSENTER_OPCODE {
107        log::debug!("handling SYSENTER via #UD");
108
109        stack.stack.iret.rip = stack.stack.scratch.rcx;
110        stack.stack.iret.rsp = stack.stack.scratch.r11;
111
112        super::super::syscall::x86_64_check_sysenter(stack);
113        super::super::syscall::x86_64_do_syscall(stack);
114        return;
115    }
116
117    // Otherwise handle the exception as normal.
118
119    unwind::prepare_panic();
120
121    log::error!("EXCEPTION: Invalid Opcode");
122    log::error!("Stack: {:#x?}", stack);
123
124    unwind::unwind_stack_trace();
125
126    unsafe {
127        loop {
128            super::halt();
129        }
130    }
131}

133pub fn breakpoint(stack: &mut InterruptErrorStack) {
134    // We will need to prevent RIP from going out of sync with
135    // instructions.
136    //
137    // So we will set the RIP to RIP - 1, pointing to the int3
138    // instruction.
139    stack.stack.iret.rip -= 1;
140}

142pub(super) fn page_fault(stack: &mut InterruptErrorStack) {
143    let pf_resume = unsafe { *PF_RESUME };
144    if !pf_resume.is_zero() {
145        stack.stack.iret.rip = pf_resume.as_u64();
146        return;
147    }
148
149    let accessed_address = controlregs::read_cr2();
150    let reason = PageFaultErrorCode::from_bits_truncate(stack.code);
151
152    // We cannot directly check if we want to handle the page fault by checking
153    // if the CS register contains the RPL_3 flag since, we also want to handle the
154    // situation where we are trying to access a user provided buffer in the kernel and
155    // its not mapped. So we handle the page fault if the accessed address is less then the
156    // MAX userland address and we only signal kill the process if its trying to access
157    // a non-mapped memory region while in RPL_3.
158    let userland_last_address = super::super::task::userland_last_address();
159
160    // prints out the error information for this page fault.
161    let print_info = || {
162        log::error!("");
163        log::error!("FS={:#x}", unsafe { io::rdmsr(io::IA32_FS_BASE) },);
164        log::error!("GS={:#x}", unsafe { io::rdmsr(io::IA32_GS_BASE) });
165        log::error!("");
166        log::error!("accessed address: {:#x}", accessed_address);
167        log::error!("reason: {:?}", reason);
168        log::error!("");
169        log::error!("stack: {:#x?}", stack);
170    };
171
172    if accessed_address < userland_last_address && scheduler::is_initialized()
173        || stack.stack.iret.is_user()
174    {
175        let signal = scheduler::get_scheduler()
176            .current_task()
177            .vm
178            .handle_page_fault(reason, accessed_address);
179
180        if !signal && stack.stack.iret.is_user() {
181            log::error!("Segmentation fault");
182            print_info();
183
184            let task = scheduler::get_scheduler().current_task();
185
186            log::error!(
187                "process: (tid={}, pid={})",
188                task.tid().as_usize(),
189                task.pid().as_usize()
190            );
191
192            log::error!(
193                "process: (path=`{}`)",
194                task.path()
195                    .expect("userland application does not have a path set")
196            );
197
198            task.file_table.log();
199
200            if LOG_PF_PTABLE {
201                scheduler::get_scheduler().log_ptable();
202            }
203
204            unwind::unwind_stack_trace();
205
206            let task = scheduler::get_scheduler().current_task();
207            task.signal(aero_syscall::signal::SIGSEGV);
208            return;
209        } else if !signal {
210        } else {
211            return;
212        }
213    }
214
215    unwind::prepare_panic();
216
217    log::error!("Page fault");
218    print_info();
219
220    unwind::unwind_stack_trace();
221
222    unsafe {
223        loop {
224            super::halt();
225        }
226    }
