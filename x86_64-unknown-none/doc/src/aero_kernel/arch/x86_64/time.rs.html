18//! The PIT (Programmable Interval Timer) chip basically consists of an oscillator,
19//! a prescaler and 3 independent frequency dividers and it is used to create time intervals
20//! and calculate *estimate* time since epoch.
21//!
22//! **Notes**: <https://wiki.osdev.org/Programmable_Interval_Timer>
23
24use core::sync::atomic::{AtomicUsize, Ordering};
25
26use aero_syscall::TimeSpec;
27
28use super::apic;
29
30use crate::arch::interrupts;
31use crate::arch::interrupts::InterruptStack;
32
33use crate::arch::io;
34use crate::utils::sync::Mutex;
35
36const PIT_FREQUENCY_HZ: usize = 1000;
37pub const PIT_DIVIDEND: usize = 1193182;
38
39static UPTIME_RAW: AtomicUsize = AtomicUsize::new(0);
40static UPTIME_SEC: AtomicUsize = AtomicUsize::new(0);
41
42pub static EPOCH: AtomicUsize = AtomicUsize::new(usize::MAX);
43pub static REALTIME_CLOCK: Mutex<aero_syscall::TimeSpec> = Mutex::new(aero_syscall::TimeSpec {
44    tv_sec: 0,
45    tv_nsec: 0,
46});
47
48pub fn get_uptime_ticks() -> usize {
49    UPTIME_SEC.load(Ordering::SeqCst)
50}
51
52pub fn get_realtime_clock() -> TimeSpec {
53    REALTIME_CLOCK.lock_irq().clone()
54}
55
56/// Returns the current amount of PIT ticks.
57pub fn get_current_count() -> u16 {
58    unsafe {
59        io::outb(0x43, 0);
60
61        let lower = io::inb(0x40) as u16;
62        let higher = io::inb(0x40) as u16;
63
64        (higher << 8) | lower
65    }
66}
67
68pub fn set_reload_value(new_count: u16) {
69    // Channel 0, lo/hi access mode, mode 2 (rate generator)
70    unsafe {
71        io::outb(0x43, 0x34);
72        io::outb(0x40, new_count as u8);
73        io::outb(0x40, (new_count >> 8) as u8);
74    }
75}
76
77pub fn set_frequency(frequency: usize) {
78    let mut new_divisor = PIT_DIVIDEND / frequency;
79
80    if PIT_DIVIDEND % frequency > frequency / 2 {
81        new_divisor += 1;
82    }
83
84    set_reload_value(new_divisor as u16);
85}
86
87fn pit_irq_handler(_stack: &mut InterruptStack) {
88    {
89        let interval = aero_syscall::TimeSpec {
90            tv_sec: 0,
91            tv_nsec: (1000000000 / PIT_FREQUENCY_HZ) as isize,
92        };
93
94        let mut this = REALTIME_CLOCK.lock_irq();
95
96        if this.tv_nsec + interval.tv_nsec > 999999999 {
97            let diff = (this.tv_nsec + interval.tv_nsec) - 1000000000;
98
99            this.tv_nsec = diff;
100            this.tv_sec += 1;
101        } else {
102            this.tv_nsec += interval.tv_nsec
103        }
104
105        this.tv_sec += interval.tv_sec;
106    }
107
108    let value = UPTIME_RAW.fetch_add(1, Ordering::Relaxed); // Increment uptime raw ticks.
109
110    if value % PIT_FREQUENCY_HZ == 0 {
111        UPTIME_SEC.fetch_add(1, Ordering::Relaxed); // Increment uptime seconds
112        crate::syscall::time::check_timers();
113    }
114}
115
116/// This function is responsible for initializing the PIT chip and setting
117/// up the IRQ.
118pub fn init() {
119    apic::get_local_apic().timer_calibrate();
120
121    REALTIME_CLOCK.lock().tv_sec = EPOCH.load(Ordering::SeqCst) as _;
122
123    set_frequency(PIT_FREQUENCY_HZ);
124
125    let pit_vector = interrupts::allocate_vector();
126    interrupts::register_handler(pit_vector, pit_irq_handler);
127
128    apic::io_apic_setup_legacy_irq(0, pit_vector, 1); // Set up the IRQ.
129}
