184#[derive(Debug, Copy, Clone)]
185#[repr(transparent)]
186pub struct SegmentSelector(u16);
187
188impl SegmentSelector {
189    pub const fn empty() -> Self {
190        Self(0)
191    }
192
193    pub const fn new(index: u16, privilege_level: PrivilegeLevel) -> Self {
194        Self(index << 3 | (privilege_level as u16))
195    }
196
197    pub const fn bits(&self) -> u16 {
198        self.0
199    }
200
201    pub const fn from_bits(value: u16) -> Self {
202        Self(value)
203    }
204
205    pub const fn privilege_level(&self) -> PrivilegeLevel {
206        match self.bits() & 0b11 {
207            0 => PrivilegeLevel::Ring0,
208            3 => PrivilegeLevel::Ring3,
209            _ => unreachable!(),
210        }
211    }
212}

214#[repr(C, packed)]
215struct GdtDescriptor {
216    /// The size of the table subtracted by 1.
217    /// The size of the table is subtracted by 1 as the maximum value
218    /// of `size` is 65535, while the GDT can be up to 65536 bytes.
219    size: u16,
220    /// The linear address of the table.
221    offset: u64,
222}
223
224impl GdtDescriptor {
225    /// Create a new GDT descriptor.
226    #[inline]
227    pub const fn new(size: u16, offset: u64) -> Self {
228        Self { size, offset }
229    }
230}

232#[derive(Debug, Clone, Copy)]
233#[repr(C)]
234pub(super) struct GdtEntry {
235    limit_low: u16,
236    base_low: u16,
237    base_middle: u8,
238    access_byte: u8,
239    /// The limit high and the flags.
240    ///
241    /// **Note**: Four bits of the variable is the limit and rest four bits of the
242    /// variable are the flags.
243    limit_hi_flags: u8,
244    base_hi: u8,
245}
246
247impl GdtEntry {
248    const NULL: Self = Self::new(GdtAccessFlags::NULL, GdtEntryFlags::empty());
249
250    const fn new(access_flags: u8, entry_flags: GdtEntryFlags) -> Self {
251        Self {
252            limit_low: 0x00,
253            base_low: 0x00,
254            base_middle: 0x00,
255            access_byte: access_flags,
256            limit_hi_flags: entry_flags.bits() & 0xF0,
257            base_hi: 0x00,
258        }
259    }
260
261    fn set_offset(&mut self, offset: u32) {
262        self.base_low = offset as u16;
263        self.base_middle = (offset >> 16) as u8;
264        self.base_hi = (offset >> 24) as u8;
265    }
266
267    fn set_limit(&mut self, limit: u32) {
268        self.limit_low = limit as u16;
269        self.limit_hi_flags = self.limit_hi_flags & 0xF0 | ((limit >> 16) as u8) & 0x0F;
270    }
271
272    fn set_raw<T>(&mut self, value: T) {
273        unsafe {
274            *(ptr::addr_of_mut!(*self).cast::<T>()) = value;
275        }
276    }
277}

279/// Although hardware task-switching is not supported in 64-bit mode, a task
280/// state segment is required since the it holds information important to 64-bit
281/// mode and its not directly related to task switching mechanism.
282#[repr(C, packed)]
283pub struct Tss {
284    reserved: u32, // offset 0x00
285
286    /// The full 64-bit canonical forms of the stack pointers (RSP) for
287    /// privilege levels 0-2.
288    pub rsp: [u64; 3], // offset 0x04
289    pub reserved2: u64, // offset 0x1C
290
291    /// The full 64-bit canonical forms of the interrupt stack table
292    /// (IST) pointers.
293    pub ist: [u64; 7], // offset 0x24
294    reserved3: u64, // offset 0x5c
295    reserved4: u16, // offset 0x64
296
297    /// The 16-bit offset to the I/O permission bit map from the 64-bit
298    /// TSS base.
299    pub iomap_base: u16, // offset 0x66
300}
301
302#[cpu_local(subsection = "tss")]
303pub static mut TSS: Tss = Tss {
304    reserved: 0,
305    rsp: [0; 3],
306    reserved2: 0,
307    ist: [0; 7],
308    reserved3: 0,
309    reserved4: 0,
310    iomap_base: 0,
311};

313/// Initialize the bootstrap GDT which is required to initialize TLS (Thread Local Storage)
314/// support so, after the kernel heap we will map the TLS section and initialize the *actual* GDT
315/// and then each CPU will have it's own GDT but we only will have to define it once as a
316/// `#[thread_local]`.
317pub fn init_boot() {
318    unsafe {
319        let gdt_descriptor = GdtDescriptor::new(
320            (mem::size_of::<[GdtEntry; BOOT_GDT_ENTRY_COUNT]>() - 1) as u16,
321            addr_of!(BOOT_GDT).addr() as u64,
322        );
323
324        load_gdt(&gdt_descriptor);
325    }
326
327    // Load the GDT segments.
328    unsafe {
329        load_cs(SegmentSelector::new(
330            GdtEntryIndex::KERNEL_CODE,
331            PrivilegeLevel::Ring0,
332        ));
333
334        load_ds(SegmentSelector::new(
335            GdtEntryIndex::KERNEL_DATA,
336            PrivilegeLevel::Ring0,
337        ));
338
339        load_es(SegmentSelector::new(
340            GdtEntryIndex::KERNEL_DATA,
341            PrivilegeLevel::Ring0,
342        ));
343
344        load_fs(SegmentSelector::new(
345            GdtEntryIndex::KERNEL_DATA,
346            PrivilegeLevel::Ring0,
347        ));
348
349        load_gs(SegmentSelector::new(
350            GdtEntryIndex::KERNEL_TLS,
351            PrivilegeLevel::Ring0,
352        ));
353
354        load_ss(SegmentSelector::new(
355            GdtEntryIndex::KERNEL_DATA,
356            PrivilegeLevel::Ring0,
357        ));
358    }
359}

369/// Initialize the *actual* GDT stored in TLS.
370///
371/// ## Safety
372/// The heap must be initialized before this function is called.
373pub fn init() {
374    let gdt = unsafe {
375        let gdt_ent_size = core::mem::size_of::<GdtEntry>();
376        let gdt_ent_align = core::mem::align_of::<GdtEntry>();
377
378        let gdt_size = gdt_ent_size * GDT_ENTRY_COUNT;
379        let layout = Layout::from_size_align_unchecked(gdt_size, gdt_ent_align);
380
381        let ptr = alloc_zeroed(layout).cast::<GdtEntry>();
382        core::slice::from_raw_parts_mut::<GdtEntry>(ptr, GDT_ENTRY_COUNT)
383    };
384
385    // Copy over the GDT template:
386    gdt.copy_from_slice(&GDT);
387
388    unsafe {
389        let tss_ptr = TSS.addr().as_mut_ptr::<Tss>();
390
391        gdt[GdtEntryIndex::TSS as usize].set_offset(tss_ptr as u32);
392        gdt[GdtEntryIndex::TSS as usize].set_limit(mem::size_of::<Tss>() as u32);
393        gdt[GdtEntryIndex::TSS_HI as usize].set_raw((tss_ptr as u64) >> 32);
394
395        TSS.rsp[0] = STK.as_ptr().offset(4096 * 16) as u64;
396
397        let gdt_descriptor = GdtDescriptor::new(
398            (mem::size_of::<[GdtEntry; GDT_ENTRY_COUNT]>() - 1) as u16,
399            gdt.as_ptr() as u64,
400        );
401
402        load_gdt(&gdt_descriptor);
403
404        // Reload the GDT segments.
405        load_cs(SegmentSelector::new(
406            GdtEntryIndex::KERNEL_CODE,
407            PrivilegeLevel::Ring0,
408        ));
409        load_ds(SegmentSelector::new(
410            GdtEntryIndex::KERNEL_DATA,
411            PrivilegeLevel::Ring0,
412        ));
413        load_es(SegmentSelector::new(
414            GdtEntryIndex::KERNEL_DATA,
415            PrivilegeLevel::Ring0,
416        ));
417        load_ss(SegmentSelector::new(
418            GdtEntryIndex::KERNEL_DATA,
419            PrivilegeLevel::Ring0,
420        ));
421
422        // Load the Task State Segment.
423        load_tss(SegmentSelector::new(
424            GdtEntryIndex::TSS,
425            PrivilegeLevel::Ring0,
426        ));
427    }
428
429    // // Now we update the per-cpu storage to store a reference
430    // // to the per-cpu GDT.
431    // tls::get_percpu().gdt = gdt;
432}

434#[inline(always)]
435unsafe fn load_cs(selector: SegmentSelector) {
436    // NOTE: We cannot directly move into CS since x86 requires the IP
437    // and CS set at the same time. To do this, we need push the new segment
438    // selector and return value onto the stack and far return to reload CS and
439    // continue execution.
440    //
441    // We also cannot use a far call or a far jump since we would only be
442    // able to jump to 32-bit instruction pointers. Only Intel supports for
443    // 64-bit far calls/jumps in long-mode, AMD does not.
444    asm!(
445        "push {selector}",
446        "lea {tmp}, [1f + rip]",
447        "push {tmp}",
448        "retfq",
449        "1:",
450        selector = in(reg) u64::from(selector.bits()),
451        tmp = lateout(reg) _,
452    );
453}

455#[inline(always)]
456unsafe fn load_ds(selector: SegmentSelector) {
457    asm!("mov ds, {0:x}", in(reg) selector.bits(), options(nomem, nostack))
458}
459
460#[inline(always)]
461unsafe fn load_es(selector: SegmentSelector) {
462    asm!("mov es, {0:x}", in(reg) selector.bits(), options(nomem, nostack))
463}
464
465#[inline(always)]
466unsafe fn load_fs(selector: SegmentSelector) {
467    asm!("mov fs, {0:x}", in(reg) selector.bits(), options(nomem, nostack))
468}
469
470#[inline(always)]
471unsafe fn load_gs(selector: SegmentSelector) {
472    asm!("mov gs, {0:x}", in(reg) selector.bits(), options(nomem, nostack))
473}
474
475#[inline(always)]
476unsafe fn load_ss(selector: SegmentSelector) {
477    asm!("mov ss, {0:x}", in(reg) selector.bits(), options(nomem, nostack))
478}
479
480#[inline(always)]
481unsafe fn load_tss(selector: SegmentSelector) {
482    asm!("ltr {0:x}", in(reg) selector.bits(), options(nostack, nomem));
483}
484
485#[inline(always)]
486unsafe fn load_gdt(gdt_descriptor: &GdtDescriptor) {
487    asm!("lgdt [{}]", in(reg) gdt_descriptor, options(nostack));
