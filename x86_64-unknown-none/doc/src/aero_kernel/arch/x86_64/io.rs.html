55/// Wrapper function to the `outb` assembly instruction used to do the
56/// 8-bit low level port output.
57#[inline]
58pub unsafe fn outb(port: u16, value: u8) {
59    asm!(
60       "out dx, al",
61       in("dx") port,
62       in("al") value,
63       options(preserves_flags, nomem, nostack)
64    );
65}
66
67/// Wrapper function to the `inb` assembly instruction used to do the
68/// 8-bit low level port input.
69#[inline]
70pub unsafe fn inb(port: u16) -> u8 {
71    let ret: u8;
72
73    asm!(
74        "in al, dx",
75        in("dx") port,
76        out("al") ret,
77        options(preserves_flags, nomem, nostack)
78    );
79
80    ret
81}

83/// Wrapper function to the `outw` assembly instruction used to do the
84/// 16-bit low level port output.
85#[inline]
86pub unsafe fn outw(port: u16, value: u16) {

89        in("dx") port,

139/// This function is called after every `outb` and `outl` instruction as on older machines
140/// its necessary to give the PIC some time to react to commands as they might not
141/// be processed quickly.
142#[inline]
143pub unsafe fn wait() {
144    outb(0x80, 0)
145}

147/// Wrapper function to the `wrmsr` assembly instruction used
148/// to write 64 bits to msr register.
149pub unsafe fn wrmsr(msr: u32, value: u64) {
150    let low = value as u32;
151    let high = (value >> 32) as u32;
152
153    asm!("wrmsr", in("ecx") msr, in("eax") low, in("edx") high, options(nomem));
154}
155
156/// Wrapper function to the `rdmsr` assembly instruction used
157// to read 64 bits msr register.
158#[inline]
159pub unsafe fn rdmsr(msr: u32) -> u64 {
160    let (high, low): (u32, u32);
161
162    asm!("rdmsr", out("eax") low, out("edx") high, in("ecx") msr, options(nomem));
163
164    ((high as u64) << 32) | (low as u64)
165}

167#[inline]
168pub fn delay(cycles: usize) {
169    unsafe {
170        for _ in 0..cycles {
171            inb(0x80);
172        }
173    }
174}

176pub trait InOut {
177    unsafe fn port_in(port: u16) -> Self;
178    unsafe fn port_out(port: u16, value: Self);
179}
180
181impl InOut for u8 {
182    unsafe fn port_in(port: u16) -> u8 {
183        inb(port)
184    }
185
186    unsafe fn port_out(port: u16, value: u8) {
187        outb(port, value);
188    }
189}
190
191impl InOut for u16 {
192    unsafe fn port_in(port: u16) -> u16 {
193        inw(port)
194    }
195
196    unsafe fn port_out(port: u16, value: u16) {
197        outw(port, value);
198    }
199}
200
201impl InOut for u32 {
202    unsafe fn port_in(port: u16) -> u32 {
203        inl(port)
204    }
205
206    unsafe fn port_out(port: u16, value: u32) {
207        outl(port, value);
208    }
209}

213#[derive(Copy, Clone)]
214pub struct BasedPort {
215    base: u16,

218impl BasedPort {
219    pub fn new(base: u16) -> BasedPort {
220        BasedPort { base }
221    }
222
223    pub fn read_offset<V: InOut>(&self, offset: u16) -> V {
224        unsafe { V::port_in(self.base + offset) }
225    }
226
227    pub fn write_offset<V: InOut>(&mut self, offset: u16, value: V) {
228        unsafe { V::port_out(self.base + offset, value) }
229    }

262/// # Safety
263/// The caller must ensure that the swap operation does not cause any undefined behavior.
264#[inline(always)]
265pub unsafe fn swapgs() {
266    asm!("swapgs", options(nostack, preserves_flags));
267}

269// XXX: There is no variant of the {rd,wr}gsbase instructions for accessing `KERNEL_GS_BASE`, so we
270// wrap those in two `swapgs` instructions. This approach is still faster than reading/writing from
271// the `KERNEL_GS_BASE` MSR.
272
273/// Sets the inactive `GS` segment's base address.
274#[indirect]
275pub fn set_inactive_gsbase() -> fn(base: VirtAddr) {
276    unsafe fn with_wrgsbase(base: VirtAddr) {
277        swapgs();
278        wrgsbase(base);
279        swapgs();
280    }
281
282    unsafe fn with_wrmsr(base: VirtAddr) {
283        wrmsr(IA32_KERNEL_GSBASE, base.as_u64());
284    }
285
286    if super::has_fsgsbase() {
287        with_wrgsbase
288    } else {
289        with_wrmsr
290    }
291}
292
293/// Returns the inactive `GS` segment's base address.
294#[indirect]
295pub fn get_inactive_gsbase() -> fn() -> VirtAddr {
296    fn with_rdgsbase() -> VirtAddr {
297        // SAFETY: The GS base is swapped back to the original value after the read.
298        unsafe { swapgs() };
299        let base = rdgsbase();
300        unsafe { swapgs() };
301
302        base
303    }
304
305    fn with_rdmsr() -> VirtAddr {
306        VirtAddr::new(unsafe { rdmsr(IA32_KERNEL_GSBASE) })
307    }
308
309    if super::has_fsgsbase() {
310        with_rdgsbase
311    } else {
312        with_rdmsr
313    }
314}
