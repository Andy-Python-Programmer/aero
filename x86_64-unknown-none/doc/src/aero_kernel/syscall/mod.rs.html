37#[derive(Default)]
38pub struct ExecArgs {
39    pub inner: Vec<Box<[u8]>>,
40}
41
42impl ExecArgs {
43    pub fn push(&mut self, arg: &[u8]) {
44        self.inner.push(arg.into());
45    }
46
47    pub fn extend(&mut self, args: &[Box<[u8]>]) {
48        for arg in args {
49            self.push(arg);
50        }
51    }
52
53    pub fn push_into_stack(&self, stack: &mut StackHelper) -> Vec<u64> {
54        let mut tops = Vec::with_capacity(self.inner.len());
55
56        for slice in self.inner.iter() {
57            unsafe {
58                stack.write(0u8);
59                stack.write_bytes(slice);
60            }
61
62            tops.push(stack.top());
63        }
64
65        tops
66    }
67}

69pub fn exec_args_from_slice(args: usize, size: usize) -> ExecArgs {
70    // NOTE: Arguments must be moved into kernel space before we utilize them.
71    //
72    // struct SliceReference {
73    //    ptr: *const usize,
74    //    len: usize,
75    // }
76    let data = args as *const [usize; 2];
77    let slice = unsafe { core::slice::from_raw_parts(data, size) };
78
79    // todo(andy): use `with_capacity` to avoid reallocation.
80    let mut result = Vec::new();
81
82    for inner in slice {
83        let mut boxed = Box::new_uninit_slice(inner[1]);
84        let ptr = inner[0] as *const MaybeUninit<u8>;
85
86        unsafe {
87            boxed.as_mut_ptr().copy_from(ptr, inner[1]);
88
89            let inner_slice = boxed.assume_init();
90            result.push(inner_slice);
91        }
92    }
93
94    ExecArgs { inner: result }
95}

107pub(super) struct SysLog {
108    name: &'static str,
109    /// The result of the syscall.
110    result: Option<Result<usize, SyscallError>>,
111    /// The formatted argument values.
112    args: Vec<String>,
113}
114
115impl SysLog {
116    pub fn new(name: &'static str) -> Self {
117        Self {
118            name,
119            result: None,
120            args: Vec::new(),
121        }
122    }
123
124    pub fn add_argument<T>(mut self, value: T) -> Self
125    where
126        T: core::fmt::Display,
127    {
128        self.args.push(alloc::format!("{value}"));
129        self
130    }
131
132    pub fn add_argument_dbg<T>(mut self, value: T) -> Self
133    where
134        T: core::fmt::Debug,
135    {
136        self.args.push(alloc::format!("{value:?}"));
137        self
138    }
139
140    pub fn set_result(mut self, result: Result<usize, SyscallError>) -> Self {
141        self.result = Some(result);
142        self
143    }
144
145    pub fn flush(self) {
146        let mut result = String::new();
147
148        if self.result.unwrap().is_ok() {
149            result.push_str("\x1b[1;32m");
150        } else {
151            result.push_str("\x1b[1;31m");
152        }
153
154        result.push_str(alloc::format!("{}\x1b[0m(", self.name).as_str());
155
156        for (i, e) in self.args.iter().enumerate() {
157            if i != 0 {
158                result.push_str(", ");
159            }
160
161            result.push_str(e);
162        }
163
164        result.push_str(alloc::format!(") = {:?}", self.result.unwrap()).as_str());
165        log::trace!("{result}");
166
167        if self.result.unwrap().is_err() {
168            crate::unwind::unwind_stack_trace();
169        }
170    }
171}

173pub fn generic_do_syscall(
174    a: usize,
175    b: usize,
176    c: usize,
177    d: usize,
178    e: usize,
179    f: usize,
180    g: usize,
181) -> usize {
182    let result = match a {
183        SYS_EXIT => process::exit(b),
184        SYS_SHUTDOWN => process::shutdown(),
185        SYS_FORK => process::fork(),
186        SYS_MMAP => process::mmap(b, c, d, e, f, g),
187        SYS_MUNMAP => process::munmap(b, c),
188        SYS_MPROTECT => process::mprotect(b, c, d),
189        SYS_EXEC => process::exec(b, c, d, e, f, g),
190        SYS_LOG => process::log(b, c),
191        SYS_UNAME => process::uname(b),
192        SYS_WAITPID => process::waitpid(b, c, d),
193        SYS_GETPID => process::getpid(),
194        SYS_GETPPID => process::getppid(),
195        SYS_GETTID => process::gettid(),
196        SYS_GETHOSTNAME => process::gethostname(b, c),
197        SYS_SETHOSTNAME => process::sethostname(b, c),
198        SYS_INFO => process::info(b),
199        SYS_SIGACTION => process::sigaction(b, c, d, e),
200        SYS_SIGPROCMASK => process::sigprocmask(b, c, d),
201        SYS_CLONE => process::clone(b, c),
202        SYS_KILL => process::kill(b, c),
203        SYS_BACKTRACE => process::backtrace(),
204        SYS_TRACE => process::trace(),
205        SYS_SETPGID => process::setpgid(b, c),
206        SYS_SETSID => process::setsid(),
207        SYS_GETPGID => process::getpgid(b),
208
209        SYS_READ => fs::read(b, c, d),
210        SYS_OPEN => fs::open(b, c, d, e, f),
211        SYS_CLOSE => fs::close(b),
212        SYS_WRITE => fs::write(b, c, d),
213        SYS_GETDENTS => fs::getdents(b, c, d),
214        SYS_GETCWD => fs::getcwd(b, c),
215        SYS_CHDIR => fs::chdir(b, c, d),
216        SYS_MKDIR_AT => fs::mkdirat(b, c, d),
217        SYS_RMDIR => fs::rmdir(b, c),
218        SYS_IOCTL => fs::ioctl(b, c, d),
219        SYS_SEEK => fs::seek(b, c, d),
220        SYS_ACCESS => fs::access(b, c, d, e, f),
221        SYS_PIPE => fs::pipe(b, c),
222        SYS_UNLINK => fs::unlink(b, c, d, e),
223        SYS_DUP => fs::dup(b, c),
224        SYS_DUP2 => fs::dup2(b, c, d),
225        SYS_FCNTL => fs::fcntl(b, c, d),
226        SYS_STAT => fs::stat(b, c, d),
227        SYS_FSTAT => fs::fstat(b, c, d, e, f),
228        SYS_READ_LINK => fs::read_link(b, c, d, e),
229        SYS_EVENT_FD => fs::event_fd(b, c),
230        SYS_LINK => fs::link(b, c, d, e),
231        SYS_POLL => fs::poll(b, c, d, e),
232        SYS_RENAME => fs::rename(b, c, d, e),
233        SYS_SYMLINK_AT => fs::symlink(b, c, d, e, f),
234
235        // epoll calls:
236        SYS_EPOLL_CREATE => fs::epoll_create(b),
237        SYS_EPOLL_CTL => fs::epoll_ctl(b, c, d, e),
238        SYS_EPOLL_PWAIT => fs::epoll_pwait(b, c, d, e, f),
239
240        SYS_SOCKET => net::socket(b, c, d),
241        SYS_BIND => net::bind(b, c, d),
242        SYS_CONNECT => net::connect(b, c, d),
243        SYS_LISTEN => net::listen(b, c),
244        SYS_ACCEPT => net::accept(b, c, d),
245        SYS_SOCK_RECV => net::sock_recv(b, c, d),
246        SYS_SOCK_SEND => net::sock_send(b, c, d),
247        SYS_SOCKET_PAIR => net::socket_pair(b, c, d, e),
248        SYS_SOCK_SHUTDOWN => net::shutdown(b, c),
249        SYS_GETPEERNAME => net::get_peername(b, c, d),
250        SYS_GETSOCKNAME => net::get_sockname(b, c, d),
251        SYS_SETSOCKOPT => net::setopt(a, b, c, d, e),
252
253        SYS_GETTIME => time::gettime(b, c),
254        SYS_SLEEP => time::sleep(b),
255
256        SYS_SETITIMER => time::setitimer(b, c, d),
257        SYS_GETITIMER => time::getitimer(b, c),
258
259        SYS_IPC_SEND => ipc::send(b, c, d),
260        SYS_IPC_RECV => ipc::recv(b, c, d, e),
261        SYS_IPC_DISCOVER_ROOT => ipc::discover_root(),
262        SYS_IPC_BECOME_ROOT => ipc::become_root(),
263
264        SYS_FUTEX_WAIT => futex::wait(b, c, d),
265        SYS_FUTEX_WAKE => futex::wake(b),
266
267        // Syscall aliases (this should be handled in aero_syscall)
268        SYS_MKDIR => fs::mkdirat(aero_syscall::AT_FDCWD as _, b, c),
269
270        SYS_DEBUG => tag_memory(b, c, d, e),
271
272        _ => {
273            log::error!("invalid syscall: {:#x}", a);
274            Err(SyscallError::ENOSYS)
275        }
276    };
277
278    aero_syscall::syscall_result_as_usize(result)
279}

281#[syscall]
282pub fn tag_memory(ptr: *const u8, size: usize, tag: &str) -> Result<usize, SyscallError> {
283    use crate::userland::scheduler;
284    use alloc::string::ToString;
285
286    let addr = ptr as usize;
287
288    let thread = scheduler::current_thread();
289    thread
290        .mem_tags
291        .lock()
292        .insert(addr..(addr + size), tag.to_string());
293
294    Ok(0)
295}
