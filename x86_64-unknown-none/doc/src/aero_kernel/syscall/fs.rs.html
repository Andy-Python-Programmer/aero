37#[derive(Debug, Copy, Clone)]

40impl FileDescriptor {
41    /// Returns the file handle associated with this file descriptor.
42    ///
43    /// ## Errors
44    /// * `EBADFD`: The file descriptor is not a valid open file descriptor.
45    pub fn handle(&self) -> aero_syscall::Result<Arc<FileHandle>> {
46        scheduler::current_thread()
47            .file_table
48            .get_handle(self.0)
49            .ok_or(SyscallError::EBADFD)
50    }
51}

53impl fmt::Display for FileDescriptor {
54    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
55        if let Ok(file_handle) = self.handle() {
56            let path = file_handle.inode.absolute_path();
57            write!(f, "{{ {} -> {} }}", self.0, path)
58        } else {
59            // invalid file descriptor
60            write!(f, "{{ {} -> INVALID }}", self.0)
61        }
62    }
63}

104#[syscall]
105pub fn open(fd: usize, path: &Path, flags: usize, _mode: usize) -> Result<usize, SyscallError> {
106    let current_thread = scheduler::current_thread();
107    let at = match fd as isize {
108        AT_FDCWD if !path.is_absolute() => current_thread.cwd_dirent(),
109        _ if !path.is_absolute() => {
110            let ent = FileDescriptor::from_usize(fd).handle()?.inode.clone();
111            assert!(ent.inode().metadata()?.is_directory());
112            ent
113        }
114        _ => fs::root_dir().clone(),
115    };
116
117    let mut flags = OpenFlags::from_bits(flags).ok_or(SyscallError::EINVAL)?;
118
119    if !flags.intersects(OpenFlags::O_RDONLY | OpenFlags::O_RDWR | OpenFlags::O_WRONLY) {
120        flags.insert(OpenFlags::O_RDONLY);
121    }
122
123    let mut lookup_mode = LookupMode::None;
124
125    if flags.contains(OpenFlags::O_CREAT) {
126        lookup_mode = LookupMode::Create;
127    }
128
129    let inode = fs::lookup_path_with(at, path, lookup_mode, true)?;
130
131    if flags.contains(OpenFlags::O_DIRECTORY) && !inode.inode().metadata()?.is_directory() {
132        return Err(SyscallError::ENOTDIR);
133    }
134
135    if flags.contains(OpenFlags::O_TRUNC) {
136        inode.inode().truncate(0)?;
137    }
138
139    Ok(current_thread.file_table.open_file(inode.clone(), flags)?)
140}

168        .current_task()
169        .file_table

180#[syscall]
181pub fn chdir(fd: usize, path: &Path) -> Result<usize, SyscallError> {
182    let current_thread = scheduler::current_thread();
183    let at = match fd as isize {
184        AT_FDCWD if !path.is_absolute() => current_thread.cwd_dirent(),
185        _ if !path.is_absolute() => {
186            let ent = FileDescriptor::from_usize(fd).handle()?.inode.clone();
187            assert!(ent.inode().metadata()?.is_directory());
188            ent
189        }
190        _ => fs::root_dir().clone(),
191    };
192
193    if path.is_empty() {
194        current_thread.set_cwd(at);
195        return Ok(0);
196    }
197
198    let ent = fs::lookup_path_with(at, path, LookupMode::None, true)?;
199    if !ent.inode().metadata()?.is_directory() {
200        return Err(SyscallError::ENOTDIR);
201    }
202
203    current_thread.set_cwd(ent);
204    Ok(0)
205}

207#[syscall]
208pub fn mkdirat(dfd: usize, path: &Path) -> Result<usize, SyscallError> {
209    // NOTE: If the pathname given in pathname is relative, then it is interpreted
210    // relative to the directory referred to by the file descriptor (rather than relative
211    // to the current working directory of the calling task, as is done by mkdir() for a
212    // relative pathname).
213    let (parent_inode, child) = if path.is_absolute() {
214        let (path, child) = path.parent_and_basename();
215        (fs::lookup_path(path)?.inode(), child)
216    } else {
217        // If pathname is relative and fd is the special value AT_FDCWD, then
218        // pathname is interpreted relative to the current working directory of the
219        // calling task.
220        if dfd as isize == aero_syscall::AT_FDCWD {
221            let cwd = scheduler::get_scheduler().current_task().cwd_dirent();
222            (cwd.inode(), path.as_str())
223        } else {
224            let handle = scheduler::get_scheduler()
225                .current_task()
226                .file_table
227                .get_handle(dfd)
228                .ok_or(SyscallError::EBADFD)?;
229
230            (handle.inode(), path.as_str())
231        }
232    };
233
234    if !parent_inode.metadata()?.is_directory() {
235        // A component of path is not a directory.
236        return Err(SyscallError::ENOTDIR);
237    }
238
239    if ["", ".", ".."].contains(&path.as_str()) {
240        // Cannot create a directory with a name of "", ".", or "..".
241        return Err(SyscallError::EEXIST);
242    }
243
244    parent_inode.mkdir(child)?;
245    Ok(0x00)
246}

248#[syscall]
249pub fn rmdir(path: &Path) -> Result<usize, SyscallError> {
250    let (_, child) = path.parent_and_basename();
251    let inode = fs::lookup_path(path)?;
252
253    if !inode.inode().metadata()?.is_directory() {
254        // ENOTDIR: A component used as a directory in pathname, is not in fact,
255        // a directory.
256        return Err(SyscallError::ENOTDIR);
257    }
258
259    inode.inode().rmdir(child)?;
260    inode.drop_from_cache();
261    Ok(0x00)
262}

277#[syscall]
278pub fn ioctl(fd: FileDescriptor, command: usize, argument: usize) -> Result<usize, SyscallError> {
279    let handle = fd.handle()?;
280
281    match command {
282        // Sets the close-on-exec file descriptor flag. This is equivalent
283        // to `fcntl(fd, F_SETFD, FD_CLOEXEC)`
284        FIOCLEX => {
285            let flags = handle.flags();
286            handle.set_flags(flags | OpenFlags::O_CLOEXEC);
287            Ok(0)
288        }
289
290        FIONBIO => {
291            let flags = handle.flags();
292            handle.set_flags(flags | OpenFlags::O_NONBLOCK);
293            Ok(0)
294        }
295
296        // Handle file specific ioctl:
297        _ => Ok(handle.inode().ioctl(command, argument)?),
298    }
299}

307#[syscall]
308pub fn pipe(fds: &mut [i32; 2], flags: usize) -> Result<usize, SyscallError> {
309    let flags = OpenFlags::from_bits(flags).ok_or(SyscallError::EINVAL)?;
310    let pipe = Pipe::new();
311
312    let entry = DirEntry::from_inode(pipe, String::from("<pipe>"));
313
314    let flags_1 = OpenFlags::O_RDONLY | flags;
315    let flags_2 = OpenFlags::O_WRONLY | flags;
316
317    let current_task = scheduler::get_scheduler().current_task();
318
319    let fd1 = current_task.file_table.open_file(entry.clone(), flags_1)?;
320    let fd2 = current_task.file_table.open_file(entry, flags_2);
321
322    // If there was an error in opening the second file descriptor,
323    // then close the first file descriptor. Just to be safe :^)
324    let fd2 = match fd2 {
325        Err(err) => {
326            current_task.file_table.close_file(fd1);
327            return Err(err.into());
328        }
329
330        Ok(fd2) => fd2,
331    };
332
333    fds[0] = fd1 as i32;
334    fds[1] = fd2 as i32;
335
336    Ok(0x00)
337}

364    let at = match fd as isize {
365        AT_FDCWD if !path.is_absolute() => scheduler::current_thread().cwd_dirent(),

378const SETFL_MASK: OpenFlags = OpenFlags::from_bits_truncate(
379    OpenFlags::O_APPEND.bits()
380        | OpenFlags::O_NONBLOCK.bits()
381        // | OpenFlags::O_NDELAY.bits()
382        | OpenFlags::O_DIRECT.bits()

386#[syscall]
387pub fn fcntl(fd: FileDescriptor, command: usize, arg: usize) -> Result<usize, SyscallError> {
388    let handle = fd.handle()?;
389
390    match command {
391        // F_DUPFD_CLOEXEC and F_DUPFD:
392        //
393        // Duplicate the file descriptor `fd` using the lowest-numbered
394        // available file descriptor greater than or equal to `arg`. This is
395        // different from `dup2(2)`, which uses exactly the file descriptor
396        // specified.
397        //
398        // F_DUPFD_CLOEXEC additionally sets the close-on-exec flag for the duplicate
399        // file descriptor.
400        aero_syscall::prelude::F_DUPFD => scheduler::current_thread().file_table.duplicate(
401            fd.into(),
402            DuplicateHint::GreatorOrEqual(arg),
403            handle.flags(),
404        ),
405
406        aero_syscall::prelude::F_DUPFD_CLOEXEC => scheduler::current_thread().file_table.duplicate(
407            fd.into(),
408            DuplicateHint::GreatorOrEqual(arg),
409            handle.flags() | OpenFlags::O_CLOEXEC,
410        ),
411
412        // Get the value of file descriptor flags.
413        aero_syscall::prelude::F_GETFD => {
414            let flags = handle.flags();
415            let mut result = FdFlags::empty();
416
417            if flags.contains(OpenFlags::O_CLOEXEC) {
418                result.insert(FdFlags::CLOEXEC);
419            }
420
421            Ok(result.bits())
422        }
423
424        // Set the value of file descriptor flags:
425        aero_syscall::prelude::F_SETFD => {
426            let mut flags = handle.flags();
427            let fd_flags = FdFlags::from_bits_truncate(arg);
428
429            if fd_flags.contains(FdFlags::CLOEXEC) {
430                flags.insert(OpenFlags::O_CLOEXEC);
431            } else {
432                flags.remove(OpenFlags::O_CLOEXEC);
433            }
434
435            handle.set_flags(flags);
436            Ok(0)
437        }
438
439        // Get the value of file status flags:
440        aero_syscall::prelude::F_GETFL => Ok(handle.flags().bits()),
441
442        aero_syscall::prelude::F_SETFL => {
443            let flags = OpenFlags::from_bits_truncate(arg);
444            let old_flags = handle.flags();
445            handle.set_flags((flags & SETFL_MASK) | (old_flags & !SETFL_MASK));
446
447            Ok(0)
448        }
449
450        aero_syscall::prelude::F_SETLKW | aero_syscall::prelude::F_SETLK => {
451            log::warn!("fcntl: F_SETLKW,F_SETLK are a stub!");
452            Ok(0)
453        }
454
455        _ => {
456            log::error!("fcntl: unknown command {command}");
457            Ok(0)
458        }
459    }
460}

462#[syscall]
463pub fn fstat(fd: usize, path: &Path, flags: usize, stat: &mut Stat) -> Result<usize, SyscallError> {
464    let at = match fd as isize {
465        AT_FDCWD if !path.is_absolute() => scheduler::current_thread().cwd_dirent(),
466        _ if !path.is_absolute() => FileDescriptor::from_usize(fd).handle()?.inode.clone(),
467        _ => fs::root_dir().clone(),
468    };
469
470    // TODO: derive(SysArg) for bitflags.
471    let flags = AtFlags::from_bits(flags).ok_or(SyscallError::EINVAL)?;
472    assert!(!flags.intersects(AtFlags::EACCESS | AtFlags::REMOVEDIR));
473
474    if path.is_empty() {
475        if !flags.contains(AtFlags::EMPTY_PATH) {
476            return Err(SyscallError::EINVAL);
477        }
478
479        *stat = at.inode().stat()?;
480        return Ok(0);
481    }
482
483    let resolve_last = !flags.contains(AtFlags::SYMLINK_NOFOLLOW);
484    let ent = fs::lookup_path_with(at, path, LookupMode::None, resolve_last)?;
485    *stat = ent.inode().stat()?;
486    Ok(0)
487}

496#[syscall]
497pub fn read_link(path: &Path, buffer: &mut [u8]) -> Result<usize, SyscallError> {
498    // XXX: lookup_path with automatically resolve the link.
499    let cwd = if !path.is_absolute() {
500        scheduler::current_thread().cwd_dirent()
501    } else {
502        fs::root_dir().clone()
503    };
504
505    let file = fs::lookup_path_with(cwd.clone(), path, LookupMode::None, false)?.inode();
506    if !file.metadata()?.is_symlink() {
507        return Err(SyscallError::EINVAL);
508    }
509
510    let resolved_path = file.resolve_link()?;
511    let resolved_path = if resolved_path.is_absolute() {
512        resolved_path
513    } else {
514        cwd.absolute_path().join(resolved_path)
515    };
516
517    let size = core::cmp::min(resolved_path.as_str().len(), buffer.len());
518
519    log::warn!("Orig: {path:?} -> {resolved_path}");
520
521    buffer[..size].copy_from_slice(&resolved_path.as_bytes()[..size]);
522    Ok(size)
523}

539/// Used to add, modify, or remove entries in the interest list of the
540/// epoll instance referred to by the file descriptor. It requests that
541/// the operation be performed for the target file descriptor.
542#[syscall]
543pub fn epoll_ctl(
544    epfd: FileDescriptor,
545    mode: usize,
546    fd: usize,
547    event: &mut EPollEvent,
548) -> Result<usize, SyscallError> {
549    let epfd = epfd.handle()?;
550
551    let epoll = epfd
552        .inode()
553        .downcast_arc::<EPoll>()
554        .ok_or(SyscallError::EINVAL)?;
555
556    match mode {
557        EPOLL_CTL_ADD => {
558            epoll.add_event(fd, *event)?;
559            Ok(0)
560        }
561
562        EPOLL_CTL_DEL => {
563            epoll.remove_event(fd)?;
564            Ok(0)
565        }
566
567        EPOLL_CTL_MOD => {
568            epoll.update_event(fd, *event)?;
569            Ok(0)
570        }
571
572        _ => unreachable!("epoll_ctl: unknown mode {mode}"),
573    }
574}

576#[syscall]
577pub fn epoll_pwait(
578    epfd: FileDescriptor,
579    event: &mut [EPollEvent],
580    timeout: usize,
581    sigmask: usize,
582) -> Result<usize, SyscallError> {
583    let max_events = event.len();
584
585    let current_task = scheduler::get_scheduler().current_task();
586    let signals = current_task.signals();
587
588    let epfd = epfd.handle()?;
589    let epfd = epfd
590        .inode()
591        .downcast_arc::<EPoll>()
592        .ok_or(SyscallError::EINVAL)?;
593
594    let mut old_mask = 0;
595
596    // Update the signal mask.
597    signals.set_mask(SigProcMask::Set, Some(sigmask as u64), Some(&mut old_mask));
598
599    let result = epfd.wait(event, max_events, timeout)?;
600
601    // Restore the original signal mask.
602    signals.set_mask(SigProcMask::Set, Some(old_mask), None);
603    Ok(result)
604}

621/// Creates a new link (also known as a hard link) to an existing
622/// file.
623#[syscall]
624pub fn link(src_path: &Path, dest_path: &Path) -> Result<usize, SyscallError> {
625    let src = fs::lookup_path(src_path)?;
626    let (dest_dir, dest_name) = dest_path.parent_and_basename();
627
628    let dest_dir = fs::lookup_path(dest_dir)?.inode();
629
630    // Cannot create a hardlink to a file on a different filesystem.
631    //
632    // SAFETY: The pointers to the file system are valid since we know that there are
633    // strong references to it.
634    //
635    // TODO: Should this be moved to the inode impl?
636    if !Weak::ptr_eq(
637        &dest_dir.weak_filesystem().unwrap(),
638        &src.inode().weak_filesystem().unwrap(),
639    ) {
640        return Err(SyscallError::EINVAL);
641    }
642
643    dest_dir.link(dest_name, src)?;
644    Ok(0)
645}

647fn do_poll(fds: &mut [PollFd], timeout: Option<&TimeSpec>) -> Result<usize, SyscallError> {
648    let current_task = scheduler::get_scheduler().current_task();
649
650    let mut poll_table = PollTable::default();
651    let mut n = 0;
652    let mut refds = alloc::vec![];
653
654    // Iterate over all the registered events and check if they are ready.
655    for (i, fd) in fds.iter_mut().enumerate() {
656        fd.revents = PollEventFlags::empty();
657
658        // TODO: If an invalid file descriptor is provided then return EBADFD. Not implemented
659        // currently, since the init process (libc?) tries to POLL on the stdout, stdin and
660        // stdout file descriptors which are currently not present.
661        //
662        // One possible solution is to open the file descriptors when the init process
663        // is a kernel process?
664        let handle = match current_task.file_table.get_handle(fd.fd as usize) {
665            Some(v) => v,
666            None => {
667                return Ok(0);
668            }
669        };
670
671        let ready: PollEventFlags = handle.inode().poll(None)?.into();
672
673        if !(ready & fd.events).is_empty() {
674            // The registered event is ready; increment the number of ready events
675            // and update revents mask for this event.
676            fd.revents = ready & fd.events;
677            n += 1;
678            continue;
679        }
680
681        // Not ready; add the event to the poll table.
682        handle.inode().poll(Some(&mut poll_table))?;
683        refds.push((handle, i));
684    }
685
686    // If all events are ready, we can return now.
687    if n > 0 {
688        return Ok(n);
689    }
690
691    // Start the timer if timeout specified, if not, we can block indefinitely.
692    if let Some(timeout) = timeout {
693        // If the timeout is zero, then we have to return without blocking.
694        if timeout.tv_nsec == 0 && timeout.tv_sec == 0 {
695            return Ok(0);
696        }
697    }
698
699    'search: loop {
700        scheduler::get_scheduler().await_io()?;
701
702        for (handle, index) in refds.iter() {
703            let pollfd = &mut fds[*index];
704            let ready: PollEventFlags = handle.inode().poll(None)?.into();
705
706            if !(ready & pollfd.events).is_empty() {
707                pollfd.revents = ready & pollfd.events;
708                break 'search Ok(1);
709            }
710        }
711    }
712}

714#[syscall]
715pub fn poll(fds: &mut [PollFd], timeout: usize, sigmask: usize) -> Result<usize, SyscallError> {
716    // Nothing to poll on.
717    if fds.is_empty() {
718        return Ok(0);
719    }
720
721    // The timeout can be NULL.
722    let timeout = if timeout != 0x00 {
723        Some(crate::utils::validate_ptr(timeout as *const TimeSpec)?)
724    } else {
725        None
726    };
727
728    let current_task = scheduler::get_scheduler().current_task();
729    let signals = current_task.signals();
730
731    let mut old_mask = 0;
732
733    // Update the signal mask.
734    signals.set_mask(SigProcMask::Set, Some(sigmask as u64), Some(&mut old_mask));
735
736    let n = do_poll(fds, timeout)?;
737
738    // Restore the original signal mask.
739    signals.set_mask(SigProcMask::Set, Some(old_mask), None);
740    Ok(n)
741}

743#[syscall]
744pub fn rename(src: &Path, dest: &Path) -> Result<usize, SyscallError> {
745    let src = fs::lookup_path(src)?;
746    let (dest, name) = {
747        let (dir, name) = dest.parent_and_basename();
748        (fs::lookup_path(dir)?, name)
749    };
750
751    dest.inode().rename(src.clone(), name)?;
752
753    cache::dcache().rehash(src.clone(), || {
754        src.set_name(name);
755        src.set_parent(dest);
756    });
757    Ok(0)
758}

760#[syscall]
761pub fn symlink(link_dirfd: usize, target: &Path, linkpath: &Path) -> Result<usize, SyscallError> {
762    // TODO(andypython): the following code is reused in a couple of places. Isolate this inside the
763    // syscall parsing for FileDescriptor with an argument as a generic specifing the allowance
764    // of this value.
765    //
766    // If the pathname given in `linkpath` is relative, then it is interpreted relative to the
767    // directory referred to by the file descriptor `link_dirfd`.
768    let at = match link_dirfd as isize {
769        AT_FDCWD if !linkpath.is_absolute() => scheduler::current_thread().cwd_dirent(),
770        _ if !linkpath.is_absolute() => FileDescriptor::from_usize(link_dirfd)
771            .handle()?
772            .inode
773            .clone(),
774        _ => fs::root_dir().clone(),
775    };
776
777    let ent = fs::lookup_path_with(at, linkpath, LookupMode::Create, false)?;
778    ent.inode().symlink(target)?;
779
780    Ok(0)
781}
