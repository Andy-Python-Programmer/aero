18//! Netlink Sockets
19//!
20//! Netlink sockets are used for inter-process communication (IPC) between both the kernel and
21//! userspace processes, and between different userspace processes, in a way similar to the Unix
22//! domain sockets.
23//!
24//! Netlink is designed and used for transferring miscellaneous networking information between the
25//! kernel space and userspace processes. Networking utilities, such as the `iproute2` family use
26//! Netlink to communicate with the kernel from userspace.
27

42struct NetlinkBuilder {
43    buffer: Vec<u8>,
44}
45
46impl NetlinkBuilder {
47    fn new() -> Self {
48        // crate::scheduler::get_scheduler().for_each_task(|task| {
49        //     if task
50        //         .executable
51        //         .lock()
52        //         .as_ref()
53        //         .map(|x| x.absolute_path_str().contains("kit"))
54        //         .unwrap_or_default()
55        //     {
56        //         task.enable_systrace();
57        //     }
58        // });
59
60        Self { buffer: Vec::new() }
61    }
62
63    fn header(&mut self, header: &netlink::nlmsghdr) {
64        self.buffer.extend_from_slice(unsafe {
65            core::slice::from_raw_parts(
66                &header as *const _ as *const u8,
67                core::mem::size_of::<netlink::nlmsghdr>(),
68            )
69        });
70
71        self.buffer_align();
72    }
73
74    fn message(&mut self, message: &netlink::rtmsg) {
75        self.buffer.extend_from_slice(unsafe {
76            core::slice::from_raw_parts(
77                &message as *const _ as *const u8,
78                core::mem::size_of::<netlink::rtmsg>(),
79            )
80        });
81
82        self.buffer_align();
83    }
84
85    fn rtattr<T>(&mut self, ty: RtAttrType, data: T) {
86        let rta_len = netlink::rta_length(core::mem::size_of::<T>() as u32);
87
88        let attr = netlink::rtattr {
89            rta_len: rta_len.try_into().unwrap(),
90            rta_type: ty,
91        };
92
93        let padding = rta_len as usize - core::mem::size_of::<T>();
94
95        self.buffer.extend_from_slice(unsafe {
96            core::slice::from_raw_parts(
97                &attr as *const _ as *const u8,
98                core::mem::size_of::<netlink::rtattr>(),
99            )
100        });
101
102        self.buffer.extend_from_slice(unsafe {
103            core::slice::from_raw_parts(&data as *const _ as *const u8, core::mem::size_of::<T>())
104        });
105
106        self.buffer.resize(self.buffer.len() + padding, 0);
107        self.buffer_align();
108    }
109
110    /// Aligns the buffer to the netlink message alignment.
111    fn buffer_align(&mut self) {
112        let aligned_len = netlink::nlmsg_align(self.buffer.len() as u32);
113
114        // Resize will not truncate the buffer since [`nlmsg_align`]` only rounds up.
115        self.buffer.resize(aligned_len as usize, 0);
116    }
117
118    fn build(self) -> Vec<u8> {
119        let mut buffer = self.buffer;
120
121        let msg_len = buffer.len();
122        let msg_hdr = unsafe { &mut *buffer.as_mut_ptr().cast::<netlink::nlmsghdr>() };
123
124        msg_hdr.nlmsg_len = msg_len as u32;
125        buffer
126    }
127}
128
129pub struct NetLinkSocket {
130    recv_queue: Mutex<Vec<Vec<u8>>>,
131    recv_wq: WaitQueue,
132}
133
134impl NetLinkSocket {
135    pub fn new() -> Arc<Self> {
136        Arc::new(Self {
137            recv_queue: Mutex::new(Vec::new()),
138            recv_wq: WaitQueue::new(),
139        })
140    }
141
142    fn validate_message<'a, T>(header: &'a netlink::nlmsghdr, payload: &'a [u8]) -> &'a T {
143        let hdr_len = core::mem::size_of::<netlink::nlmsghdr>() as u32;
144        let msg_len = core::mem::size_of::<T>() as u32;
145
146        // TODO(andypython): send an error message instead of panicking.
147        assert!(header.nlmsg_len == hdr_len + msg_len);
148
149        // FIXME(andypython): use bytemuck to cast the payload to T.
150        unsafe { &*payload.as_ptr().cast::<T>() }
151    }
152
153    fn send_route_packet(&self, header: &netlink::nlmsghdr) {
154        let mut builder = NetlinkBuilder::new();
155
156        builder.header(&netlink::nlmsghdr {
157            nlmsg_type: MessageType::RtmNewRoute,
158            nlmsg_flags: MessageFlags::MULTI,
159            nlmsg_seq: header.nlmsg_seq,
160            nlmsg_pid: 0,
161            nlmsg_len: 0,
162        });
163
164        builder.message(&netlink::rtmsg {
165            rtm_family: AF_INET as u8,
166            rtm_dst_len: 0, // FIXME
167            rtm_src_len: 0,
168            rtm_tos: 0,
169            rtm_table: netlink::RT_TABLE_MAIN,
170
171            rtm_protocol: 0,
172            rtm_scope: 0,
173            rtm_type: 0,
174            rtm_flags: 0,
175        });
176
177        builder.rtattr(RtAttrType::Table, netlink::RT_TABLE_MAIN);
178
179        // Qemu SLIRP
180        builder.rtattr(RtAttrType::Dst, Ipv4Addr::new(10, 0, 2, 15));
181        builder.rtattr(RtAttrType::Gateway, Ipv4Addr::new(10, 0, 2, 2));
182
183        self.recv_queue.lock().push(builder.build());
184        self.recv_wq.notify();
185    }
186
187    fn get_route(&self, header: &netlink::nlmsghdr, payload: &[u8]) {
188        assert!(header
189            .nlmsg_flags
190            .contains(MessageFlags::REQUEST | MessageFlags::DUMP));
191
192        let payload = Self::validate_message::<netlink::rtgenmsg>(header, payload);
193        let rtgen_family = payload.rtgen_family as u32;
194
195        assert!(rtgen_family == AF_UNSPEC || rtgen_family == AF_NETLINK);
196
197        self.send_route_packet(header);
198    }
199}

222    fn recv(
223        &self,
224        message_hdr: &mut MessageHeader,
225        flags: socket::MessageFlags,
226    ) -> fs::Result<usize> {
227        // FIXME(andypython): All of the message header and iovec logic should be moved to
228        // syscall::net::recvmsg() instead.
229
230        if let Some(addr) = message_hdr.name_mut::<netlink::sockaddr_nl>() {
231            *addr = netlink::sockaddr_nl {
232                nl_family: AF_NETLINK,
233                nl_pad: 0,
234                nl_pid: 0,
235                nl_groups: 0,
236            };
237        }
238
239        let mut queue = self
240            .recv_wq
241            .block_on(&self.recv_queue, |queue| !queue.is_empty())?;
242
243        let mut bytes_copied = 0;
244        dbg!(message_hdr.iovecs_mut());
245        let mut iovecs = message_hdr.iovecs_mut().to_vec();
246
247        while let Some(data) = queue.pop() {
248            if let Some((index, ref mut iovec)) = iovecs
249                .iter_mut()
250                .enumerate()
251                .find(|(_, iovec)| iovec.len() >= data.len())
252            {
253                let iovec = iovec.as_slice_mut();
254                assert!(iovec.len() >= data.len());
255
256                let copy = core::cmp::min(iovec.len(), data.len());
257                iovec[..copy].copy_from_slice(&data[..copy]);
258
259                bytes_copied += copy;
260                iovecs.remove(index);
261            } else if flags.contains(socket::MessageFlags::TRUNC) && bytes_copied == 0 {
262                message_hdr.flags = socket::MessageFlags::TRUNC.bits() as i32;
263                return Ok(data.len());
264            } else {
265                unimplemented!()
266            }
267        }
268
269        Ok(bytes_copied)
270    }
271
272    fn send(
273        &self,
274        message_hdr: &mut MessageHeader,
275        flags: socket::MessageFlags,
276    ) -> fs::Result<usize> {
277        log::warn!("netlink::send(flags={flags:?})");
278
279        // FIXME(andypython): figure out the message header stuff...
280        let data = message_hdr
281            .iovecs()
282            .iter()
283            .flat_map(|e| e.as_slice())
284            .copied()
285            .collect::<Vec<_>>();
286
287        let hdr_size = core::mem::size_of::<netlink::nlmsghdr>();
288
289        let mut offset = 0;
290
291        while offset + hdr_size <= data.len() {
292            let header = unsafe { &*(data.as_ptr().cast::<netlink::nlmsghdr>().byte_add(offset)) };
293            let payload = &data[offset + hdr_size..];
294
295            match header.nlmsg_type {
296                MessageType::Done => break,
297                MessageType::Error => {
298                    unimplemented!("netlink::send: error message received");
299                }
300
301                MessageType::RtmGetRoute => self.get_route(header, payload),
302
303                ty => unimplemented!("netlink::send: unknown message type {ty:?}"),
304            }
305
306            offset += header.nlmsg_len as usize;
307        }
308
309        Ok(data.len())
310    }
