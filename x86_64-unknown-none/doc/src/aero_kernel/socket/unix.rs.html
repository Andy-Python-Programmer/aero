39fn path_from_unix_sock(address: &SocketAddrUnix) -> fs::Result<&Path> {
40    // The abstract namespace socket allows the creation of a socket
41    // connection which does not require a path to be created.
42    let abstract_namespaced = address.path[0] == 0;
43    assert!(!abstract_namespaced);
44    assert!(address.path[1] != 0, "unnamed UNIX socket");
45
46    let path_len = address
47        .path
48        .iter()
49        .position(|&c| c == 0)
50        .unwrap_or(address.path.len());
51
52    let path_str = core::str::from_utf8(&address.path[..path_len])
53        .ok()
54        .ok_or(FileSystemError::InvalidPath)?;
55
56    Ok(Path::new(path_str))
57}

59#[derive(Debug, Default)]
60pub struct Message {
61    data: Vec<u8>,
62    // TODO: Keep track of the sender of the message here?
63}
64
65impl Message {
66    pub fn new(data: Vec<u8>) -> Self {
67        Self { data }
68    }
69}

71#[derive(Default)]
72pub struct MessageQueue {
73    messages: VecDeque<Message>,
74}
75
76impl MessageQueue {
77    pub fn is_empty(&self) -> bool {
78        self.messages.is_empty()
79    }
80
81    pub fn read(&mut self, buffer: &mut [u8]) -> usize {
82        if let Some(message) = self.messages.front_mut() {
83            let message_len = message.data.len();
84            let size = core::cmp::min(buffer.len(), message_len);
85
86            buffer[..size].copy_from_slice(&message.data[..size]);
87
88            if size < message_len {
89                message.data.drain(..size);
90                return size;
91            }
92
93            self.messages.pop_front();
94            size
95        } else {
96            unreachable!("MessageQueue::read() called when queue is empty");
97        }
98    }
99
100    pub fn write(&mut self, buffer: &[u8]) {
101        let message = Message::new(buffer.to_vec());
102        self.messages.push_back(message);
103    }
104}

106pub struct AcceptQueue {
107    sockets: VecDeque<Arc<UnixSocket>>,
108    backlog: usize,
109}
110
111impl AcceptQueue {
112    /// # Parameters
113    /// * `backlog`: The maximum number of pending connections that the queue can hold.
114    pub fn new(backlog: usize) -> Self {
115        Self {
116            sockets: VecDeque::with_capacity(backlog),
117            backlog,
118        }
119    }
120
121    /// Returns `true` if the queue contains no pending connections.
122    pub fn is_empty(&self) -> bool {
123        self.sockets.is_empty()
124    }
125
126    /// Adds the given socket to the queue. Returns `EAGAIN` if the
127    /// queue is full.
128    pub fn push(&mut self, socket: Arc<UnixSocket>) -> Result<(), SyscallError> {
129        if self.backlog == self.sockets.len() {
130            return Err(SyscallError::EAGAIN);
131        }
132
133        self.sockets.push_back(socket);
134        Ok(())
135    }
136
137    /// Removes the first pending connection from the queue and
138    /// returns it, or [`None`] if it is empty.
139    pub fn pop(&mut self) -> Option<Arc<UnixSocket>> {
140        self.sockets.pop_front()
141    }
142
143    /// Updates the maximum number of pending connections that the
144    /// queue can hold. Returns `EINVAL` if the new backlog is smaller
145    /// than the current number of pending connections.
146    pub fn set_backlog(&mut self, backlog: usize) -> Result<(), SyscallError> {
147        if backlog < self.sockets.len() {
148            return Err(SyscallError::EINVAL);
149        }
150
151        self.backlog = backlog;
152        Ok(())
153    }
154}

156#[derive(Default)]
157enum UnixSocketState {
158    /// The socket is not connected.
159    #[default]
160    Disconnected,
161
162    /// The socket is listening for new connections.
163    Listening(AcceptQueue),
164
165    /// The socket has connected to a peer.
166    Connected(Arc<UnixSocket>),
167}
168
169impl UnixSocketState {
170    /// Returns `true` if the socket is connected.
171    fn is_connected(&self) -> bool {
172        matches!(self, Self::Connected(_))
173    }
174
175    fn queue(&mut self) -> Option<&mut AcceptQueue> {
176        match self {
177            Self::Listening(q) => Some(q),
178            _ => None,
179        }
180    }
181}

184struct UnixSocketInner {
185    /// The address that the socket has been bound to.
186    address: Option<SocketAddrUnix>,

188    state: UnixSocketState,

191pub struct UnixSocket {
192    inner: Mutex<UnixSocketInner>,
193    buffer: Mutex<MessageQueue>,
194    wq: WaitQueue,
195    weak: Weak<UnixSocket>,
196}
197
198impl UnixSocket {
199    pub fn new() -> Arc<Self> {
200        Arc::new_cyclic(|weak| Self {
201            inner: Mutex::new(UnixSocketInner::default()),
202
203            buffer: Mutex::new(MessageQueue::default()),
204            wq: WaitQueue::new(),
205            weak: weak.clone(),
206        })
207    }
208
209    pub fn connect_pair(a: &DirCacheItem, b: &DirCacheItem) -> fs::Result<()> {
210        let a = a
211            .inode()
212            .downcast_arc::<UnixSocket>()
213            .ok_or(FileSystemError::NotSocket)?;
214
215        let b = b
216            .inode()
217            .downcast_arc::<UnixSocket>()
218            .ok_or(FileSystemError::NotSocket)?;
219
220        a.inner.lock_irq().state = UnixSocketState::Connected(b.clone());
221        b.inner.lock_irq().state = UnixSocketState::Connected(a);
222        Ok(())
223    }
224
225    pub fn sref(&self) -> Arc<Self> {
226        self.weak.upgrade().unwrap()
227    }
228}

230impl INodeInterface for UnixSocket {
231    fn metadata(&self) -> fs::Result<Metadata> {
232        Ok(Metadata {
233            id: 0,
234            file_type: FileType::Socket,
235            size: 0,
236            children_len: 0,
237        })
238    }
239
240    fn read_at(
241        &self,
242        flags: OpenFlags,
243        _offset: usize,
244        user_buffer: &mut [u8],
245    ) -> fs::Result<usize> {
246        if self.buffer.lock_irq().is_empty() && flags.is_nonblock() {
247            return Err(FileSystemError::WouldBlock);
248        }
249
250        let mut buffer = self.wq.block_on(&self.buffer, |e| !e.is_empty())?;
251
252        let read = buffer.read(user_buffer);
253        Ok(read)
254    }
255
256    fn write_at(&self, _offset: usize, buffer: &[u8]) -> fs::Result<usize> {
257        let inner = self.inner.lock_irq();
258        let peer = match inner.state {
259            UnixSocketState::Connected(ref peer) => peer,
260            _ => return Err(FileSystemError::NotConnected),
261        };
262
263        peer.buffer.lock_irq().write(buffer);
264        peer.wq.notify_all();
265
266        Ok(buffer.len())
267    }
268
269    fn listen(&self, backlog: usize) -> Result<(), SyscallError> {
270        let mut inner = self.inner.lock_irq();
271        let is_bound = inner.address.is_some();
272
273        match &mut inner.state {
274            // We cannot listen on a socket that has not been bound.
275            UnixSocketState::Disconnected if is_bound => {
276                inner.state = UnixSocketState::Listening(AcceptQueue::new(backlog));
277                Ok(())
278            }
279
280            UnixSocketState::Listening(queue) => {
281                queue.set_backlog(backlog)?;
282                Ok(())
283            }
284
285            _ => unreachable!(),
286        }
287    }
288
289    fn bind(&self, address: SocketAddrRef, _length: usize) -> fs::Result<()> {
290        let address = address.as_unix().ok_or(FileSystemError::NotSupported)?;
291        let path = path_from_unix_sock(address)?;
292
293        if fs::lookup_path(path).is_ok() {
294            return Err(FileSystemError::EntryExists);
295        }
296
297        let (parent, name) = path.parent_and_basename();
298        DirEntry::from_socket_inode(fs::lookup_path(parent)?, String::from(name), self.sref())?;
299
300        let mut inner = self.inner.lock_irq();
301        inner.address = Some(address.clone());
302
303        Ok(())
304    }
305
306    fn connect(&self, address: SocketAddrRef, _length: usize) -> fs::Result<()> {
307        let address = address.as_unix().ok_or(FileSystemError::NotSupported)?;
308        let path = path_from_unix_sock(address)?;
309        let socket = fs::lookup_path(path)?;
310
311        let target = socket
312            .inode()
313            .as_unix_socket()?
314            .downcast_arc::<UnixSocket>()
315            .ok_or(FileSystemError::NotSocket)?;
316
317        let mut itarget = target.inner.lock_irq();
318
319        let queue = match &mut itarget.state {
320            UnixSocketState::Listening(queue) => queue,
321            _ => return Err(FileSystemError::ConnectionRefused),
322        };
323
324        queue.push(self.sref()).unwrap();
325        target.wq.notify_all();
326        core::mem::drop(itarget); // release the lock
327
328        let _ = self.wq.block_on(&self.inner, |e| e.state.is_connected())?;
329        Ok(())
330    }
331
332    fn accept(&self, address: Option<(VirtAddr, &mut u32)>) -> fs::Result<Arc<UnixSocket>> {
333        let mut inner = self.wq.block_on(&self.inner, |e| {
334            e.state.queue().is_some_and(|x| !x.is_empty())
335        })?;
336
337        let queue = inner
338            .state
339            .queue()
340            .ok_or(FileSystemError::ConnectionRefused)?;
341
342        let peer = queue.pop().expect("UnixSocket::accept(): backlog is empty");
343        let sock = Self::new();
344        sock.inner.lock_irq().address.clone_from(&inner.address);
345
346        {
347            let mut sock_inner = sock.inner.lock_irq();
348            sock_inner.state = UnixSocketState::Connected(peer.clone());
349        }
350
351        {
352            let mut peer_data = peer.inner.lock_irq();
353            peer_data.state = UnixSocketState::Connected(sock.clone());
354        }
355
356        // THIS SHOULD NOT BE DONE HERE
357        if let Some((address, length)) = address {
358            let mut address = unsafe { UserRef::new(address) };
359
360            if let Some(paddr) = inner.address.as_ref() {
361                *address = paddr.clone();
362            } else {
363                *address = SocketAddrUnix::default();
364                address.family = AF_UNIX;
365            }
366
367            // THIS IS WRONG use addr.name_len() + offset_of!(sockaddr_un, path)
368            *length = core::mem::size_of::<SocketAddrUnix>() as u32;
369        }
370
371        peer.wq.notify_all();
372        Ok(sock)
373    }

375    fn recv(
376        &self,
377        fd_flags: OpenFlags,
378        header: &mut MessageHeader,
379        flags: MessageFlags,
380    ) -> fs::Result<usize> {
381        // assert!(flags.is_empty());
382
383        let inner = self.inner.lock_irq();
384
385        let peer = match &inner.state {
386            UnixSocketState::Connected(peer) => peer,
387            _ => return Err(FileSystemError::NotConnected),
388        };
389
390        if self.buffer.lock_irq().is_empty() && fd_flags.is_nonblock() {
391            return Err(FileSystemError::WouldBlock);
392        }
393
394        let mut buffer = self.wq.block_on(&self.buffer, |e| !e.is_empty())?;
395
396        if let Some(addr) = header.name_mut::<SocketAddrUnix>() {
397            *addr = peer.inner.lock_irq().address.as_ref().cloned().unwrap();
398        }
399
400        Ok(header
401            .iovecs_mut()
402            .iter_mut()
403            .map(|iovec| buffer.read(iovec.as_slice_mut()))
404            .sum::<usize>())
405    }
406
407    fn send(&self, header: &mut MessageHeader, _flags: MessageFlags) -> fs::Result<usize> {
408        // FIXME(andyython): figure out the message header stuff...
409        let data = header
410            .iovecs()
411            .iter()
412            .flat_map(|e| e.as_slice())
413            .copied()
414            .collect::<Vec<_>>();
415
416        self.write_at(0, &data)
417    }
418
419    fn poll(&self, table: Option<&mut PollTable>) -> fs::Result<PollFlags> {
420        let buffer = self.buffer.lock_irq();
421        let inner = self.inner.lock_irq();
422
423        if let Some(e) = table {
424            e.insert(&self.wq)
425        }
426
427        let mut events = PollFlags::OUT;
428
429        if let UnixSocketState::Listening(queue) = &inner.state {
430            if !queue.is_empty() {
431                events.insert(PollFlags::IN);
432                return Ok(events);
433            }
434        }
435
436        if !buffer.is_empty() {
437            events.insert(PollFlags::IN);
438        }
439
440        Ok(events)
441    }
