29use crate::mem::paging::VirtAddr;

31#[derive(Debug)]
32pub enum SocketAddr {
33    Inet(SocketAddrInet),
34    Netlink(sockaddr_nl),
35    Unix(SocketAddrUnix),
36}
37
38#[derive(Debug)]
39pub enum SocketAddrRef<'a> {
40    Unix(&'a SocketAddrUnix),
41    INet(&'a SocketAddrInet),
42    // TODO: https://docs.huihoo.com/doxygen/linux/kernel/3.7/structsockaddr__nl.html
43    Netlink,
44}
45
46impl<'a> SocketAddrRef<'a> {
47    pub fn from_family(address: VirtAddr, family: u32) -> Result<Self> {
48        match family {
49            AF_UNIX => Ok(SocketAddrRef::Unix(address.read_mut::<SocketAddrUnix>()?)),
50            AF_INET => Ok(SocketAddrRef::INet(address.read_mut::<SocketAddrInet>()?)),
51            AF_NETLINK => Ok(SocketAddrRef::Netlink),
52
53            _ => Err(SyscallError::EINVAL),
54        }
55    }
56
57    pub fn from_ifreq(ifreq: &IfReq) -> Result<Self> {
58        // SAFETY???
59        let family = unsafe { ifreq.data.addr.sa_family };
60        SocketAddrRef::from_family(
61            VirtAddr::new(&unsafe { ifreq.data.addr } as *const _ as _),
62            family,
63        )
64    }
65
66    /// Converts the socket address into a unix socket address. Returns [`None`] if
67    /// the address is not a unix socket address.
68    pub fn as_unix(&self) -> Option<&'a SocketAddrUnix> {
69        match self {
70            SocketAddrRef::Unix(address) => Some(address),
71            _ => None,
72        }
73    }
74
75    pub fn as_inet(&self) -> Option<&'a SocketAddrInet> {
76        match self {
77            SocketAddrRef::INet(addr) => Some(addr),
78            _ => None,
79        }
80    }
81}
