28/// Structure representing a *virtual* address space. The address space
29/// contains a reference of the page table allocated for this address space.
30pub struct AddressSpace {
31    cr3: PhysFrame,
32}
33
34impl AddressSpace {
35    /// Allocates a new *virtual* address space.
36    pub fn new() -> Result<Self, MapToError<Size4KiB>> {
37        let cr3 = unsafe {
38            let frame = FRAME_ALLOCATOR
39                .allocate_frame()
40                .ok_or(MapToError::FrameAllocationFailed)?;
41
42            let phys_addr = frame.start_address();
43            let virt_addr = phys_addr.as_hhdm_virt();
44
45            phys_addr.as_vm_frame().unwrap().inc_ref_count();
46
47            let page_table: *mut PageTable = virt_addr.as_mut_ptr();
48            let page_table = &mut *page_table;
49
50            let current_table = active_level_4_table();
51
52            // Zero out the page table entries from the range 0..256.
53            for i in 0..256 {
54                page_table[i].set_unused();
55            }
56
57            // Map the higher half of the kernel's address space into this address
58            // space.
59            for i in 256..512 {
60                page_table[i] = current_table[i].clone();
61            }
62
63            frame
64        };
65
66        Ok(Self { cr3 })
67    }
68
69    /// Returns the current active address space.
70    pub fn this() -> Self {
71        #[cfg(target_arch = "x86_64")]
72        {
73            let cr3 = {
74                // Get the value of the Cr3 register.
75                let value: u64;
76
77                unsafe {
78                    asm!("mov {}, cr3", out(reg) value, options(nomem));
79                }
80
81                let addr = PhysAddr::new(value & 0x_000f_ffff_ffff_f000);
82
83                PhysFrame::containing_address(addr)
84            };
85
86            Self { cr3 }
87        }
88
89        #[cfg(target_arch = "aarch64")]
90        unimplemented!()
91    }
92
93    pub fn switch(&self) {
94        #[cfg(target_arch = "x86_64")]
95        {
96            let cr3 = self.cr3().start_address().as_u64();
97
98            unsafe {
99                asm!("mov cr3, {}", in(reg) cr3, options(nostack)); // Load the new address space
100            }
101        }
102
103        #[cfg(target_arch = "aarch64")]
104        unimplemented!()
105    }
106
107    /// Returns a reference to the page table frame allocated for this address
108    /// space.
109    pub fn cr3(&self) -> PhysFrame {
110        self.cr3
111    }
112
113    /// Returns a mutable reference to the page table allocated for this
114    /// address space.
115    pub fn page_table(&mut self) -> &'static mut PageTable {
116        unsafe { &mut *(self.cr3.start_address().as_hhdm_virt().as_mut_ptr()) }
117    }
118
119    /// Returns a mutable reference to the mapper pointing to the page table
120    /// allocated for this address space.
121    pub fn offset_page_table(&mut self) -> OffsetPageTable {
122        unsafe { OffsetPageTable::new(self.page_table(), crate::PHYSICAL_MEMORY_OFFSET) }
123    }
124}
125
126pub fn alloc_boxed_buffer<T>(size: usize) -> Box<[T]> {
127    unsafe { Box::new_zeroed_slice(size).assume_init() }
