18//! Due to internal-fragmentation in the buddy frame allocator, we cannot allocate large
19//! amount of contiguous physical memory. We instead use [`vmalloc`] to allocate virtually
20//! contiguous memory. The allocator uses a red-black tree to keep track of the free memory
21//! so we can allocate and free memory efficiently.
22//!
23//! An area is reserved for [`vmalloc`] in the kernel address space, starting
24//! at [`VMALLOC_VIRT_START`] and ending at [`VMALLOC_VIRT_END`].
25
26use alloc::boxed::Box;
27use alloc::collections::VecDeque;
28use intrusive_collections::*;
29use spin::Once;
30
31use crate::utils::sync::{Mutex, MutexGuard};
32
33use super::paging::*;
34use super::AddressSpace;
35
36pub(super) const VMALLOC_MAX_SIZE: usize = 128 * 1024 * 1024; // 128 GiB
37pub(super) const VMALLOC_START: VirtAddr = VirtAddr::new(0xfffff80000000000);
38pub(super) const VMALLOC_END: VirtAddr =
39    VirtAddr::new(0xfffff80000000000 + VMALLOC_MAX_SIZE as u64);
40
41static VMALLOC: Once<Mutex<Vmalloc>> = Once::new();
42
43struct VmallocAreaProtected {
44    addr: VirtAddr,
45    size: usize,
46}
47
48impl VmallocAreaProtected {
49    fn new(addr: VirtAddr, size: usize) -> Self {
50        Self { addr, size }
51    }
52}
53
54struct VmallocArea {
55    // NOTE: Since there are equal amount of read and write operations we are going to
56    // protect the data using a [`Mutex`].
57    protected: Mutex<VmallocAreaProtected>,
58    link: RBTreeLink,
59}
60
61impl VmallocArea {
62    fn new(addr: VirtAddr, size: usize) -> Self {
63        Self {
64            protected: Mutex::new(VmallocAreaProtected::new(addr, size)),
65            link: Default::default(),
66        }
67    }
68}
69
70impl<'a> KeyAdapter<'a> for VmallocAreaAdaptor {
71    type Key = usize;
72
73    fn get_key(&self, this: &'a VmallocArea) -> Self::Key {
74        // NOTE: We use the size of the vmalloc area as the key for the red-black tree
75        // so when we are allocating or deallocating memory we can find a large enough, free
76        // vmalloc area efficiently.
77        this.protected.lock().size
78    }
79}
80
81intrusive_collections::intrusive_adapter!(VmallocAreaAdaptor = Box<VmallocArea>: VmallocArea { link: RBTreeLink });
82
83pub(super) struct Vmalloc {
84    free_list: VecDeque<VmallocArea>,
85}
86
87impl Vmalloc {
88    fn new() -> Self {
89        let mut this = Self {
90            free_list: VecDeque::new(),
91        };
92
93        this.free_list
94            .push_back(VmallocArea::new(VMALLOC_START, VMALLOC_MAX_SIZE));
95
96        this
97    }
98
99    pub(super) fn alloc(&mut self, npages: usize) -> Option<VirtAddr> {
100        // +1: area for the guard page.
101        let size_bytes = (npages + 1) * Size4KiB::SIZE as usize;
102
103        let (i, area) = self
104            .free_list
105            .iter()
106            .enumerate()
107            .find(|(_, area)| area.protected.lock().size >= size_bytes)?;
108
109        let mut area_p = area.protected.lock();
110        let address = area_p.addr;
111
112        if area_p.size > size_bytes {
113            area_p.addr += size_bytes;
114            area_p.size -= size_bytes;
115        } else {
116            // NOTE: the area is has exactly the requested size, so we can remove it
117            // from the free list.
118            core::mem::drop(area_p); // unlock
119
120            self.free_list.remove(i);
121        }
122
123        // subtract the size of the guard page since we are not required to allocate
124        // a frame for that area.
125        let size_bytes = size_bytes - Size4KiB::SIZE as usize;
126
127        let mut address_space = AddressSpace::this();
128        let mut offset_table = address_space.offset_page_table();
129
130        let page_range = {
131            let start_page: Page = Page::containing_address(address);
132            let end_page = Page::containing_address(address + size_bytes);
133
134            Page::range(start_page, end_page)
135        };
136
137        // map the pages at the allocated address.
138        for page in page_range {
139            let frame: PhysFrame<Size4KiB> = FRAME_ALLOCATOR
140                .allocate_frame()
141                .expect("vmalloc: physical memory exhausted");
142
143            unsafe {
144                offset_table.map_to(
145                    page,
146                    frame,
147                    PageTableFlags::PRESENT | PageTableFlags::WRITABLE,
148                )
149            }
150            .unwrap()
151            .flush();
152        }
153
154        Some(address)
155    }
156
157    pub(super) fn dealloc(&mut self, addr: VirtAddr, npages: usize) {
158        // +1: area for the guard page.
159        let size = (npages + 1) * Size4KiB::SIZE as usize;
160
161        // check if this block can be merged into another block.
162        let merge = self
163            .free_list
164            .iter()
165            .find(|area| addr + size == area.protected.lock().addr);
166
167        if let Some(merge) = merge {
168            let mut merge = merge.protected.lock();
169
170            merge.addr = addr;
171            merge.size += size;
172        } else {
173            // We add it to the back of the free list since, its more likely
174            // to find larger free areas in the front of the list.
175            self.free_list.push_back(VmallocArea::new(addr, size));
176        }
177
178        // subtract the size of the guard page since its not mapped.
179        let size = size - Size4KiB::SIZE as usize;
180
181        let mut address_space = AddressSpace::this();
182        let mut offset_table = address_space.offset_page_table();
183
184        let page_range = {
185            let start_page: Page = Page::containing_address(addr);
186            let end_page = Page::containing_address(addr + size);
187
188            Page::range(start_page, end_page)
189        };
190
191        for page in page_range {
192            // unmap the page at the address which in turn will deallocate
193            // the frame (refcnt == 0).
194            offset_table.unmap(page).unwrap().1.flush();
195        }
196    }
197}
198
199pub fn init() {
200    VMALLOC.call_once(|| Mutex::new(Vmalloc::new()));
201}
202
203/// ## Panics
204/// * If the `vmalloc` allocator is not initialized.
205pub(super) fn get_vmalloc() -> MutexGuard<'static, Vmalloc> {
206    VMALLOC
207        .get()
208        .expect("get_vmalloc: not initialized")
209        .lock_irq()
