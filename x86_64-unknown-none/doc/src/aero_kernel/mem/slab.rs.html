18use core::ptr::NonNull;
19
20use intrusive_collections::UnsafeRef;
21
22use crate::mem::paging::*;
23use crate::utils::sync::Mutex;
24
25#[repr(C)]
26pub struct SlabHeader {
27    /// Reference to the slab pool.
28    ptr: UnsafeRef<SmallSlab>,
29}
30
31impl SlabHeader {
32    /// Gets the [`SlabHeader`] from an allocated object.
33    pub fn from_object<'a>(ptr: *const u8) -> &'a Self {
34        assert!(!ptr.is_null());
35
36        let ptr = (ptr as usize & !(0xfff)) as *mut SlabHeader;
37        unsafe { &*ptr }
38    }
39
40    /// Returns the slab pool to which this header belongs to.
41    pub fn as_slab(&self) -> &SmallSlab {
42        self.ptr.as_ref()
43    }
44}
45
46const_assert_eq!(core::mem::size_of::<SlabHeader>(), 8);
47
48unsafe impl Send for SlabHeader {}
49unsafe impl Sync for SlabHeader {}
50
51/// For small slabs, the [`BufCtl`]s are stored inline.
52struct BufCtl(Option<NonNull<BufCtl>>);
53
54impl BufCtl {
55    const NULL: Self = Self(None);
56
57    /// Constructs a [`BufCtl`] from a raw pointer.
58    const fn from_ptr(ptr: *mut BufCtl) -> Self {
59        assert!(!ptr.is_null());
60
61        // SAFETY: We have verified above that the pointer is non-null.
62        Self(Some(unsafe { NonNull::new_unchecked(ptr) }))
63    }
64}
65
66const_assert_eq!(core::mem::size_of::<BufCtl>(), 8);
67
68unsafe impl Send for BufCtl {}
69unsafe impl Sync for BufCtl {}
70
71/// Used for allocations smaller than `1/8` of a page.
72pub struct SmallSlab {
73    /// Size of the slab.
74    size: usize,
75    first_free: Mutex<BufCtl>,
76}
77
78impl SmallSlab {
79    pub const fn new(size: usize) -> Self {
80        assert!(size.is_power_of_two());
81
82        Self {
83            size,
84            first_free: Mutex::new(BufCtl::NULL),
85        }
86    }
87
88    pub fn alloc(&self) -> *mut u8 {
89        let mut first_free = self.first_free.lock_irq();
90
91        if let Some(entry) = first_free.0 {
92            *first_free = BufCtl(unsafe { entry.as_ref() }.0);
93            entry.as_ptr().cast()
94        } else {
95            drop(first_free);
96
97            self.expand();
98            self.alloc()
99        }
100    }
101
102    pub fn dealloc(&self, ptr: *mut u8) {
103        assert!(!ptr.is_null());
104
105        let mut first_free = self.first_free.lock_irq();
106
107        let mut new_head = BufCtl::from_ptr(ptr.cast());
108        new_head.0 = first_free.0;
109        *first_free = new_head;
110    }
111
112    fn expand(&self) {
113        let frame: PhysFrame<Size4KiB> = FRAME_ALLOCATOR.allocate_frame().expect("slab: OOM");
114
115        let ptr = frame.start_address().as_hhdm_virt().as_mut_ptr::<u8>();
116        let header_size =
117            align_up(core::mem::size_of::<SlabHeader>() as u64, self.size as u64) as usize;
118
119        let avaliable_size = Size4KiB::SIZE as usize - header_size;
120        let slab_ptr = unsafe { &mut *ptr.cast::<SlabHeader>() };
121
122        // SAFETY: We are constructing an [`UnsafeRef`] from ourselves which is a valid reference.
123        slab_ptr.ptr = unsafe { UnsafeRef::from_raw(self as *const _) };
124
125        let first_free = unsafe { ptr.add(header_size).cast() };
126        *self.first_free.lock_irq() = BufCtl::from_ptr(first_free);
127
128        // Initialize the free-list:
129        //
130        // For objects smaller than 1/8 of a page, A slab is built by allocating a 4KiB page,
131        // placing the slab header at the end, and dividing the rest into equal-size buffers:
132        //
133        // ------------------------------------------------------
134        // | buffer | buffer | buffer | buffer | slab header
135        // ------------------------------------------------------
136        //                          4KiB
137        let max = (avaliable_size / self.size) - 1;
138        let fact = self.size / 8;
139
140        for i in 0..max {
141            unsafe {
142                let entry = first_free.add(i * fact);
143                let next = first_free.add((i + 1) * fact);
144
145                (*entry).0 = Some(NonNull::new_unchecked(next));
146            }
147        }
148
149        unsafe {
150            let entry = &mut *first_free.add(max * fact);
151            *entry = BufCtl::NULL;
152        }
153    }
154
155    pub fn size(&self) -> usize {
156        self.size
157    }
158}
