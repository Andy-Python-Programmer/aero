36const BUDDY_SIZE: [u64; 10] = [
37    Size4KiB::SIZE,       // 4 KiB
38    Size4KiB::SIZE * 2,   // 8 KiB
39    Size4KiB::SIZE * 4,   // 16 KiB
40    Size4KiB::SIZE * 8,   // 32 KiB
41    Size4KiB::SIZE * 16,  // 64 KiB
42    Size4KiB::SIZE * 32,  // 128 KiB
43    Size4KiB::SIZE * 64,  // 256 KiB
44    Size4KiB::SIZE * 128, // 512 KiB
45    Size4KiB::SIZE * 256, // 1 MiB
46    Size2MiB::SIZE,       // 2 MiB
47];
48
49const fn order_from_size(size: u64) -> usize {
50    // UNSTABLE: We cannot make an iterator from `BUDDY_SIZE` or use a for loop
51    //           in const context.
52    let mut order = 0;
53
54    while order < BUDDY_SIZE.len() {
55        let buddy_size = BUDDY_SIZE[order];
56        if buddy_size >= size {
57            return order;
58        }
59
60        order += 1;
61    }
62
63    unreachable!()
64}

66pub struct LockedFrameAllocator(Mutex<GlobalFrameAllocator>);

68impl LockedFrameAllocator {
69    /// Constructs a new uninitialized and locked version of the global frame
70    /// allocator.
71    pub(super) const fn new_uninit() -> Self {
72        let bstrap_ref = BootAllocRef {
73            inner: core::ptr::null(),
74        };
75
76        Self(Mutex::new(GlobalFrameAllocator {
77            buddies: [
78                Bitmap::empty(bstrap_ref),
79                Bitmap::empty(bstrap_ref),
80                Bitmap::empty(bstrap_ref),
81                Bitmap::empty(bstrap_ref),
82                Bitmap::empty(bstrap_ref),
83                Bitmap::empty(bstrap_ref),
84                Bitmap::empty(bstrap_ref),
85                Bitmap::empty(bstrap_ref),
86                Bitmap::empty(bstrap_ref),
87                Bitmap::empty(bstrap_ref),
88            ],
89            free: [0; 10],
90
91            base: PhysAddr::zero(),
92            end: PhysAddr::zero(),
93        }))
94    }
95
96    /// Initializes the inner locked global frame allocator.
97    pub(super) fn init(&self, memory_map: &mut limine::response::MemoryMapResponse) {
98        *self.0.lock_irq() = GlobalFrameAllocator::new(memory_map);
99    }
100
101    pub fn dealloc(&self, addr: PhysAddr, size_bytes: usize) {
102        let order = order_from_size(size_bytes as u64);
103
104        let mut allocator = self.0.lock_irq();
105        allocator.deallocate_frame_inner(addr, order);
106    }
107
108    pub fn alloc(&self, size_bytes: usize) -> Option<PhysAddr> {
109        let order = order_from_size(size_bytes as u64);
110
111        let mut allocator = self.0.lock_irq();
112        allocator.allocate_frame_inner(order)
113    }
114
115    pub fn alloc_zeroed(&self, size_bytes: usize) -> Option<PhysAddr> {
116        let addr = self.alloc(size_bytes)?;
117        addr.as_hhdm_virt().as_bytes_mut(size_bytes).fill(0);
118
119        Some(addr)
120    }
121}

149struct RangeMemoryIter<'a> {
150    iter: core::slice::Iter<'a, &'a memory_map::Entry>,
151
152    cursor_base: PhysAddr,
153    cursor_end: PhysAddr,
154}
155
156impl<'a> Iterator for RangeMemoryIter<'a> {
157    type Item = MemoryRange;
158
159    fn next(&mut self) -> Option<Self::Item> {
160        if self.cursor_base >= self.cursor_end {
161            if let Some(entry) = loop {
162                // We need to find out the next usable memory range from
163                // the memory map and set the cursor to the start of it.
164                let next = self.iter.next()?;
165
166                if next.entry_type == memory_map::EntryType::USABLE {
167                    break Some(next);
168                }
169            } {
170                self.cursor_base = PhysAddr::new(entry.base).align_up(Size4KiB::SIZE);
171                self.cursor_end = PhysAddr::new(entry.base + entry.length);
172            } else {
173                // We reached the end of the memory map.
174                return None;
175            }
176        }
177
178        let typee = MemoryRangeType::Usable;
179
180        let range = MemoryRange {
181            addr: self.cursor_base,
182            size: self.cursor_end - self.cursor_base,
183            typee,
184        };
185
186        self.cursor_base = self.cursor_end.align_up(Size4KiB::SIZE);
187        Some(range)
188    }
189}

219struct BootAlloc {
220    memory_ranges: Mutex<&'static mut [MemoryRange]>,
221}
222
223impl BootAlloc {
224    fn new(memory_ranges: &'static mut [MemoryRange]) -> Self {
225        Self {
226            memory_ranges: Mutex::new(memory_ranges),
227        }
228    }
229
230    fn allocate_inner(&self, size: usize) -> *mut u8 {
231        let size = align_up(size as u64, Size4KiB::SIZE);
232        for range in self.memory_ranges.lock().iter_mut().rev() {
233            if range.size >= size {
234                let addr = range.addr;
235
236                range.addr += size;
237                range.size -= size;
238
239                return addr.as_hhdm_virt().as_mut_ptr();
240            }
241        }
242
243        unreachable!("pmm: bootstrap allocator is out of memory")
244    }
245}

247#[derive(Debug, Clone, Copy)]
248struct BootAllocRef {
249    inner: *const BootAlloc,
250}
251
252impl BootAllocRef {
253    const fn new(inner: &BootAlloc) -> Self {
254        Self {
255            inner: inner as *const _,
256        }
257    }
258
259    fn get_inner(&self) -> &BootAlloc {
260        unsafe { &*self.inner }
261    }
262}
263
264unsafe impl Allocator for BootAllocRef {
265    fn allocate(&self, layout: Layout) -> Result<NonNull<[u8]>, AllocError> {
266        let this = self.get_inner();
267
268        let aligned_size = align_up(layout.size() as _, layout.align() as _) as usize;
269        let ptr = this.allocate_inner(aligned_size);
270
271        // SAFETY: `allocate_inner` is garunteed to return a valid, non-null pointer.
272        let ptr = unsafe { NonNull::new_unchecked(ptr) };
273        Ok(NonNull::slice_from_raw_parts(ptr, aligned_size))
274    }
275
276    unsafe fn deallocate(&self, _ptr: NonNull<u8>, _layout: Layout) {
277        unreachable!("pmm: bootstrap allocator cannot deallocate")
278    }
279}

285/// Buddy allocator combines power-of-two allocator with free buffer coalescing.
286///
287/// ## Overview
288///
289/// Overview of the buddy allocation algorithm:
290///
291/// * Memory is broken up into large blocks of pages where each block is a power of two number of
292///   pages.
293///
294/// * If a block of the desired size is not available, a larger block is broken up in half and the
295///   two blocks are marked as buddies then one half is used for the allocation and the other half
296///   is marked free.
297///
298/// * The blocks are continuously halved as necessary until a block of the desired size is
299///   available.
300///
301/// * When a block is later freed, the buddy is examined and the two coalesced if it is free.
302pub struct GlobalFrameAllocator {
303    buddies: [Bitmap<BootAllocRef>; 10],
304    free: [usize; 10],
305
306    base: PhysAddr,
307    end: PhysAddr,
308}
309
310impl GlobalFrameAllocator {
311    fn new(memory_map_resp: &mut limine::response::MemoryMapResponse) -> Self {
312        let memory_map = memory_map_resp.entries_mut();
313
314        let requested_size = (core::mem::size_of::<MemoryRange>() * memory_map.len()) as u64;
315
316        let entry = memory_map
317            .iter_mut()
318            .find(|entry| {
319                entry.entry_type == memory_map::EntryType::USABLE && entry.length >= requested_size
320            })
321            .expect("OOM");
322
323        let region = PhysAddr::new(entry.base);
324
325        entry.base += requested_size;
326        entry.length -= requested_size;
327
328        let mut iter = memory_map_resp.entries().iter();
329
330        let cursor = iter
331            .next()
332            .expect("stivale2: unexpected end of the memory map");
333
334        let ranges = unsafe {
335            let virt_addr = region.as_hhdm_virt();
336
337            core::slice::from_raw_parts_mut::<MemoryRange>(
338                virt_addr.as_mut_ptr(),
339                requested_size as usize,
340            )
341        };
342
343        let range_iter = RangeMemoryIter {
344            iter,
345
346            cursor_base: PhysAddr::new(cursor.base),
347            cursor_end: PhysAddr::new(cursor.base + cursor.length),
348        };
349
350        // Lets goo! Now lets initialize the bootstrap allocator so we can initialize
351        // our efficient buddy allocator. We need a separate allocator since some computers
352        // such as Macs have a shitload of memory map entries so, we cannt assume the amount
353        // of maximum mmap entries and allocate space for it on the stack instead. God damn it.
354        let mut i = 0;
355
356        for range in range_iter {
357            ranges[i] = range;
358            i += 1;
359        }
360
361        let base = ranges[0].addr;
362        let end = ranges[i - 1].addr + ranges[i - 1].size;
363
364        let bootstrapper = BootAlloc::new(&mut ranges[..i]);
365        let bref = BootAllocRef::new(&bootstrapper);
366
367        let mut this = Self {
368            base,
369            end,
370
371            buddies: [
372                Bitmap::empty(bref),
373                Bitmap::empty(bref),
374                Bitmap::empty(bref),
375                Bitmap::empty(bref),
376                Bitmap::empty(bref),
377                Bitmap::empty(bref),
378                Bitmap::empty(bref),
379                Bitmap::empty(bref),
380                Bitmap::empty(bref),
381                Bitmap::empty(bref),
382            ],
383            free: [0; 10],
384        };
385
386        let size = this.end - this.base;
387
388        // Allocate the buddies using prealloc:
389        for (i, bsize) in BUDDY_SIZE.iter().enumerate() {
390            let chunk = size / bsize;
391            this.buddies[i] = Bitmap::new_in(bref, chunk as usize);
392        }
393
394        for region in bref.get_inner().memory_ranges.lock().iter() {
395            if region.typee == MemoryRangeType::Usable {
396                this.insert_range(region.addr, region.addr + region.size);
397            }
398        }
399
400        this
401    }
402
403    fn frame_count(&self) -> usize {
404        (self.end.as_u64() / Size4KiB::SIZE) as usize
405    }
406
407    /// Find the perfect buddy order for the provided address range.
408    fn find_order(&self, address: PhysAddr, chunk_size: u64) -> usize {
409        for order in (0..BUDDY_SIZE.len()).rev() {
410            let size = BUDDY_SIZE[order];
411
412            // Too big...
413            if size > chunk_size {
414                continue;
415            }
416
417            let mask = BUDDY_SIZE[order] - 1;
418
419            if mask & address.as_u64() != 0 {
420                continue;
421            }
422
423            return order;
424        }
425
426        0
427    }
428
429    fn get_bit_idx(&self, addr: PhysAddr, order: usize) -> usize {
430        let offset = addr - self.base;
431        (offset / BUDDY_SIZE[order]) as usize
432    }
433
434    fn set_bit(&mut self, addr: PhysAddr, order: usize) -> bool {
435        let idx = self.get_bit_idx(addr, order);
436
437        let buddy = &mut self.buddies[order];
438        let change = !buddy.is_set(idx);
439
440        if change {
441            buddy.set(idx, true);
442            self.free[order] += 1;
443        }
444
445        change
446    }
447
448    #[cfg(test)]
449    fn is_free(&self, addr: PhysAddr, order: usize) -> bool {
450        let idx = self.get_bit_idx(addr, order);
451
452        let buddy = &self.buddies[order];
453        buddy.is_set(idx)
454    }
455
456    /// Inserts the provided memory range.
457    fn insert_range(&mut self, base: PhysAddr, end: PhysAddr) {
458        let mut remaining = end - base;
459        let mut current = base;
460
461        while remaining > 0 {
462            let order = self.find_order(current, remaining);
463            let size = BUDDY_SIZE[order];
464
465            self.set_bit(current, order);
466
467            current += size;
468            remaining -= size;
469        }
470    }
471
472    /// Finds a free chunk with the provided `order`.
473    fn find_free(&mut self, order: usize) -> Option<PhysAddr> {
474        let buddy = &mut self.buddies[order];
475        let first_free = buddy.find_first_set()?;
476
477        buddy.set(first_free, false);
478        self.free[order] -= 1;
479
480        Some(self.base.align_up(BUDDY_SIZE[order]) + (BUDDY_SIZE[order] * first_free as u64))
481    }
482
483    fn clear_bit(&mut self, addr: PhysAddr, order: usize) -> bool {
484        let idx = self.get_bit_idx(addr, order);
485
486        let buddy = &mut self.buddies[order];
487        let change = buddy.is_set(idx);
488
489        if change {
490            buddy.set(idx, false);
491            self.free[order] -= 1;
492        }
493
494        change
495    }
496
497    fn get_buddy(&self, addr: PhysAddr, order: usize) -> PhysAddr {
498        let size = BUDDY_SIZE[order];
499        let base = addr.align_down(size * 2);
500
501        if base == addr {
502            addr + size
503        } else {
504            base
505        }
506    }
507
508    fn deallocate_frame_inner(&mut self, mut addr: PhysAddr, mut order: usize) {
509        while order < BUDDY_SIZE.len() {
510            if order < BUDDY_SIZE.len() - 1 {
511                let buddy = self.get_buddy(addr, order);
512
513                if self.clear_bit(buddy, order) {
514                    addr = core::cmp::min(addr, buddy);
515                    order += 1;
516                } else {
517                    self.set_bit(addr, order);
518                    break;
519                }
520            } else {
521                self.set_bit(addr, order);
522                break;
523            }
524        }
525    }
526
527    fn allocate_frame_inner(&mut self, order: usize) -> Option<PhysAddr> {
528        let size = BUDDY_SIZE[order];
529
530        // Loop through the list of buddies until we can find one that can give us
531        // the requested memory.
532        for (i, &bsize) in BUDDY_SIZE[order..].iter().enumerate() {
533            let i = i + order;
534
535            if self.free[i] > 0 {
536                let result = self.find_free(i)?;
537                let mut remaining = bsize - size;
538
539                if remaining > 0 {
540                    for j in (0..=i).rev() {
541                        let sizee = BUDDY_SIZE[j];
542
543                        while remaining >= sizee {
544                            self.set_bit(result + (remaining - sizee) + size, j);
545                            remaining -= sizee;
546                        }
547                    }
548                }
549
550                return Some(result);
551            }
552        }
553
554        None
555    }
556}

573pub struct VmFrame {
574    ref_count: AtomicUsize,
575}
576
577impl VmFrame {
578    fn new() -> Self {
579        Self {
580            ref_count: AtomicUsize::new(0),
581        }
582    }
583
584    pub fn dec_ref_count(&self) {
585        let ref_count = self.ref_count.load(Ordering::SeqCst);
586
587        if ref_count != 0 {
588            self.ref_count.store(ref_count - 1, Ordering::SeqCst);
589        }
590    }
591
592    pub fn inc_ref_count(&self) {
593        self.ref_count.fetch_add(1, Ordering::SeqCst);
594    }
595
596    pub fn ref_count(&self) -> usize {
597        self.ref_count.load(Ordering::SeqCst)
598    }
