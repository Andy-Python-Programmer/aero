71#[derive(Debug, Clone, Copy, PartialEq)]
72pub enum TaskState {
73    Runnable,
74    Zombie,
75    AwaitingIo,
76    AwaitingIoDeadline,
77}
78
79impl From<u8> for TaskState {
80    fn from(x: u8) -> Self {
81        match x {
82            0 => TaskState::Runnable,
83            1 => TaskState::Zombie,
84            2 => TaskState::AwaitingIo,
85            3 => TaskState::AwaitingIoDeadline,
86            _ => panic!("invalid task state"),
87        }
88    }

115struct Zombies {
116    list: Mutex<LinkedList<SchedTaskAdapter>>,
117    block: WaitQueue,
118}
119
120impl Zombies {
121    fn new() -> Self {
122        Self {
123            list: Mutex::new(Default::default()),
124            block: WaitQueue::new(),
125        }
126    }
127
128    fn add_zombie(&self, zombie: Arc<Task>) {
129        assert!(!zombie.link.is_linked());
130        assert_eq!(zombie.state(), TaskState::Zombie);
131
132        let mut list = self.list.lock();
133
134        log::debug!("making process a zombie: (pid={:?})", zombie.pid());
135
136        list.push_back(zombie);
137        self.block.notify_all();
138    }
139
140    fn waitpid(
141        &self,
142        pids: &[usize],
143        status: &mut u32,
144        flags: WaitPidFlags,
145    ) -> SignalResult<usize> {
146        let mut captured = None;
147
148        self.block.block_on(&self.list, |l| {
149            let mut cursor = l.front_mut();
150
151            while let Some(t) = cursor.get() {
152                for pid in pids {
153                    if t.pid().as_usize() == *pid {
154                        captured = Some((t.pid(), t.exit_status().clone()));
155                        cursor.remove();
156
157                        return true;
158                    }
159                }
160
161                cursor.move_next();
162            }
163
164            if flags.contains(WaitPidFlags::WNOHANG) {
165                return true;
166            }
167
168            false
169        })?;
170
171        if let Some((tid, exit_status)) = captured {
172            // mlibc/abis/linux/wait.h (`W_EXITCODE`)
173            match exit_status {
174                ExitStatus::Normal(code) => {
175                    *status = (code as u32) << 8;
176                }
177
178                ExitStatus::Signal(signal) => {
179                    *status = signal as u32;
180                }
181            }
182
183            Ok(tid.as_usize())
184        } else {
185            // If `WNOHANG` was specified in flags and there were no children in a waitable
186            // state, then waipid() returns 0 immediately.
187            *status = 0;
188            Ok(0)
189        }
190    }
191}

486    pub fn waitpid(
487        &self,
488        pid: isize,
489        status: &mut u32,
490        flags: WaitPidFlags,
491    ) -> SignalResult<usize> {
492        if pid == -1 {
493            // wait for any child process if no specific process is requested.
494            //
495            // NOTE: we collect all of the zombie list's process IDs with the children
496            // list since the child could have been removed from the children list and
497            // become a zombie before the parent had a chance to wait for it.
498            let mut pids = self
499                .zombies
500                .list
501                .lock_irq()
502                .iter()
503                .map(|e| e.pid().as_usize())
504                .collect::<alloc::vec::Vec<_>>();
505
506            pids.extend(self.children.lock_irq().iter().map(|e| e.pid().as_usize()));
507            self.zombies.waitpid(&pids, status, flags)
508        } else {
509            self.zombies.waitpid(&[pid as _], status, flags)
510        }
511    }

518        &self,
519        executable: &DirCacheItem,
520
521        argv: Option<ExecArgs>,
522        envv: Option<ExecArgs>,
523    ) -> Result<(), MapToError<Size4KiB>> {

525            *self.cwd.write() = Some(Cwd::new())
526        }
527
528        // if executable.absolute_path_str().contains("gcc")
529        //     || executable.absolute_path_str().contains("ls")
530        // {
531        // self.enable_systrace();
532        // }
533
534        *self.mem_tags.lock() = HashMap::new();
535        self.file_table.close_on_exec();
536
537        self.file_table.log();
538
539        *self.executable.lock() = Some(executable.clone());
540
541        let vm = self.vm();
542        vm.clear();
543
544        // Clear the signals that are pending for this task on exec.
545        self.signals().clear();
546
547        self.arch_task_mut().exec(vm, executable, argv, envv)
548    }

688    pub fn signal(&self, signal: usize) -> bool {
689        match self.signals().trigger(signal, false) {
690            TriggerResult::Triggered => {
691                self.wake_up();
692                true
693            }
694
695            TriggerResult::Ignored => false,
696
697            TriggerResult::Blocked => {
698                // Find other thread in process to notify
699                let process_leader = self.process_leader();
700
701                if !process_leader.signals().is_blocked(signal) {
702                    process_leader.wake_up();
703
704                    return true;
705                }
706
707                for c in process_leader
708                    .children
709                    .lock()
710                    .iter()
711                    .filter(|t| t.pid() == self.pid())
712                {
713                    if !c.signals().is_blocked(signal) {
714                        c.wake_up();
715
716                        return true;
717                    }
718                }
719
720                false
721            }
722        }
723    }

725    pub(super) fn make_zombie(&self) {
726        self.detach();
727        self.arch_task_mut().dealloc();
728
729        if let Some(parent) = self.get_parent() {
730            parent.remove_child(self);
731            parent.zombies.add_zombie(self.this());
732
733            if self.is_process_leader() {
734                parent.signal(aero_syscall::signal::SIGCHLD);
735            }
736        }
737    }
738
739    pub fn systrace(&self) -> bool {
740        self.systrace.load(Ordering::SeqCst)
741    }
742
743    pub fn enable_systrace(&self) {
744        self.systrace.store(true, Ordering::SeqCst);
745    }
746
747    pub fn detach(&self) {
748        let mut controlling_terminal = self.controlling_terminal.lock_irq();
749
750        if let Some(term) = controlling_terminal.as_ref() {
751            term.detach(self.sref.upgrade().unwrap());
752            *controlling_terminal = None;
753        }
754    }
755
756    pub fn attach(&self, terminal: Arc<dyn TerminalDevice>) {
757        if self.is_session_leader() {
758            terminal.attach(self.sref.upgrade().unwrap());
759            *self.controlling_terminal.lock_irq() = Some(terminal);
760        } else {
761            // FIXME: If its not the session leader then we needs to be a part of the the same
762            // session as the terminal's foreground group.
763            *self.controlling_terminal.lock_irq() = Some(terminal);
764        }
765    }
766
767    /// Returns the controlling terminal of the task.
768    pub fn controlling_terminal(&self) -> Option<Arc<dyn TerminalDevice>> {
769        self.controlling_terminal.lock_irq().clone()
770    }
771
772    /// Returns whether the task is the session leader (`pid` == `sid`).
773    pub fn is_session_leader(&self) -> bool {
774        self.session_id() == self.pid().as_usize()
775    }
776
777    /// Returns whether the task is the group leader (`pid` == `gid`).
778    pub fn is_group_leader(&self) -> bool {
779        self.group_id() == self.pid().as_usize()
780    }
781
782    /// Returns the group identifier of the task (`GID`).
783    pub fn group_id(&self) -> usize {
784        self.gid.load(Ordering::SeqCst)
785    }
786
787    /// Returns the session identifier of the task (`SID`).
788    pub fn session_id(&self) -> usize {
789        self.sid.load(Ordering::SeqCst)
790    }
791
792    /// Sets the session identifier of the task (`SID`) to `session_id`.
793    pub(super) fn set_session_id(&self, session_id: usize) {
794        self.sid.store(session_id, Ordering::SeqCst);
795    }
796
797    /// Sets the group identifier of the task (`GID`) to `group_id`.
798    pub(super) fn set_group_id(&self, group_id: usize) {
799        self.gid.store(group_id, Ordering::SeqCst);
800    }
801}
