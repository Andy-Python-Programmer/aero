25/// Process Group
26///
27/// A process group is a collection of one or more processes that are grouped together so that
28/// they can be manipulated as a single entity.
29pub struct Group {
30    /// Unique identifier of the process group.
31    id: usize,
32    /// Processes part of the process group.
33    tasks: Mutex<HashMap<TaskId, Arc<Task>>>,
34}
35
36impl Group {
37    /// Creates a new process group.
38    pub fn new(leader: Arc<Task>) -> Arc<Self> {
39        leader.set_group_id(leader.pid().as_usize());
40
41        let id = leader.pid().as_usize();
42
43        let mut tasks = HashMap::new();
44        tasks.insert(leader.pid(), leader);
45
46        Arc::new(Self {
47            id,
48            tasks: Mutex::new(tasks),
49        })
50    }
51
52    pub fn id(&self) -> usize {
53        self.id
54    }
55
56    pub fn register_task(&self, task: Arc<Task>) {
57        assert!(self.tasks.lock_irq().insert(task.pid(), task).is_none());
58    }
59
60    pub fn remove_task(&self, task: &Arc<Task>) {
61        assert!(self.tasks.lock_irq().remove(&task.pid()).is_some());
62    }
63
64    pub fn is_empty(&self) -> bool {
65        self.tasks.lock_irq().is_empty()
66    }
67
68    pub fn signal(&self, target: usize) {
69        for (_, task) in self.tasks.lock_irq().iter() {
70            log::error!("Sending signal to task: {:?}", task.path());
71
72            task.signal(target);
73        }
74    }
75}

77/// Process Session
78pub struct Session {
79    groups: Mutex<HashMap<usize, Arc<Group>>>,
80}
81
82impl Session {
83    /// Creates a new process session.
84    pub fn new(leader: Arc<Task>) -> Arc<Self> {
85        leader.set_session_id(leader.pid().as_usize());
86
87        let mut groups = HashMap::new();
88        groups.insert(leader.pid().as_usize(), Group::new(leader));
89
90        Arc::new(Self {
91            groups: Mutex::new(groups),
92        })
93    }
94
95    pub fn find(&self, target: &Arc<Task>) -> Option<Arc<Group>> {
96        self.groups.lock_irq().get(&target.group_id()).cloned()
97    }
98
99    pub fn register_task(&self, task: Arc<Task>) {
100        assert!(!task.is_session_leader());
101
102        let mut groups = self.groups.lock_irq();
103        if let Some(group) = groups.get(&task.group_id()) {
104            assert!(!task.is_group_leader());
105            group.register_task(task);
106        } else {
107            assert!(task.is_group_leader());
108            groups.insert(task.group_id(), Group::new(task));
109        }
110    }
111
112    pub fn remove_task(&self, task: &Arc<Task>) {
113        let mut groups = self.groups.lock();
114        let group = groups
115            .get(&task.group_id())
116            .expect("Session::remove_task: ESRCH");
117
118        group.remove_task(task);
119
120        if group.is_empty() {
121            assert!(task.is_group_leader());
122            groups.remove(&task.group_id());
123        }
124    }
125
126    pub fn is_empty(&self) -> bool {
127        self.groups.lock_irq().is_empty()
128    }
129}

131pub struct SessionList(Mutex<HashMap<usize, Arc<Session>>>);
132
133impl SessionList {
134    /// Creates and registers a new process session with the given `leader` as the leader task.
135    pub fn create_session(&self, leader: Arc<Task>) {
136        self.0
137            .lock_irq()
138            .insert(leader.pid().as_usize(), Session::new(leader));
139    }
140
141    pub fn find_group(&self, target: &Arc<Task>) -> Option<Arc<Group>> {
142        self.0.lock_irq().get(&target.session_id())?.find(target)
143    }
144
145    pub fn register_task(&self, task: Arc<Task>) {
146        assert!(task.is_process_leader());
147
148        let sessions = self.0.lock_irq();
149
150        if let Some(session) = sessions.get(&task.session_id()) {
151            session.register_task(task);
152        } else {
153            drop(sessions);
154            self.create_session(task);
155        }
156    }
157
158    pub fn remove_task(&self, task: &Arc<Task>) {
159        let mut sessions = self.0.lock_irq();
160        let session = sessions
161            .get(&task.session_id())
162            .expect("SessionList::remove_task: ESRCH");
163
164        session.remove_task(task);
165
166        if session.is_empty() {
167            assert!(task.is_session_leader());
168            sessions.remove(&task.session_id());
169        }
170    }
171
172    pub fn isolate(&self, task: &Arc<Task>) {
173        assert!(!task.is_group_leader() && !task.is_session_leader());
174
175        let leader = task.process_leader();
176
177        {
178            let mut sessions = self.0.lock_irq();
179            sessions.remove(&task.session_id());
180        }
181
182        self.create_session(leader)
183    }
184}

187    pub static ref SESSIONS: SessionList = SessionList(Mutex::new(HashMap::new()));
