46bitflags::bitflags! {
47    #[derive(Debug, Copy, Clone, PartialEq)]
48    pub struct VmFlag: u8 {
49        // currently active flags
50        const READ      = MMapProt::PROT_READ.bits() as _;
51        const WRITE     = MMapProt::PROT_WRITE.bits() as _;
52        const EXEC      = MMapProt::PROT_EXEC.bits() as _;
53
54        // limits
55        const MAY_READ  = 1 << 3;
56        const MAY_WRITE = 1 << 4;
57        const MAY_EXEC  = 1 << 5;
58
59        const SHARED    = 1 << 6;
60    }
61}

63const VM_PROT_MASK: VmFlag =
64    VmFlag::from_bits_retain(VmFlag::READ.bits() | VmFlag::WRITE.bits() | VmFlag::EXEC.bits());

66impl From<MMapProt> for VmFlag {
67    #[inline]
68    fn from(value: MMapProt) -> Self {
69        Self::from_bits_retain(value.bits().try_into().unwrap())
70    }
71}

73impl From<VmFlag> for PageTableFlags {
74    fn from(flags: VmFlag) -> Self {
75        let mut value = PageTableFlags::empty();

77        if !flags.contains(VmFlag::EXEC) {
78            value.insert(PageTableFlags::NO_EXECUTE);
79        }
80
81        if flags.contains(VmFlag::WRITE) {
82            value.insert(PageTableFlags::WRITABLE);
83        }
84
85        value
86    }

95pub enum ElfLoadError {
96    /// Unexpected file system error occurred on an IO operation on the file.
97    IOError(FileSystemError),
98    /// The PT1 header has an invalid magic number.
99    InvalidMagic,
100    /// The ELF header contains an invalid class.
101    InvalidClass,
102    /// The provided program header index is invalid.
103    InvalidProgramHeaderIndex,
104    /// Unexpected file system error occurred when memory mapping an
105    /// ELF segment.
106    MemoryMapError,
107}

109fn parse_elf_header<'header>(file: &DirCacheItem) -> Result<Header<'header>, ElfLoadError> {
110    // 1. Read the ELF PT1 header:
111    let pt1_hdr_slice = Box::leak(mem::alloc_boxed_buffer::<u8>(ELF_PT1_SIZE));
112
113    file.inode()
114        .read_at(0, pt1_hdr_slice)
115        .map_err(ElfLoadError::IOError)?;
116
117    let pt1_header: &'header _ = unsafe { &*pt1_hdr_slice.as_ptr().cast::<HeaderPt1>() };
118
119    // 2. Ensure that the header has the correct magic number:
120    if pt1_header.magic != ELF_HEADER_MAGIC {
121        return Err(ElfLoadError::InvalidMagic);
122    }
123
124    let pt2_header = match pt1_header.class() {
125        // 3. Read the 64-bit PT2 header:
126        Class::SixtyFour => {
127            let pt2_hdr_slice = Box::leak(mem::alloc_boxed_buffer::<u8>(ELF_PT2_64_SIZE));
128
129            file.inode()
130                .read_at(ELF_PT1_SIZE, pt2_hdr_slice)
131                .map_err(ElfLoadError::IOError)?;
132
133            let pt2_header_ptr = pt2_hdr_slice.as_ptr();
134            let pt2_header: &'header _ = unsafe { &*pt2_header_ptr.cast::<HeaderPt2_<P64>>() };
135
136            Ok(HeaderPt2::Header64(pt2_header))
137        }
138
139        // 3. Read the 32-bit PT2 header:
140        Class::ThirtyTwo => {
141            unimplemented!("parse_elf_header: 32-bit executables are not implemented")
142        }
143
144        // SAFETY: ensure the PT1 header has a valid class.
145        Class::None | Class::Other(_) => Err(ElfLoadError::InvalidClass),
146    }?;
147
148    Ok(Header {
149        pt1: pt1_header,
150        pt2: pt2_header,
151    })
152}

154fn parse_program_header<'pheader>(
155    file: &DirCacheItem,
156    header: Header<'pheader>,
157    index: u16,
158) -> Result<ProgramHeader<'pheader>, ElfLoadError> {
159    let pt2 = &header.pt2;
160
161    // SAFETY: ensure that the provided program header index is valid.
162    if !(index < pt2.ph_count() && pt2.ph_offset() > 0 && pt2.ph_entry_size() > 0) {
163        return Err(ElfLoadError::InvalidProgramHeaderIndex);
164    }
165
166    // 1. Calculate the start offset and size of the program header:
167    let start = pt2.ph_offset() as usize + index as usize * pt2.ph_entry_size() as usize;
168    let size = pt2.ph_entry_size() as usize;
169
170    // 2. Read the 64-bit program header:
171    let phdr_buffer = Box::leak(mem::alloc_boxed_buffer::<u8>(size));
172
173    file.inode()
174        .read_at(start, phdr_buffer)
175        .map_err(ElfLoadError::IOError)?;
176
177    let phdr_ptr = phdr_buffer.as_ptr();
178
179    match header.pt1.class() {
180        // 3. Cast and return the 64-bit program header:
181        Class::SixtyFour => {
182            let phdr: &'pheader _ = unsafe { &*phdr_ptr.cast::<ProgramHeader64>() };
183            Ok(ProgramHeader::Ph64(phdr))
184        }
185
186        // 3. Cast and return the 32-bit program header:
187        Class::ThirtyTwo => {
188            let phdr: &'pheader _ = unsafe { &*phdr_ptr.cast::<ProgramHeader32>() };
189            Ok(ProgramHeader::Ph32(phdr))
190        }
191
192        // SAFETY: ensure the PT1 header has a valid class.
193        Class::None | Class::Other(_) => Err(ElfLoadError::InvalidClass),
194    }
195}

202impl Shebang {

204        let interpreter = fs::lookup_path(path.as_ref()).map_err(ElfLoadError::IOError)?;

207            interpreter,
208            argument,
209        })
210    }

213/// Returns [`true`] if the provided executable (`bin`) contains a shebang
214/// at the start.
215fn contains_shebang(bin: &DirCacheItem) -> Result<bool, ElfLoadError> {
216    let shebang = &mut [0u8; 2];
217
218    bin.inode()
219        .read_at(0, shebang)
220        .map_err(ElfLoadError::IOError)?;
221
222    Ok(shebang[0] == b'#' && shebang[1] == b'!')
223}
224
225fn parse_shebang(bin: &DirCacheItem) -> Result<Option<Shebang>, ElfLoadError> {
226    if !contains_shebang(bin)? {
227        return Ok(None);
228    }
229
230    // Syntax: #![whitespace]interpreter_path [single-argument][new-line]
231    //
232    // NOTE: We set the position to `2` since we skip the `#!` prefix.
233    let mut idx = 2;
234
235    let read_at_index = |idx: usize| -> Result<char, ElfLoadError> {
236        let c = &mut [0u8; 1];
237
238        bin.inode().read_at(idx, c).map_err(ElfLoadError::IOError)?;
239
240        Ok(c[0] as char)
241    };
242
243    // 1. check for the optional whitespace (ignore it):
244    if read_at_index(idx)? == ' ' {
245        idx += 1;
246    }
247
248    // we build the string with `16` capicity to avoid reallocations.
249    let mut path = String::with_capacity(16);
250    let mut arg = String::with_capacity(16);
251
252    // 2. parse the interpreter path:
253    loop {
254        let char = read_at_index(idx)?;
255
256        if char == ' ' {
257            idx += 1;
258            break;
259        } else if char == '\n' {
260            // there is no argument, early return:
261            return Ok(Some(Shebang::new(path, arg)?));
262        }
263
264        idx += 1;
265        path.write_char(char)
266            .expect("parse_shebang: internal error");
267    }
268
269    // 3. parse the argument:
270    loop {
271        let char = read_at_index(idx)?;
272        idx += 1;
273
274        if char == '\n' || char == ' ' {
275            return Ok(Some(Shebang::new(path, arg)?));
276        }
277
278        arg.write_char(char).expect("parse_shebang: internal error");
279    }
280}

298struct ProgramHeaderIter<'this> {
299    file: DirCacheItem,
300    header: Header<'this>,
301    next_index: usize,
302}
303
304impl<'this> ProgramHeaderIter<'this> {
305    fn new(header: Header<'this>, file: DirCacheItem) -> Self {
306        Self {
307            file,
308            header,
309
310            next_index: 0,
311        }
312    }
313}
314
315impl<'this> Iterator for ProgramHeaderIter<'this> {
316    type Item = ProgramHeader<'this>;
317
318    fn next(&mut self) -> Option<Self::Item> {
319        let count = self.header.pt2.ph_count() as usize;
320
321        // We have reached at the end of the program header array.
322        if self.next_index >= count {
323            return None;
324        }
325
326        // Parse and return the program header.
327        let result = parse_program_header(&self.file, self.header, self.next_index as u16).ok();
328
329        // Increment the next index.
330        self.next_index += 1;
331        result
332    }
333}

384impl Mapping {
385    pub fn set_protection(&mut self, protection: MMapProt) -> aero_syscall::Result<()> {
386        if (protection.contains(MMapProt::PROT_READ) && !self.flags.contains(VmFlag::MAY_READ))
387            || (protection.contains(MMapProt::PROT_WRITE)
388                && !self.flags.contains(VmFlag::MAY_WRITE))
389            || (protection.contains(MMapProt::PROT_EXEC) && !self.flags.contains(VmFlag::MAY_EXEC))
390        {
391            return Err(aero_syscall::SyscallError::EACCES);
392        }
393
394        self.flags = (self.flags & !VM_PROT_MASK) | protection.into();
395        Ok(())
396    }

399    pub fn protection(&self) -> VmFlag {
400        self.flags & VM_PROT_MASK

403    /// Handler routine for private anonymous pages. Since its an anonymous page is not
404    /// backed by a file, we have to alloctate a frame and map it at the faulted address.
405    fn handle_pf_private_anon(
406        &mut self,
407        offset_table: &mut OffsetPageTable,
408        reason: PageFaultErrorCode,
409        address: VirtAddr,
410    ) -> bool {
411        let addr_aligned = address.align_down(Size4KiB::SIZE);
412
413        if !reason.contains(PageFaultErrorCode::PROTECTION_VIOLATION) {
414            let frame: PhysFrame =
415                PhysFrame::containing_address(pmm_alloc(BuddyOrdering::Size4KiB));
416
417            unsafe {
418                offset_table.map_to(
419                    Page::containing_address(addr_aligned),
420                    frame,
421                    // NOTE: We dont need to remove the writeable flag from this mapping, since
422                    // the writeable flag will be removed from the parent and child on fork so,
423                    // the mapping gets copied on write.
424                    PageTableFlags::USER_ACCESSIBLE | PageTableFlags::PRESENT | self.flags.into(),
425                )
426            }
427            .expect("Failed to identity map userspace private mapping")
428            .flush();
429
430            true
431        } else if reason.contains(PageFaultErrorCode::CAUSED_BY_WRITE) {
432            self.handle_cow(offset_table, addr_aligned, false)
433        } else {
434            if !self.refresh_flags {
435                return false;
436            }
437
438            unsafe {
439                // The page is present but most likely the flags need to be updated after
440                // mprotect(2).
441                let page: Page<Size4KiB> = Page::containing_address(address);
442                offset_table
443                    .update_flags(
444                        page,
445                        PageTableFlags::USER_ACCESSIBLE
446                            | PageTableFlags::PRESENT
447                            | self.flags.into(),
448                    )
449                    .unwrap()
450                    .flush();
451            }
452
453            self.refresh_flags = false;
454            true
455        }
456    }

458    /// Handler routine for pages backed by a file. This function will allocate a frame and
459    /// read a page-sized amount from the disk into the allocated frame. Then it maps
460    /// the allocated frame at the faulted address.
461    fn handle_pf_file(
462        &mut self,
463        offset_table: &mut OffsetPageTable,
464        reason: PageFaultErrorCode,
465        addr: VirtAddr,
466    ) -> bool {
467        if let Some(file) = self.file.as_ref() {
468            let offset = align_down(
469                (addr - self.start_addr) + file.offset as u64,
470                Size4KiB::SIZE,
471            );
472
473            let addr = addr.align_down(Size4KiB::SIZE);
474            let size = Size4KiB::SIZE.min(file.size as u64 - (addr - self.start_addr));
475
476            return if self.flags.contains(VmFlag::SHARED) {
477                self.handle_pf_shared_file(offset_table, reason, addr, offset as _, size as _)
478            } else {
479                self.handle_pf_private_file(offset_table, reason, addr, offset as _, size as _)
480            };
481        }
482
483        false
484    }
485
486    fn handle_pf_private_file(
487        &mut self,
488        offset_table: &mut OffsetPageTable,
489        reason: PageFaultErrorCode,
490
491        addr: VirtAddr,
492        offset: usize,
493        size: usize,
494    ) -> bool {
495        let mmap_file = self.file.as_mut().unwrap();
496        let page_cache = if let MMapPage::PageCache(page_cache) =
497            mmap_file.file.inode().mmap_v2(offset).unwrap()
498        {
499            page_cache
500        } else {
501            todo!()
502        };
503
504        if !reason.contains(PageFaultErrorCode::PROTECTION_VIOLATION)
505            && !reason.contains(PageFaultErrorCode::CAUSED_BY_WRITE)
506        {
507            let frame = if size == Size4KiB::SIZE as usize {
508                page_cache.page()
509            } else {
510                // The end needs to be zeroed out so we cannot directly map the cached page.
511                let page: Page = Page::containing_address(page_cache.data_addr().as_hhdm_virt());
512
513                let new_frame: PhysFrame = PhysFrame::containing_address(
514                    FRAME_ALLOCATOR
515                        .alloc_zeroed(Size4KiB::SIZE as usize)
516                        .unwrap(),
517                );
518
519                let new_slice = new_frame.as_slice_mut::<u8>();
520                new_slice[..size].copy_from_slice(unsafe {
521                    core::slice::from_raw_parts(page.start_address().as_ptr::<u8>(), size)
522                });
523
524                new_frame
525            };
526
527            unsafe {
528                offset_table.map_to(
529                    Page::containing_address(addr),
530                    frame,
531                    PageTableFlags::PRESENT
532                        | PageTableFlags::USER_ACCESSIBLE
533                        | (self.flags & !VmFlag::WRITE).into(),
534                )
535            }
536            .expect("failed to map allocated frame for private file read")
537            .flush();
538
539            true
540        } else if !reason.contains(PageFaultErrorCode::PROTECTION_VIOLATION)
541            && reason.contains(PageFaultErrorCode::CAUSED_BY_WRITE)
542        {
543            // We are writing to private file mapping so copy the content of the page.
544            let frame = mmap_file
545                .file
546                .inode()
547                .mmap(offset, size, MMapFlags::empty())
548                .expect("handle_pf_file: file does not support mmap");
549
550            unsafe {
551                offset_table.map_to(
552                    Page::containing_address(addr),
553                    frame,
554                    PageTableFlags::PRESENT | PageTableFlags::USER_ACCESSIBLE | self.flags.into(),
555                )
556            }
557            .expect("failed to map allocated frame for private file read")
558            .flush();
559
560            assert!(mmap_file.mappings.get(&addr).is_none());
561            true
562        } else if reason.contains(PageFaultErrorCode::CAUSED_BY_WRITE)
563            && reason.contains(PageFaultErrorCode::PROTECTION_VIOLATION)
564        {
565            if self.handle_cow(offset_table, addr, true) {
566                self.file.as_mut().unwrap().mappings.remove(&addr);
567                return true;
568            }
569
570            false
571        } else {
572            log::error!("    - present page read failed");
573            false
574        }
575    }

577    fn handle_pf_shared_file(
578        &mut self,
579        offset_table: &mut OffsetPageTable,
580        reason: PageFaultErrorCode,
581
582        addr: VirtAddr,
583        offset: usize,
584        _size: usize,
585    ) -> bool {
586        let mmap_file = self.file.as_mut().unwrap();
587        let mmap_page = mmap_file.file.inode().mmap_v2(offset).unwrap();
588
589        if reason.contains(PageFaultErrorCode::PROTECTION_VIOLATION) {
590            if reason.contains(PageFaultErrorCode::CAUSED_BY_WRITE) {
591                return false;
592            }
593        } else if let MMapPage::PageCache(page_cache) = &mmap_page {
594            mmap_file.mappings.insert(addr, page_cache.clone());
595        }
596
597        match mmap_page {
598            MMapPage::PageCache(page_cache) => {
599                if reason.contains(PageFaultErrorCode::CAUSED_BY_WRITE) {
600                    unsafe {
601                        offset_table.map_to(
602                            Page::containing_address(addr),
603                            page_cache.page(),
604                            PageTableFlags::PRESENT
605                                | PageTableFlags::USER_ACCESSIBLE
606                                | self.flags.into(),
607                        )
608                    }
609                    .unwrap()
610                    .flush();
611
612                    page_cache.mark_dirty();
613                } else {
614                    unsafe {
615                        offset_table.map_to(
616                            Page::containing_address(addr),
617                            page_cache.page(),
618                            PageTableFlags::PRESENT
619                                | PageTableFlags::USER_ACCESSIBLE
620                                | (self.flags & !VmFlag::WRITE).into(),
621                        )
622                    }
623                    .unwrap()
624                    .flush();
625                }
626            }
627
628            MMapPage::Direct(frame) => {
629                unsafe {
630                    offset_table.map_to(
631                        Page::containing_address(addr),
632                        frame,
633                        PageTableFlags::PRESENT
634                            | PageTableFlags::USER_ACCESSIBLE
635                            | self.flags.into(),
636                    )
637                }
638                .unwrap()
639                .flush();
640            }
641        }
642
643        true
644    }

646    /// Copies the contents of the `page` page to a newly allocated frame and maps it to
647    /// the `page` page with the provided `protection` protection flags.
648    fn map_copied(
649        offset_table: &mut OffsetPageTable,
650        page: Page<Size4KiB>,
651        flags: VmFlag,
652    ) -> Result<(), MapToError<Size4KiB>> {
653        // Allocate a new frame to hold the contents.
654        let new_frame: PhysFrame<Size4KiB> = FRAME_ALLOCATOR
655            .allocate_frame()
656            .expect("map_copied: failed to allocate frame");
657
658        let old_slice = unsafe {
659            let ptr = page.start_address().as_ptr::<u8>();
660            core::slice::from_raw_parts(ptr, Size4KiB::SIZE as _)
661        };
662
663        let new_slice = new_frame.as_slice_mut::<u8>();
664
665        // Copy the contents from the old frame to the newly allocated frame.
666        new_slice.copy_from_slice(old_slice);
667
668        // Re-map the page to the newly allocated frame and with the provided
669        // protection flags.
670        offset_table.unmap(page).unwrap().1.ignore();
671
672        // NOTE: We operate on an active page table, so we flush the changes.
673        unsafe {
674            offset_table
675                .map_to(
676                    page,
677                    new_frame,
678                    PageTableFlags::PRESENT | PageTableFlags::USER_ACCESSIBLE | flags.into(),
679                )?
680                .flush();
681        }
682
683        Ok(())
684    }

686    /// Handler routine for a COW (Copy-On-Write) pages. A COW page is shared between multiple
687    /// processes until a write occurs after which a private copy is made for the writing
688    /// process. A COW page is recognised because the VMA for the region is marked writable even
689    /// though the individual page table entry is not.
690    ///
691    /// ## Panics
692    /// * The provided `address` is not aligned to a page boundary.
693    fn handle_cow(
694        &mut self,
695        offset_table: &mut OffsetPageTable,
696        address: VirtAddr,
697        copy: bool,
698    ) -> bool {
699        debug_assert!(address.is_aligned(Size4KiB::SIZE));
700
701        let page: Page<Size4KiB> = Page::containing_address(address);
702
703        if let TranslateResult::Mapped { frame, .. } = offset_table.translate(address) {
704            let phys_addr = frame.start_address();
705
706            if let Some(vm_frame) = phys_addr.as_vm_frame() {
707                if vm_frame.ref_count() > 1 || copy {
708                    // This page is used by more then one process, so make it a private copy.
709                    Self::map_copied(offset_table, page, self.flags).unwrap();
710                } else {
711                    // This page is used by only one process, so make it writable.
712                    unsafe {
713                        offset_table.update_flags(
714                            page,
715                            PageTableFlags::PRESENT
716                                | PageTableFlags::USER_ACCESSIBLE
717                                | self.flags.into(),
718                        )
719                    }
720                    .unwrap()
721                    .flush();
722                }
723
724                return true;
725            }
726        }
727
728        false
729    }
730
731    fn unmap(
732        &mut self,
733        offset_table: &mut OffsetPageTable,
734        start: VirtAddr,
735        end: VirtAddr,
736    ) -> Result<UnmapResult, UnmapError> {
737        let mut unmap_range_inner = |range: Range<VirtAddr>| -> Result<(), UnmapError> {
738            for addr in range.step_by(Size4KiB::SIZE as usize) {
739                let page: Page = Page::containing_address(addr);
740                match offset_table.unmap(page) {
741                    Ok((_, flusher)) => flusher.flush(),
742                    Err(UnmapError::PageNotMapped) => {}
743                    Err(e) => return Err(e),
744                }
745            }
746
747            Ok(())
748        };
749
750        if end <= self.start_addr || start >= self.end_addr {
751            Ok(UnmapResult::None)
752        } else if start > self.start_addr && end < self.end_addr {
753            // The address we want to unmap is in the middle of the region. So we
754            // will need to split the mapping and update the end address accordingly.
755            unmap_range_inner(start..end)?;
756
757            let new_file = self.file.as_ref().map(|file| {
758                let offset = file.offset + (end - self.start_addr) as usize;
759                let size = file.size - (offset - file.offset);
760
761                MMapFile::new(file.file.clone(), offset, size)
762            });
763
764            let new_mapping = Mapping {
765                start_addr: end,
766                end_addr: end + (self.end_addr - end),
767                file: new_file,
768                refresh_flags: true,
769                flags: self.flags,
770            };
771
772            self.end_addr = start;
773            Ok(UnmapResult::Partial(new_mapping))
774        } else if start <= self.start_addr && end >= self.end_addr {
775            // We are unmapping the whole region.
776            unmap_range_inner(self.start_addr..self.end_addr)?;
777            Ok(UnmapResult::Full)
778        } else if start <= self.start_addr && end < self.end_addr {
779            unmap_range_inner(self.start_addr..end)?;
780
781            // Update the start address of the mapping since we have unmapped the
782            // first chunk of the mapping.
783            let offset = end - self.start_addr;
784
785            if let Some(file) = self.file.as_mut() {
786                file.offset += offset as usize;
787            }
788
789            self.start_addr = end;
790
791            Ok(UnmapResult::Start)
792        } else {
793            unmap_range_inner(start..self.end_addr)?;
794
795            // Update the end address of the mapping since we have unmapped the
796            // last chunk of the mapping.
797            self.end_addr = start;
798            Ok(UnmapResult::End)
799        }
800    }
801
802    fn size(&self) -> usize {
803        (self.end_addr - self.start_addr) as usize
804    }
805
806    fn split(&self, start: VirtAddr, end: VirtAddr) -> (Mapping, Mapping, Mapping) {
807        assert!(start > self.start_addr && end < self.end_addr);
808
809        let mut left = self.clone();
810        left.end_addr = start;
811
812        let mut mid = self.clone();
813        mid.start_addr = start;
814        mid.end_addr = end;
815
816        let mut right = self.clone();
817        right.start_addr = end;
818
819        if self.file.is_some() {
820            left.file.as_mut().unwrap().size = left.size();
821            mid.file.as_mut().unwrap().offset += left.size();
822            mid.file.as_mut().unwrap().size = mid.size();
823            right.file.as_mut().unwrap().offset += mid.size();
824            right.file.as_mut().unwrap().size = right.size();
825        }
826
827        (left, mid, right)
828    }
829}

842    fn handle_page_fault(
843        &mut self,
844        reason: PageFaultErrorCode,
845        accessed_address: VirtAddr,
846    ) -> bool {
847        if let Some(map) = self
848            .mappings
849            .iter_mut()
850            .find(|e| accessed_address >= e.start_addr && accessed_address < e.end_addr)
851        {
852            if map.protection().is_empty() {
853                return false;
854            }
855
856            if reason.contains(PageFaultErrorCode::CAUSED_BY_WRITE)
857                && !map.flags.contains(VmFlag::WRITE)
858            {
859                return false;
860            }
861
862            if reason.contains(PageFaultErrorCode::INSTRUCTION_FETCH)
863                && !map.flags.contains(VmFlag::EXEC)
864            {
865                return false;
866            }
867
868            let mut address_space = AddressSpace::this();
869            let mut offset_table = address_space.offset_page_table();
870
871            match (!map.flags.contains(VmFlag::SHARED), map.file.is_none()) {
872                (true, true) => {
873                    map.handle_pf_private_anon(&mut offset_table, reason, accessed_address)
874                }
875
876                (true | false, false) => {
877                    map.handle_pf_file(&mut offset_table, reason, accessed_address)
878                }
879
880                (false, true) => unreachable!("shared and anonymous mapping"),
881            }
882        } else {
883            log::trace!("mapping not found for address: {:#x}", accessed_address);
884            self.log();
885
886            // else the mapping does not exist, so return false.
887            false
888        }
889    }

891    fn find_fixed_mapping(
892        &mut self,
893        address: VirtAddr,
894        size: usize,
895    ) -> Option<(VirtAddr, CursorMut<Mapping>)> {
896        let mut cursor = self.mappings.cursor_front_mut();
897
898        while let Some(map) = cursor.current() {
899            if map.start_addr <= address && map.end_addr > address {
900                return None;
901            } else if map.start_addr < address {
902                cursor.move_next();
903            } else if address + size > map.start_addr {
904                return None;
905            } else {
906                break;
907            }
908        }
909
910        Some((address, cursor))
911    }
912
913    fn find_any_above(
914        &mut self,
915        address: VirtAddr,
916        size: usize,
917    ) -> Option<(VirtAddr, CursorMut<Mapping>)> {
918        if self.mappings.is_empty() {
919            return Some((address, self.mappings.cursor_front_mut()));
920        }
921
922        let mut cursor = self.mappings.cursor_back_mut();
923
924        // Search the mappings starting at the current cursor position for a big
925        // enough hole for where the address is above the provided `address`. A hole is
926        // big enough if it can hold the requested `size`. We use the first fit strategy,
927        // so it breaks as soon as a big enough hole is found.
928        while let Some(map) = cursor.current() {
929            let map_start = map.start_addr;
930
931            if map.start_addr < address {
932                cursor.move_next();
933            } else if let Some(pmap) = cursor.peek_prev() {
934                let start = core::cmp::max(address, pmap.end_addr);
935                let hole = map_start.as_u64() - start.as_u64();
936
937                if hole as usize >= size {
938                    return Some((start, cursor));
939                }
940
941                // The hole is too small
942                cursor.move_next();
943            } else {
944                let hole = map_start.as_u64() - address.as_u64();
945
946                return if hole as usize >= size {
947                    Some((address, cursor))
948                } else {
949                    // The hole is too small
950                    None
951                };
952            }
953        }
954
955        None
956    }

958    fn mmap(
959        &mut self,
960        address: VirtAddr,
961        size: usize,
962        flags: MMapFlags,
963        offset: usize,
964        file: Option<DirCacheItem>,
965        vm_flags: VmFlag,
966    ) -> Option<VirtAddr> {
967        let z = file.clone();
968
969        // Offset is required to be a multiple of page size.
970        if (offset as u64 & (Size4KiB::SIZE - 1)) != 0 {
971            log::warn!("mmap: offset is not a multiple of page size");
972            return None;
973        }
974
975        // The provided size should be non-zero.
976        if size == 0 {
977            log::warn!("mmap: size is zero");
978            return None;
979        }
980
981        if file.is_some() {
982            if flags.contains(MMapFlags::MAP_ANONYOMUS) {
983                log::warn!("mmap: cannot map a file with the anonymous flag");
984                return None;
985            }
986        } else {
987            if !flags.contains(MMapFlags::MAP_ANONYOMUS) {
988                log::warn!("mmap: mappings not backed by a file cannot be anonymous");
989                return None;
990            }
991
992            if flags.contains(MMapFlags::MAP_SHARED) {
993                log::warn!("mmap: anonymous mappings cannot be shared");
994                return None;
995            }
996        }
997
998        // TODO: align_up may overflow. return if size_aligned == 0
999        let size_aligned = align_up(size as _, Size4KiB::SIZE);
1000
1001        let x = if address == VirtAddr::zero() {
1002            // We need to find a free mapping above 0x7000_0000_0000.
1003            self.find_any_above(VirtAddr::new(0x7000_0000_0000), size_aligned as _)
1004        } else if flags.contains(MMapFlags::MAP_FIXED) {
1005            if !address.is_aligned(Size4KiB::SIZE) {
1006                log::warn!("mmap: fixed mapping address is not page aligned");
1007                return None;
1008            }
1009
1010            if (address + size_aligned) > userland_last_address() {
1011                log::warn!("mmap: fixed mapping address is out of range");
1012                return None;
1013            }
1014
1015            self.munmap(address, size_aligned as usize); // Unmap any existing mappings.
1016            self.find_fixed_mapping(address, size_aligned as _)
1017        } else {
1018            self.find_any_above(address, size)
1019        }
1020        .map(|(addr, mut cursor)| {
1021            // Merge same mappings instead of creating a new one.
1022            if let Some(prev) = cursor.peek_prev() {
1023                if prev.end_addr == addr
1024                    && prev.flags == vm_flags
1025                    && prev.file.is_none()
1026                    && file.is_none()
1027                {
1028                    prev.end_addr = addr + size_aligned;
1029                    return addr;
1030                }
1031            }
1032
1033            cursor.insert_before(Mapping {
1034                start_addr: addr,
1035                end_addr: addr + size_aligned,
1036
1037                file: file.map(|f| MMapFile::new(f, offset, size)),
1038                refresh_flags: true,
1039                flags: vm_flags,
1040            });
1041
1042            addr
1043        });
1044
1045        if x.is_none() {
1046            log::warn!("mmap failed");
1047            self.log();
1048
1049            dbg!(
1050                address,
1051                size,
1052                vm_flags,
1053                flags,
1054                offset,
1055                z.map(|f| f.absolute_path())
1056            );
1057
1058            crate::unwind::unwind_stack_trace();
1059        }
1060
1061        x
1062    }

1080                    mmap.flags,

1082            }

1086    fn load_bin<'header>(
1087        &mut self,
1088        bin: &DirCacheItem,
1089        argv: Option<ExecArgs>,
1090        envv: Option<ExecArgs>,
1091    ) -> Result<LoadedBinary<'header>, ElfLoadError> {
1092        // check for a shebang before proceeding.
1093        if let Some(shebang) = parse_shebang(bin)? {
1094            log::debug!(
1095                "shebang: (interpreter={}, argument={})",
1096                shebang.interpreter.absolute_path(),
1097                shebang.argument
1098            );
1099
1100            let mut largv = ExecArgs::default();
1101
1102            largv.push(shebang.interpreter.absolute_path().as_bytes());
1103
1104            if !shebang.argument.is_empty() {
1105                largv.push(shebang.argument.as_bytes());
1106            }
1107
1108            largv.push(bin.absolute_path().as_bytes());
1109
1110            if let Some(argv) = argv {
1111                largv.extend(&argv.inner[1..])
1112            }
1113
1114            return self.load_bin(&shebang.interpreter, Some(largv), envv);
1115        }
1116
1117        let elf = Elf::new(bin.clone())?;
1118        let header = &elf.header;
1119
1120        let load_offset = VirtAddr::new(
1121            if header.pt2.type_().as_type() == header::Type::SharedObject {
1122                0x4000_0000_u64
1123            } else {
1124                0u64
1125            },
1126        );
1127
1128        let mut entry_point = load_offset + header.pt2.entry_point();
1129
1130        log::debug!("entry point: {:#x}", entry_point);
1131        log::debug!("entry point type: {:?}", header.pt2.type_().as_type());
1132
1133        let mut base_addr = VirtAddr::zero();
1134
1135        for header in elf.program_iter() {
1136            let header_type = header
1137                .get_type()
1138                .expect("load_bin: failed to get program header type");
1139
1140            let header_flags = header.flags();
1141
1142            if header_type == xmas_elf::program::Type::Load {
1143                let virtual_start = VirtAddr::new(header.virtual_addr()).align_down(Size4KiB::SIZE)
1144                    + load_offset.as_u64();
1145
1146                if base_addr == VirtAddr::zero() {
1147                    base_addr = virtual_start;
1148                }
1149
1150                let virtual_end = VirtAddr::new(header.virtual_addr() + header.mem_size())
1151                    .align_up(Size4KiB::SIZE)
1152                    + load_offset.as_u64();
1153
1154                let virtual_fend = VirtAddr::new(header.virtual_addr() + header.file_size())
1155                    + load_offset.as_u64();
1156
1157                let data_size = virtual_fend - virtual_start;
1158                let aligned_data_size = align_up(data_size, Size4KiB::SIZE);
1159
1160                let file_offset = align_down(header.offset(), Size4KiB::SIZE);
1161
1162                let mut flags = VmFlag::MAY_READ | VmFlag::MAY_WRITE | VmFlag::MAY_EXEC;
1163
1164                if header_flags.is_read() {
1165                    flags.insert(VmFlag::READ);
1166                }
1167
1168                if header_flags.is_write() {
1169                    flags.insert(VmFlag::WRITE);
1170                }
1171
1172                if header_flags.is_execute() {
1173                    flags.insert(VmFlag::EXEC);
1174                }
1175
1176                #[rustfmt::skip]
1177                // The last non-bss frame of the segment consists partly of data and partly of bss
1178                // memory, which must be zeroed. Unfortunately, the file representation might have
1179                // reused the part of the frame that should be zeroed to store the next segment. This
1180                // means that we can't simply overwrite that part with zeroes, as we might overwrite
1181                // other data this way.
1182                //
1183                // Example:
1184                //
1185                //  XXXXXXXXXXXXXXX000000YYYYYYY000ZZZZZZZZZZZ     virtual memory (XYZ are data)
1186                //  |·············|     /·····/   /·········/
1187                //  |·············| ___/·····/   /·········/
1188                //  |·············|/·····/‾‾‾   /·········/
1189                //  |·············||·····|/·̅·̅·̅·̅·̅·····/‾‾‾‾
1190                //  XXXXXXXXXXXXXXXYYYYYYYZZZZZZZZZZZ              file memory (zeros are not saved)
1191                //  '       '       '       '        '
1192                //  The areas filled with dots (`·`) indicate a mapping between virtual and file
1193                //  memory. We see that the data regions `X`, `Y`, `Z` have a valid mapping, while
1194                //  the regions that are initialized with 0 have not.
1195                //
1196                //  The ticks (`'`) below the file memory line indicate the start of a new frame. We
1197                //  see that the last frames of the `X` and `Y` regions in the file are followed
1198                //  by the bytes of the next region. So we can't zero these parts of the frame
1199                //  because they are needed by other memory regions.
1200                let address = self
1201                    .mmap(
1202                        virtual_start,
1203                        data_size as usize,
1204                        MMapFlags::MAP_PRIVATE | MMapFlags::MAP_FIXED,
1205                        file_offset as usize,
1206                        Some(bin.clone()),
1207                        flags
1208                    )
1209                    .ok_or(ElfLoadError::MemoryMapError)?;
1210
1211                let virtual_fend = address + aligned_data_size;
1212
1213                if virtual_fend < virtual_end {
1214                    let bss_size = virtual_end - virtual_fend;
1215
1216                    self.mmap(
1217                        virtual_fend,
1218                        bss_size as usize,
1219                        MMapFlags::MAP_PRIVATE | MMapFlags::MAP_ANONYOMUS | MMapFlags::MAP_FIXED,
1220                        0,
1221                        None,
1222                        flags,
1223                    )
1224                    .ok_or(ElfLoadError::MemoryMapError)?;
1225                }
1226            } else if header_type == xmas_elf::program::Type::Tls {
1227            } else if header_type == xmas_elf::program::Type::Interp {
1228                let ld = fs::lookup_path(fs::Path::new("/usr/lib/ld.so")).unwrap();
1229
1230                let res = self.load_bin(&ld, None, None)?;
1231                entry_point = res.entry_point;
1232            }
1233        }
1234
1235        Ok(LoadedBinary {
1236            elf,
1237            entry_point,
1238
1239            base_addr,
1240
1241            argv,
1242            envv,
1243        })
1244    }

1252    fn munmap(&mut self, address: VirtAddr, size: usize) -> bool {
1253        let start = address.align_up(Size4KiB::SIZE);
1254        let end = (address + size).align_up(Size4KiB::SIZE);
1255
1256        let mut cursor = self.mappings.cursor_front_mut();
1257        let mut success = false;
1258
1259        let mut address_space = AddressSpace::this();
1260        let mut offset_table = address_space.offset_page_table();
1261
1262        while let Some(map) = cursor.current() {
1263            if map.end_addr <= start {
1264                cursor.move_next();
1265            } else {
1266                match map.unmap(&mut offset_table, start, end) {
1267                    Ok(result) => match result {
1268                        UnmapResult::None => return success,
1269                        UnmapResult::Start => return true,
1270
1271                        UnmapResult::Full => {
1272                            success = true;
1273                            cursor.remove_current();
1274                        }
1275
1276                        UnmapResult::Partial(mapping) => {
1277                            cursor.insert_after(mapping);
1278                            return true;
1279                        }
1280
1281                        UnmapResult::End => {
1282                            success = true;
1283                            cursor.move_next();
1284                        }
1285                    },
1286
1287                    Err(_) => return false,
1288                }
1289            }
1290        }
1291
1292        success
1293    }

1295    fn mprotect(
1296        &mut self,
1297        addr: VirtAddr,
1298        size: usize,
1299        prot: MMapProt,
1300    ) -> aero_syscall::Result<()> {
1301        let start = addr.align_up(Size4KiB::SIZE);
1302        let end = (addr + size).align_up(Size4KiB::SIZE);
1303
1304        let mut cursor = self.mappings.cursor_front_mut();
1305
1306        while let Some(map) = cursor.current() {
1307            if map.end_addr <= start {
1308                cursor.move_next();
1309            } else if end <= map.start_addr || start >= map.end_addr {
1310                break;
1311            } else if start > map.start_addr && end < map.end_addr {
1312                // The address we want to unmap is in the middle of the region. So we
1313                // will need to split the mapping and update the end address accordingly.
1314                let (left, mut mid, right) = map.split(start, end);
1315                mid.set_protection(prot)?;
1316
1317                cursor.insert_after(right);
1318                cursor.insert_after(mid);
1319                cursor.insert_after(left);
1320                cursor.remove_current();
1321                break;
1322            } else if start <= map.start_addr && end >= map.end_addr {
1323                // full
1324                map.set_protection(prot)?;
1325                cursor.move_next();
1326            } else if start <= map.start_addr && end < map.end_addr {
1327                // start
1328                let mut mapping = map.clone();
1329                mapping.end_addr = end;
1330                mapping.set_protection(prot)?;
1331
1332                map.start_addr = end;
1333                cursor.insert_before(mapping);
1334                break;
1335            } else {
1336                // end
1337                let mut mapping = map.clone();
1338                mapping.start_addr = start;
1339                mapping.set_protection(prot)?;
1340
1341                map.end_addr = start;
1342                cursor.insert_after(mapping);
1343                cursor.move_next();
1344            }
1345        }
1346
1347        Ok(())
1348    }

1350    #[must_use]
1351    fn fork_from(&mut self, parent: &Vm) -> AddressSpace {
1352        {
1353            let parent = parent.inner.lock();
1354            self.mappings.clone_from(&parent.mappings);
1355        }
1356
1357        let mut address_space = AddressSpace::new().unwrap();
1358        let mut offset_table = address_space.offset_page_table();
1359
1360        let mut current = AddressSpace::this();
1361        let mut current = current.offset_page_table();
1362
1363        for map in self.mappings.iter().filter(|map| {
1364            // Do not copy page table entries where a page fault can map them correctly.
1365            !map.flags.contains(VmFlag::SHARED) && map.flags.contains(VmFlag::MAY_WRITE)
1366        }) {
1367            offset_table.copy_page_range(&mut current, map.start_addr..=map.end_addr);
1368        }
1369
1370        address_space
1371    }

1386    pub fn mmap(
1387        &self,
1388        address: VirtAddr,
1389        size: usize,
1390        protection: MMapProt,
1391        flags: MMapFlags,
1392        offset: usize,
1393        file: Option<Arc<FileHandle>>,
1394    ) -> Option<VirtAddr> {
1395        let mut vm_flags =
1396            VmFlag::from(protection) | VmFlag::MAY_READ | VmFlag::MAY_WRITE | VmFlag::MAY_EXEC;
1397
1398        let map_type = flags & (MMapFlags::MAP_SHARED | MMapFlags::MAP_PRIVATE);
1399
1400        match (map_type, file.as_ref()) {
1401            (MMapFlags::MAP_SHARED, Some(file)) => {
1402                vm_flags.insert(VmFlag::SHARED);
1403
1404                if !file.is_writable() {
1405                    if protection.contains(MMapProt::PROT_WRITE) {
1406                        return None; // EACCES
1407                    }
1408
1409                    // The mapping is going to be read-only forever so, it can be converted into a
1410                    // private mapping.
1411                    vm_flags.remove(VmFlag::MAY_WRITE | VmFlag::SHARED);
1412                }
1413
1414                if !file.is_readable() {
1415                    return None; // EACCES
1416                }
1417
1418                // TODO: * check if the filsystem is noexec mounted and remove the MAY_EXEC flag.
1419                //       * error out if prot contains PROT_EXEC & filesystem is noexec.
1420            }
1421
1422            (MMapFlags::MAP_PRIVATE, Some(file)) => {
1423                if !file.is_readable() {
1424                    return None; // EACCES
1425                }
1426
1427                // TODO: * check if the filsystem is noexec mounted and remove the MAY_EXEC flag.
1428                //       * error out if prot contains PROT_EXEC & filesystem is noexec.
1429            }
1430
1431            (MMapFlags::MAP_SHARED, None) => vm_flags.insert(VmFlag::SHARED),
1432            _ => {}
1433        }
1434
1435        let file = file.map(|file| file.dirnode());
1436        self.inner
1437            .lock()
1438            .mmap(address, size, flags, offset, file, vm_flags)
1439    }
