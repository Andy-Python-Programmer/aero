18use core::num::ParseIntError;
19
20use spin::Once;
21
22use limine::file::File;
23
24use crate::rendy;
25
26static RAW_CMDLINE_STR: Once<&'static str> = Once::new();
27
28pub struct CommandLine {
29    /// If set, then the kernel logs will be redirected onto the framebuffer until
30    /// the kernel thread jumps to userland.
31    ///
32    /// By default, the kernel logs are not redirected.
33    pub rendy_debug: bool,
34    pub term_background: Option<&'static [u8]>,
35    pub theme_background: u32,
36}
37
38impl CommandLine {
39    fn new() -> Self {
40        Self {
41            rendy_debug: false,
42            term_background: None,
43            theme_background: rendy::DEFAULT_THEME_BACKGROUND,
44        }
45    }
46}
47
48fn resolve_module(modules: &[&File], name: &str) -> &'static [u8] {
49    modules
50        .iter()
51        .find(|m| {
52            let n = core::str::from_utf8(m.cmdline()).unwrap();
53            n == name
54        })
55        .map(|m| unsafe { core::slice::from_raw_parts(m.addr(), m.size() as usize) })
56        .expect("resolve_module: invalid operand")
57}
58
59fn parse_number(mut string: &str) -> Result<usize, ParseIntError> {
60    let is_hex = string.starts_with("0x");
61    let is_octal = string.starts_with("0o");
62
63    if is_hex {
64        string = string.trim_start_matches("0x");
65        usize::from_str_radix(string, 16)
66    } else if is_octal {
67        string = string.trim_start_matches("0o");
68        usize::from_str_radix(string, 8)
69    } else {
70        string.parse::<usize>()
71    }
72}
73
74pub fn parse(cmdline: &'static str, modules: &[&File]) -> CommandLine {
75    RAW_CMDLINE_STR.call_once(|| cmdline);
76
77    // Chew up the leading spaces.
78    let cmdline = cmdline.trim();
79    let mut result = CommandLine::new();
80
81    let bail = |argument| log::warn!("unknown kernel command line option: '{}'", argument);
82
83    for argument in cmdline.split_whitespace() {
84        match argument {
85            "rendy-dbg" => result.rendy_debug = true,
86
87            _ => {
88                let mut pair = argument.splitn(2, '=');
89
90                match pair.next() {
91                    Some(name) => {
92                        let value = pair.next().expect("missing operand");
93
94                        match name {
95                            "term-background" => {
96                                result.term_background = Some(resolve_module(modules, value))
97                            }
98
99                            "theme-background" => {
100                                let theme_bg = parse_number(value).unwrap_or_else(|e| {
101                                    log::warn!(
102                                        "parse_number: invalid operand {}, defaulting to {}",
103                                        e,
104                                        rendy::DEFAULT_THEME_BACKGROUND
105                                    );
106
107                                    rendy::DEFAULT_THEME_BACKGROUND as usize
108                                });
109
110                                result.theme_background = theme_bg as u32;
111                            }
112
113                            _ => bail(argument),
114                        }
115                    }
116
117                    None => bail(argument),
118                }
119            }
120        }
121    }
122
123    result
124}
125
126/// Returns the raw kernel command line string.
127///
128/// ## Panics
129/// * If this function was invoked before the kernel command line was parsed using [`self::parse`].
130pub fn get_raw_cmdline() -> &'static str {
131    RAW_CMDLINE_STR
132        .get()
133        .expect("get_raw_cmdline: called before cmdline was parsed")
134}

136#[cfg(test)]
137mod tests {
138    use super::*;
139
140    #[test]
141    fn number_parser_test() {
142        assert_eq!(parse_number("0xdeadbeef").unwrap(), 0xdeadbeef);
143        assert_eq!(parse_number("0o546").unwrap(), 0o546);
144        assert_eq!(parse_number("123").unwrap(), 123);
145
146        assert!(parse_number("invalid").is_err());
147        assert!(parse_number("0xinvalid").is_err());
148        assert!(parse_number("0oinvalid").is_err());
149    }
