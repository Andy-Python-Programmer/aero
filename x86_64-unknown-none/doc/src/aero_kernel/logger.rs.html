18use core::fmt::Write;
19use core::sync::atomic::{AtomicBool, Ordering};
20
21use log::{Level, LevelFilter, Metadata, Record};
22use spin::Once;
23
24use crate::userland::scheduler;
25use crate::utils::buffer::RingBuffer;
26use crate::utils::sync::Mutex;
27
28const DEFAULT_LOG_RING_BUFFER_SIZE: usize = 4096;
29
30static LOG_RING_BUFFER: Once<Mutex<RingBuffer<[u8; DEFAULT_LOG_RING_BUFFER_SIZE]>>> = Once::new();
31static LOGGER: AeroLogger = AeroLogger;
32
33static RENDY_DEBUG: AtomicBool = AtomicBool::new(false);
34
35struct AeroLogger;
36
37impl log::Log for AeroLogger {
38    fn enabled(&self, metadata: &Metadata) -> bool {
39        metadata.level() <= Level::Trace
40    }
41
42    fn log(&self, record: &Record) {
43        if self.enabled(record.metadata()) {
44            use crate::drivers::uart::*;
45
46            let file = record.file().unwrap_or("unknown");
47            let file = file.strip_prefix("aero_kernel/src/").unwrap_or(file);
48
49            let line = record.line().unwrap_or(0);
50
51            let level = record.level();
52            let rendy_dbg = RENDY_DEBUG.load(Ordering::Relaxed);
53
54            macro generic_log($($arg:tt)*) {
55                {
56                    serial_print!("{}", format_args!($($arg)*));
57                    if rendy_dbg {
58                        $crate::rendy::print!("{}", format_args!($($arg)*));
59                    }
60                }
61            }
62
63            // Append the log message to the log ring buffer.
64            let mut log_ring = LOG_RING_BUFFER.get().unwrap().lock_irq();
65            let _ = writeln!(log_ring, "[{}] {}", level, record.args());
66
67            let ticks = crate::arch::time::get_uptime_ticks();
68            serial_print!("\x1b[37;1m[{}] {file}:{line} ", ticks);
69
70            if scheduler::is_initialized() {
71                // fetch the current task, grab the TID and PID.
72                if let Some(task) = scheduler::get_scheduler().inner.current_task_optional() {
73                    serial_print!(
74                        "(tid={}, pid={}) ",
75                        task.tid().as_usize(),
76                        task.pid().as_usize()
77                    );
78                }
79            }
80
81            match record.level() {
82                Level::Info => generic_log!("\x1b[32;1minfo "), // green info
83                Level::Warn => generic_log!("\x1b[33;1mwarn "), // yellow warn
84                Level::Error => generic_log!("\x1b[32;1merror "), // red error
85                Level::Debug => generic_log!("\x1b[35;1mdebug "), // gray debug
86                Level::Trace => generic_log!("\x1b[34;1mtrace "), // blue trace
87            }
88
89            generic_log!("\x1b[0m");
90            generic_log!("{}\n", record.args());
91        }
92    }
93
94    fn flush(&self) {}
95}
96
97/// Force-unlocks the logger ring buffer to prevent a deadlock.
98///
99/// ## Safety
100/// This method is not memory safe and should be only used when absolutely necessary.
101#[inline]
102pub unsafe fn force_unlock() {
103    if let Some(l) = LOG_RING_BUFFER.get() {
104        l.force_unlock()
105    }
106}
107
108pub fn get_log_buffer() -> String {
109    LOG_RING_BUFFER
110        .get()
111        .map(|l| String::from(l.lock_irq().extract()))
112        .expect("log: attempted to get the log ring buffer before it was initialized")
113}
114
115#[inline]
116pub fn enabled_rendy_debug() -> bool {
117    RENDY_DEBUG.load(Ordering::SeqCst)
118}
119
120#[inline]
121pub fn set_rendy_debug(yes: bool) {
122    RENDY_DEBUG.store(yes, Ordering::SeqCst);
123}
124
125pub fn init() {
126    LOG_RING_BUFFER.call_once(|| Mutex::new(RingBuffer::new([0; DEFAULT_LOG_RING_BUFFER_SIZE])));
127
128    log::set_logger(&LOGGER)
129        .map(|()| log::set_max_level(LevelFilter::Trace))
130        .unwrap();
131}
