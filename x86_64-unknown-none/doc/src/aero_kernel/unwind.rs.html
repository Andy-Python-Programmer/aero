34static PANIC_HOOK_READY: AtomicBool = AtomicBool::new(false);
35pub static UNWIND_INFO: spin::Once<UnwindInfo> = spin::Once::new();

37pub fn set_panic_hook_ready(yes: bool) {
38    PANIC_HOOK_READY.store(yes, Ordering::SeqCst);
39}

41pub fn prepare_panic() {
42    // Disable interrupts as we do not want to be interrupted while
43    // we are unwinding the stack.
44    unsafe {
45        interrupts::disable_interrupts();
46    }
47
48    // Force unlock rendy and the logger ring buffer to prevent deadlock while
49    // unwinding.
50    unsafe {
51        rendy::force_unlock();
52        logger::force_unlock();
53    }
54
55    // Clear the screen if the rendy is initialized and enable
56    // rendy debug in logger.
57    if rendy::is_initialized() {
58        rendy::clear_screen(true);
59        logger::set_rendy_debug(true);
60    }
61}

73pub fn unwind_stack_trace() {
74    let _guard = IrqGuard::new();
75
76    let mut address_space = AddressSpace::this();
77    let offset_table = address_space.offset_page_table();
78
79    let kernel_elf = &UNWIND_INFO.get().unwrap().kernel_elf;
80    let mut symbol_table = None;
81
82    for section in kernel_elf.section_iter() {
83        if section.get_type() == Ok(ShType::SymTab) {
84            let section_data = section
85                .get_data(kernel_elf)
86                .expect("Failed to get kernel section data information");
87
88            if let SectionData::SymbolTable64(symtab) = section_data {
89                symbol_table = Some(symtab);
90            }
91        }
92    }
93
94    let symbol_table = symbol_table.unwrap();
95    let mut rbp: usize;
96
97    unsafe {
98        asm!("mov {}, rbp", out(reg) rbp);
99    }
100
101    // Make sure the RBP is not NULL. If it is then we cannot do the stack unwinding/tracing
102    // as no frame pointers were emitted in this build. This should only occur if you
103    // set the field `eliminate-frame-pointer` in the target file to true or manually resetting
104    // the RBP to prevent backtrace to avoid address leaks (for example when jumping to userland).
105    // If that's the case we return (resumes the parent function).
106    if rbp == 0x00 {
107        log::trace!("<empty backtrace>");
108        return;
109    }
110
111    log::trace!("{:-^80}", " BACKTRACE ");
112
113    for depth in 0../*64*/16 {
114        if let Some(rip_rbp) = rbp.checked_add(core::mem::size_of::<usize>()) {
115            let rip_rbp = VirtAddr::new(rip_rbp as u64);
116
117            if offset_table.translate_addr(rip_rbp).is_none() || !rip_rbp.is_canonical() {
118                log::trace!("{:>2}: <guard page>", depth);
119                break;
120            }
121
122            let rip = unsafe { *(rip_rbp.as_ptr::<usize>()) };
123
124            if rip == 0 {
125                break;
126            }
127
128            unsafe {
129                rbp = *(rbp as *const usize);
130            }
131
132            let mut name = None;
133
134            for data in symbol_table {
135                let st_value = data.value() as usize;
136                let st_size = data.size() as usize;
137
138                if rip >= st_value && rip < (st_value + st_size) {
139                    let mangled_name = data.get_name(kernel_elf).unwrap_or("<unknown>");
140                    let demangled_name = rustc_demangle::demangle(mangled_name);
141
142                    name = Some(demangled_name);
143                }
144            }
145
146            if let Some(name) = name {
147                log::trace!("{:>2}: 0x{:016x} - {}", depth, rip, name);
148            } else if scheduler::is_initialized() {
149                if let Some((region, tag)) = scheduler::current_thread()
150                    .mem_tags
151                    .lock()
152                    .iter()
153                    .find(|(region, _tag)| region.contains(&rip))
154                {
155                    let resolved_addr = rip - region.start;
156                    log::trace!(
157                        "{depth:>2}: 0x{rip:016x} - <userland, in={tag}, resolved_addr={resolved_addr:#x}>"
158                    );
159                } else {
160                    log::trace!("{depth:>2}: 0x{rip:016x} - <unknown>");
161                }
162            } else {
163                log::trace!("{depth:>2}: 0x{rip:016x} - <unknown>");
164            }
165        } else {
166            // RBP has been overflowed...
167            break;
168        }
169    }
170
171    // let thread = scheduler::current_thread();
172    // let tags = thread.mem_tags.lock();
173    // let mut tags = tags.iter().collect::<alloc::vec::Vec<_>>();
174    // tags.sort_by(|x, y| x.0.start.cmp(&y.0.start));
175    // for (x, s) in tags.iter() {
176    //     log::trace!("{:#x}..{:#x} {s}", x.start, x.end);
177    // }
178}

184#[panic_handler]
185fn rust_begin_unwind(info: &PanicInfo) -> ! {
186    prepare_panic();
187
188    let message = info.message();
189    let location = info.location().unwrap();
190
191    // Get the CPU ID where this panic happened and if PANIC_HOOK_READY is false
192    // then we cannot get the CPU where this panic happened.
193    let cpu_id = if PANIC_HOOK_READY.load(Ordering::SeqCst) {
194        crate::arch::tls::get_cpuid()
195    } else {
196        0x00
197    };
198
199    log::error!("cpu '{cpu_id}' panicked at {location}:");
200    log::error!("{message}");
201    log::error!("");
202
203    unwind_stack_trace();
204
205    #[cfg(feature = "ci")]
206    emu::exit_qemu(emu::ExitStatus::Success);
207
208    #[cfg(not(feature = "ci"))]
209    unsafe {
210        // Go into a halt loop to to save power.
211        loop {
212            interrupts::halt();
213        }
214    }
215}

217/// This function is automatically called after each unwinding cleanup routine finishes
218/// executing. Our task here is to *resume* the unwinding procedure by figuring out where
219/// we just came from and picking up where we left off.
220#[allow(non_snake_case)]
221#[no_mangle]
222extern "C" fn _Unwind_Resume(unwind_context_ptr: usize) -> ! {
223    log::debug!("{}", unwind_context_ptr);
224
225    unsafe {
226        interrupts::disable_interrupts();
227
228        loop {
229            interrupts::halt();
230        }
231    }
232}
