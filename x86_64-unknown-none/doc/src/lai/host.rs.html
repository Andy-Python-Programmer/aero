18macro_rules! marker {
19    ($(fn $name:tt(&self, $($pname:tt: $ptyp:ty),*) -> $ret:ty);*;) => {
20        $(fn $name(&self, $($pname: $ptyp),*) -> $ret { unimplemented!() })*
21    };
22}

24pub trait Host {
25    marker!(
26        fn scan(&self, _signature: &str, _index: usize) -> *const u8;
27        fn sleep(&self, _ms: u64) -> ();
28
29        // Port I/O functions:
30        fn outb(&self, _port: u16, _value: u8) -> ();
31        fn outw(&self, _port: u16, _value: u16) -> ();
32        fn outd(&self, _port: u16, _value: u32) -> ();
33
34        fn inb(&self, _port: u16) -> u8;
35        fn inw(&self, _port: u16) -> u16;
36        fn ind(&self, _port: u16) -> u32;
37
38        // PCI read functions:
39        fn pci_readb(&self, _seg: u16, _bus: u8, _slot: u8, _fun: u8, _offset: u16) -> u8;
40        fn pci_readw(&self, _seg: u16, _bus: u8, _slot: u8, _fun: u8, _offset: u16) -> u16;
41        fn pci_readd(&self, _seg: u16, _bus: u8, _slot: u8, _fun: u8, _offset: u16) -> u32;
42
43        // Maps count bytes from the given physical address and returns
44        // a pointer that can be used to access the memory.
45        fn map(&self, _address: usize, _count: usize) -> *mut u8;
46    );
47
48    unsafe fn alloc(&self, size: usize) -> *mut u8 {
49        let layout = Layout::from_size_align_unchecked(size, 16);
50        alloc::alloc::alloc_zeroed(layout)
51    }
52
53    unsafe fn dealloc(&self, ptr: *mut u8, size: usize) {
54        let layout = Layout::from_size_align_unchecked(size, 16);
55        alloc::alloc::dealloc(ptr, layout);
56    }
57
58    unsafe fn realloc(&self, ptr: *mut u8, new_size: usize, old_size: usize) -> *mut u8 {
59        let layout = Layout::from_size_align_unchecked(old_size, 16);
60        alloc::alloc::realloc(ptr, layout, new_size)
61    }
62}

64pub fn init(host: Arc<dyn Host>) {
65    unsafe {
66        assert!(LAI_HOST.is_none());
67        LAI_HOST = Some(host);
68    }
69}

72const LAI_DEBUG_LOG: i32 = 1;
73const LAI_WARN_LOG: i32 = 2;
74
75#[no_mangle]
76extern "C" fn laihost_log(level: i32, message: *const u8) {
77    let message = unsafe { c_str_as_str(message) };
78
79    match level {
80        LAI_DEBUG_LOG => log::debug!("{message}"),
81        LAI_WARN_LOG => log::warn!("{message}"),
82
83        _ => unreachable!("undefined log level: (message={message}, level={level})"),
84    };
85}
86
87#[no_mangle]
88extern "C" fn laihost_panic(message: *const u8) -> ! {
89    let message = unsafe { c_str_as_str(message) };
90    panic!("{message}");
91}
