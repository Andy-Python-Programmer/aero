17    /// An IPv4 address representing the broadcast address: `255.255.255.255`
18    ///
19    /// # Examples
20    ///
21    /// ```
22    /// use crabnet::network::Ipv4Addr;
23    /// assert_eq!(Ipv4Addr::BROADCAST, Ipv4Addr::new(255, 255, 255, 255));
24    /// ```
25    pub const BROADCAST: Self = Self([0xff; Self::ADDR_SIZE]);
26    /// Loopback address.
27    pub const LOOPBACK: Self = Self([127, 0, 0, 1]);
28    // FIXME(andypython): [`Ipv4Addr::NULL`] should not really be used anywhere except for testing.
29    // Consider removing it to avoid misuse/confusion.
30    pub const NULL: Self = Self([0; Self::ADDR_SIZE]);
31
32    /// Creates a new IPv4 address from four eight-bit octets.
33    ///
34    /// The result will represent the IP address `a`.`b`.`c`.`d`.
35    ///
36    /// # Examples
37    ///
38    /// ```
39    /// use crabnet::network::Ipv4Addr;
40    ///
41    /// let addr = Ipv4Addr::new(127, 0, 0, 1);
42    /// ```
43    #[must_use]
44    #[inline]
45    pub const fn new(a: u8, b: u8, c: u8, d: u8) -> Self {
46        Self([a, b, c, d])
47    }
48
49    /// Returns [`true`] if this is a broadcast address (`255.255.255.255`).
50    ///
51    /// A broadcast address has all octets set to `255` as defined in [IETF RFC 919].
52    ///
53    /// [IETF RFC 919]: https://tools.ietf.org/html/rfc919
54    ///
55    /// # Examples
56    ///
57    /// ```
58    /// use std::net::Ipv4Addr;
59    ///
60    /// assert_eq!(Ipv4Addr::new(255, 255, 255, 255).is_broadcast(), true);
61    /// assert_eq!(Ipv4Addr::new(236, 168, 10, 65).is_broadcast(), false);
62    /// ```
63    #[must_use]
64    #[inline]
65    pub const fn is_broadcast(&self) -> bool {
66        // FIXME: Equality check `self.0 == self.BROADCAST.0` does not work in const contexts.
67        u32::from_be_bytes(self.octets()) == u32::from_be_bytes(Self::BROADCAST.octets())
68    }
69
70    /// Returns `true` if `self` and `other` belong to the same subnet.
71    ///
72    /// ## Example
73    ///
74    /// ```rust
75    /// use crabnet::network::Ipv4Addr;
76    ///
77    /// let subnet_mask = Ipv4Addr::new(255, 255, 255, 0);
78    ///
79    /// let x = Ipv4Addr::new(192, 168, 1, 1);
80    /// let y = Ipv4Addr::new(192, 168, 1, 2);
81    /// let z = Ipv4Addr::new(192, 168, 2, 1);
82    ///
83    /// assert_eq!(x.is_same_subnet(y, subnet_mask), true);
84    /// assert_eq!(x.is_same_subnet(z, subnet_mask), false);
85    /// assert_eq!(y.is_same_subnet(z, subnet_mask), false);
86    /// ```
87    #[must_use]
88    pub const fn is_same_subnet(&self, other: Ipv4Addr, subnet_mask: Ipv4Addr) -> bool {
89        let subnet = u32::from_be_bytes(subnet_mask.octets());
90
91        let x = u32::from_be_bytes(other.octets());
92        let y = u32::from_be_bytes(self.octets());
93
94        (x & subnet) == (y & subnet)
95    }
96
97    /// Returns the four eight-bit integers that make up this address.
98    ///
99    /// ## Examples
100    ///
101    /// ```rust
102    /// use crabnet::network::Ipv4Addr;
103    ///
104    /// let addr = Ipv4Addr::new(192, 168, 1, 1);
105    /// assert_eq!(addr.octets(), [192, 168, 1, 1]);
106    /// ```
107    #[must_use]
108    #[inline]
109    pub const fn octets(&self) -> [u8; Self::ADDR_SIZE] {
110        self.0
111    }
112}
