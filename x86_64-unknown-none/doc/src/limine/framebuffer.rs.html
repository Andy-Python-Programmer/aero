39/// A memory model used by a framebuffer. Currently only
40/// [`MemoryModel::RGB`](Self::RGB) is defined.
41#[repr(transparent)]
42#[derive(PartialEq, Eq, Clone, Copy)]
43pub struct MemoryModel(u8);
44impl MemoryModel {
45    /// This is an RGB framebuffer.
46    pub const RGB: Self = Self(1);
47}

49/// A mode supported by the current framebuffer.
50#[repr(C)]
51pub struct VideoMode {
52    /// The pitch (distance between rows, in bytes). This is not always the same
53    /// as `(width * bpp) / 8`, as padding bytes may be added to achieve a
54    /// particular alignment.
55    pub pitch: u64,
56    /// The width of the framebuffer, in pixels.
57    pub width: u64,
58    /// The height of the framebuffer, in pixels.
59    pub height: u64,
60    /// The number of **bits** (*not bytes*) per single pixel in the framebuffer.
61    pub bpp: u16,
62    /// The memory model used by this video mode. See [`MemoryModel`] for
63    /// specific values.
64    pub memory_model: MemoryModel,
65    /// The size of the red mask, in bits. This part of the mask can be applied
66    /// with `red_value & ((1 << red_mask_size) - 1)`.
67    pub red_mask_size: u8,
68    /// The number of bits to shift the red mask to the left. This part of the
69    /// mask can be applied with `red_value << red_mask_shift`.
70    pub red_mask_shift: u8,
71    /// The size of the green mask, in bits. This part of the mask can be
72    /// applied with `green_value & ((1 << green_mask_size) - 1)`.
73    pub green_mask_size: u8,
74    /// The number of bits to shift the green mask to the left. This part of the
75    /// mask can be applied with `green_value << green_mask_shift`.
76    pub green_mask_shift: u8,
77    /// The size of the blue mask, in bits. This part of the mask can be applied
78    /// with `blue_value & ((1 << blue_mask_size) - 1)`.
79    pub blue_mask_size: u8,
80    /// The number of bits to shift the blue mask to the left. This part of the
81    /// mask can be applied with `blue_value << blue_mask_shift`.
82    pub blue_mask_shift: u8,
83}

85/// A pointer to a framebuffer.
86///
87/// # Why is this a wrapper type?
88/// Two revisions currently exist of the framebuffer type. However, the type
89/// itself has no revision field. In order to keep this type safe, we wrap the
90/// pointer with its associated revision taken from the response.
91pub struct Framebuffer<'a> {
92    revision: u64,
93    inner: &'a RawFramebuffer,
94}
95impl<'a> Framebuffer<'a> {
96    pub(crate) fn new(revision: u64, inner: &'a RawFramebuffer) -> Self {
97        Self { revision, inner }
98    }
99
100    /// The address of the framebuffer. Note that no synchronization is done on
101    /// this pointer, so you yourself must synchronize all access. In addition,
102    /// the pointer may point to uninitialized bytes at boot, so dereferencing
103    /// it will cause UB.
104    pub fn addr(&self) -> *mut u8 {
105        unsafe { self.inner.v0 }.addr.cast()
106    }
107
108    /// The width of the framebuffer in its current mode.
109    pub fn width(&self) -> u64 {
110        unsafe { self.inner.v0 }.width
111    }
112    /// The height of the framebuffer in its current mode.
113    pub fn height(&self) -> u64 {
114        unsafe { self.inner.v0 }.height
115    }
116
117    /// The pitch (distance between rows, in bytes) of the framebuffer in the
118    /// current mode. This is not always the same as `(width * bpp) / 8`, as
119    /// padding bytes may be added to achieve a particular alignment.
120    pub fn pitch(&self) -> u64 {
121        unsafe { self.inner.v0 }.pitch
122    }
123    /// The number of **bits** (*not bytes*) per pixel of the framebuffer in the
124    /// current mode.
125    pub fn bpp(&self) -> u16 {
126        unsafe { self.inner.v0 }.bpp
127    }
128
129    /// The memory model of the framebuffer in the current mode. See
130    /// [`MemoryModel`] for specific values.
131    pub fn memory_model(&self) -> MemoryModel {
132        unsafe { self.inner.v0 }.memory_model
133    }
134
135    /// The size of the red mask, in bits. This part of the mask can be applied
136    /// with `red_value & ((1 << red_mask_size) - 1)`.
137    pub fn red_mask_size(&self) -> u8 {
138        unsafe { self.inner.v0 }.red_mask_size
139    }
140    /// The number of bits to shift the red mask to the left. This part of the
141    /// mask can be applied with `red_value << red_mask_shift`.
142    pub fn red_mask_shift(&self) -> u8 {
143        unsafe { self.inner.v0 }.red_mask_shift
144    }
145    /// The size of the green mask, in bits. This part of the mask can be
146    /// applied with `green_value & ((1 << green_mask_size) - 1)`.
147    pub fn green_mask_size(&self) -> u8 {
148        unsafe { self.inner.v0 }.green_mask_size
149    }
150    /// The number of bits to shift the green mask to the left. This part of the
151    /// mask can be applied with `green_value << green_mask_shift`.
152    pub fn green_mask_shift(&self) -> u8 {
153        unsafe { self.inner.v0 }.green_mask_shift
154    }
155    /// The size of the blue mask, in bits. This part of the mask can be applied
156    /// with `blue_value & ((1 << blue_mask_size) - 1)`.
157    pub fn blue_mask_size(&self) -> u8 {
158        unsafe { self.inner.v0 }.blue_mask_size
159    }
160    /// The number of bits to shift the blue mask to the left. This part of the
161    /// mask can be applied with `blue_value << blue_mask_shift`.
162    pub fn blue_mask_shift(&self) -> u8 {
163        unsafe { self.inner.v0 }.blue_mask_shift
164    }
165
166    /// The raw EDID bytes of the display attached to this framebuffer.
167    pub fn edid(&self) -> &[u8] {
168        unsafe { core::slice::from_raw_parts(self.inner.v0.edid, self.inner.v0.edid_size as usize) }
169    }
170
171    /// The video modes supported on this framebuffer. Only available on
172    /// revision 1 and above.
173    pub fn modes(&self) -> Option<&[&VideoMode]> {
174        match self.revision {
175            0 => None,
176            1.. => Some(unsafe {
177                core::slice::from_raw_parts(
178                    self.inner.v1.modes.cast(),
179                    self.inner.v1.mode_ct as usize,
180                )
181            }),
182        }
183    }
184}
