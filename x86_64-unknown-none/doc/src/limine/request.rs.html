7macro_rules! impl_base_fns {
8    ($latest_revision:expr, $response:ty, $magic:expr, { $($(#[$attr:meta])* $field:ident: $val:expr),* $(,)? }) => {
9        /// Create a new request with the latest revision.
10        pub const fn new() -> Self {
11            Self::with_revision($latest_revision)
12        }
13
14        /// Create a new request with the given revision.
15        pub const fn with_revision(revision: u64) -> Self {
16            Self {
17                id: $magic,
18                revision,
19                response: Response::none(),
20                $($(#[$attr])* $field: $val),*
21            }
22        }
23
24        /// Get the ID of this request. This includes the magic number and the
25        /// request-specific ID.
26        pub fn id(&self) -> &[u64; 4] {
27            &self.id
28        }
29
30        /// Get the revision of this request.
31        pub fn revision(&self) -> u64 {
32            self.revision
33        }
34
35        /// Get the response to this request, if it has been set.
36        pub fn get_response(&self) -> Option<&$response> {
37            self.response.get()
38        }
39        /// Get a mutable reference to the response to this request, if it has
40        /// been set.
41        ///
42        /// Note that this method takes a mutable reference, so the request will
43        /// have to be wrapped in a mutex or similar in order to use it.
44        pub fn get_response_mut(&mut self) -> Option<&mut $response> {
45            self.response.get_mut()
46        }
47    };
48}

50macro_rules! setter {
51    ($(#[$attr:meta])* $ty:ty, $setter:ident, $with:ident, $field:ident) => {
52        $(#[$attr])*
53        /// This function operates in place.
54        ///
55        /// # Parameters
56        #[doc = concat!("- ", stringify!($field), ": The new value for the field.")]
57        pub fn $setter(&mut self, $field: impl Into<$ty>) {
58            self.$field = $field.into();
59        }
60        $(#[$attr])*
61        /// This function returns the new value.
62        ///
63        /// # Parameters
64        #[doc = concat!("- ", stringify!($field), ": The new value for the field.")]
65        pub const fn $with(mut self, $field: $ty) -> Self {
66            self.$field = $field;
67            self
68        }
69    };
70}

72macro_rules! magic {
73    ($first:expr, $second:expr) => {
74        [0xc7b1dd30df4c8b88, 0x0a82e883a194f07b, $first, $second]
75    };

78struct Response<T> {
79    inner: UnsafeCell<Option<NonNull<T>>>,
80}
81unsafe impl<T: Sync> Sync for Response<T> {}
82unsafe impl<T: Send> Send for Response<T> {}
83impl<T> Response<T> {
84    pub fn get(&self) -> Option<&T> {
85        Some(unsafe { core::ptr::read_volatile(self.inner.get())?.as_ref() })
86    }
87    pub fn get_mut(&mut self) -> Option<&mut T> {
88        Some(unsafe { core::ptr::read_volatile(self.inner.get())?.as_mut() })
89    }
90}
91impl<T> Response<T> {
92    pub const fn none() -> Self {
93        Self {
94            inner: UnsafeCell::new(None),
95        }
96    }
97}

493/// Request information about the loaded modules. See [`File`](file::File) for
494/// more information.
495///
496/// Additionally, with revision 1, this request can be used to specify
497/// additional modules to be loaded without being specified in the configuration
498/// file.
499///
500/// # Revisions
501/// - Revision 0: Initial revision.
502/// - Revision 1: added `internal_modules`
503///
504/// # Usage
505/// ```rust
506/// # use limine::{request::ModuleRequest, response::ModuleResponse, modules::*, BaseRevision};
507/// static BASE_REVISION: BaseRevision = BaseRevision::new();
508///
509/// // Try to load our own module
510/// static MODULE_REQUEST: ModuleRequest =
511///    ModuleRequest::new().with_internal_modules(&[
512///        &InternalModule::new().with_path(limine::cstr!("/path/to/a/module"))
513///    ]);
514///
515/// # fn dummy<'a>() -> Option<&'a ModuleResponse> {
516/// // ...later, in our code
517/// MODULE_REQUEST.get_response() // ...
518/// # }
519/// ```
520#[repr(C)]
521pub struct ModuleRequest {
522    id: [u64; 4],
523    revision: u64,
524    response: Response<ModuleResponse>,
525
526    // Revision 1+
527    internal_module_ct: u64,
528    internal_modules: *const *const InternalModule,
529}
530unsafe impl Sync for ModuleRequest {}
531unsafe impl Send for ModuleRequest {}
532impl ModuleRequest {
533    impl_base_fns!(
534        1,
535        ModuleResponse,
536        magic!(0x3e7e279702be32af, 0xca1c4f3bd1280cee),
537        {
538            internal_module_ct: 0,
539            internal_modules: core::ptr::null(),
540        }
541    );
542
543    /// Seet the internal modules to be loaded. Only available on revision 1+.
544    /// This function operates in place.
545    ///
546    /// # Parameters
547    /// - `modules`: The new value of the field.
548    pub fn set_internal_modules(&mut self, modules: &'static [&'static InternalModule]) {
549        self.internal_module_ct = modules.len() as u64;
550        self.internal_modules = modules.as_ptr().cast();
551    }
552
553    /// Set the internal modules to be loaded. Only available on revision 1+.
554    /// This function returns the new value.
555    ///
556    /// # Parameters
557    /// - `modules`: The new value of the field.
558    pub const fn with_internal_modules(
559        mut self,
560        modules: &'static [&'static InternalModule],
561    ) -> Self {
562        self.internal_module_ct = modules.len() as u64;
563        self.internal_modules = modules.as_ptr().cast();
564        self
565    }
566
567    /// Get the internal modules to be loaded. Only available on revision 1.
568    pub fn internal_modules(&self) -> &[&InternalModule] {
569        unsafe {
570            core::slice::from_raw_parts(
571                self.internal_modules.cast(),
572                self.internal_module_ct as usize,
573            )
574        }
575    }
576}
