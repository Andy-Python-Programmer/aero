1//! Auxiliary types for the [memory map request](crate::request::MemoryMapRequest)
2
3/// A type of entry within the memory map.
4#[repr(transparent)]
5#[derive(PartialEq, Eq, Clone, Copy)]
6pub struct EntryType(u64);
7impl EntryType {
8    /// The memory region is freely usable.
9    pub const USABLE: Self = Self(0);
10    /// The memory region is permanently reserved.
11    pub const RESERVED: Self = Self(1);
12    /// The memory region is currently used by ACPI, but can be reclaimed once
13    /// ACPI structures are no longer needed.
14    pub const ACPI_RECLAIMABLE: Self = Self(2);
15    /// The memory region is permanently reserved by ACPI, and must not be used.
16    pub const ACPI_NVS: Self = Self(3);
17    /// The memory region is unusable due to physical damage or similar errors.
18    pub const BAD_MEMORY: Self = Self(4);
19    /// The memory region is used by the bootloader, but can be reclaimed once
20    /// all responses have been processed and will no longer be used.
21    pub const BOOTLOADER_RECLAIMABLE: Self = Self(5);
22    /// The memory region is used by the kernel and modules, and as such is
23    /// permanently reserved.
24    pub const KERNEL_AND_MODULES: Self = Self(6);
25    /// The memory region is used by the framebuffer, and as such is permanently
26    /// reserved.
27    pub const FRAMEBUFFER: Self = Self(7);
28}
29impl From<u64> for EntryType {
30    fn from(val: u64) -> Self {
31        Self(val)
32    }
33}

35/// A memory map entry.
36#[repr(C)]
37pub struct Entry {
38    /// The base of the memory region, in *physical space*.
39    pub base: u64,
40    /// The length of the memory region, in bytes.
41    pub length: u64,
42    /// The type of the memory region. See [`EntryType`] for specific values.
43    pub entry_type: EntryType,
44}
