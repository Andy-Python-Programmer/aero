1//! Auxiliary types for the [module request](crate::request::ModuleRequest)
2
3use core::ffi::{c_char, CStr};
4
5use bitflags::bitflags;
6
7bitflags! {
8    /// Flags for internal modules
9    #[derive(PartialEq, Eq, Clone, Copy)]
10    pub struct ModuleFlags: u64 {
11        /// The module is required. If it is not found, the bootloader will
12        /// refuse to boot.
13        const REQUIRED = 1 << 0;
14        /// The module is GZ-compressed and will be uncompressed by the
15        /// bootloader. This is only honored on response revision 2 and greater.
16        const COMPRESSED = 1 << 1;
17    }
18}
19
20/// Create a NUL-terminated C string from a string literal
21#[macro_export]
22macro_rules! cstr {
23    () => {
24        unsafe { core::ffi::CStr::from_bytes_with_nul_unchecked(b"\0") }
25    };
26    ($s:expr) => {
27        unsafe { core::ffi::CStr::from_bytes_with_nul_unchecked(concat!($s, "\0").as_bytes()) }
28    };
29}

31/// An internal module that the kernel requests from the bootloader. Only
32/// available with request revision 1 and greater.
33#[repr(C)]
34pub struct InternalModule {
35    path: *const c_char,
36    cmdline: *const c_char,
37    flags: ModuleFlags,
38}
39unsafe impl Sync for InternalModule {}
40unsafe impl Send for InternalModule {}
41impl InternalModule {
42    /// Create a new internal module with no path, no cmdline and no flags.
43    pub const fn new() -> Self {
44        Self {
45            path: b"\0".as_ptr().cast(),
46            cmdline: b"\0".as_ptr().cast(),
47            flags: ModuleFlags::empty(),
48        }
49    }
50
51    /// Set the path of the internal module. This function returns the new value.
52    ///
53    /// # Parameters
54    /// - `path`: The new value of the field.
55    pub const fn with_path(mut self, path: &'static CStr) -> Self {
56        self.path = path.as_ptr();
57        self
58    }
59    /// Set the path of the internal module. This function operates in place.
60    ///
61    /// # Parameters
62    /// - `path`: The new value of the field.
63    pub fn set_path(&mut self, path: &'static CStr) {
64        self.path = path.as_ptr();
65    }
66
67    /// Set the command-line for the module. This function returns the new value.
68    ///
69    /// # Parameters
70    /// - `cmdline`: The new value of the field.
71    pub const fn with_cmdline(mut self, cmdline: &'static CStr) -> Self {
72        self.cmdline = cmdline.as_ptr();
73        self
74    }
75    /// Set the command-line for the module. This function operates in place.
76    ///
77    /// # Parameters
78    /// - `cmdline`: The new value of the field.
79    pub fn set_cmdline(&mut self, cmdline: &'static CStr) {
80        self.cmdline = cmdline.as_ptr();
81    }
82
83    /// Set the flags for the module. This function returns the new value.
84    ///
85    /// # Parameters
86    /// - `flags`: The new value of the field.
87    pub const fn with_flags(mut self, flags: ModuleFlags) -> Self {
88        self.flags = flags;
89        self
90    }
91    /// Set the flags for the module. This function operates in place.
92    ///
93    /// # Parameters
94    /// - `flags`: The new value of the field.
95    pub fn set_flags(&mut self, flags: ModuleFlags) {
96        self.flags = flags;
97    }
98
99    /// Returns the module's path as a byte slice with unspecified encoding.
100    pub fn path(&self) -> &[u8] {
101        unsafe { CStr::from_ptr(self.path).to_bytes() }
102    }
103    /// Returns the module's command-line as a byte slice with unspecified
104    /// encoding.
105    pub fn cmdline(&self) -> &[u8] {
106        unsafe { CStr::from_ptr(self.cmdline).to_bytes() }
107    }
108    /// Returns the module's flags.
109    pub fn flags(&self) -> ModuleFlags {
110        self.flags
111    }
112}
