32/// Support for kernel unit-testing framework.
33///
34/// ## Example
35/// ```rust,no_run
36/// #[test]
37/// fn some_test() {
38///     assert_eq!(2 + 2, 4);
39/// }
40/// ```
41#[proc_macro_attribute]
42#[proc_macro_error]
43pub fn test(attr: TokenStream, item: TokenStream) -> TokenStream {
44    test::parse(attr, item)
45}

47/// Validates input buffers, structures, path and strings auto-magically.
48///
49/// Functions that use this macro are not allowed to be `async`, `unsafe`, or `const` and must
50/// have a valid return-type of `Result<usize, AeroSyscallError>`. In addition, the function cannot
51/// have generic parameters.
52#[proc_macro_attribute]
53#[proc_macro_error]
54pub fn syscall(attr: TokenStream, item: TokenStream) -> TokenStream {
55    syscall::parse(attr, item)
56}
