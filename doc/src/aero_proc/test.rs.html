18use proc_macro::TokenStream;
19use syn::ItemFn;
20
21pub fn parse(_: TokenStream, item: TokenStream) -> TokenStream {
22    let input = syn::parse_macro_input!(item as ItemFn);
23
24    let name = &input.sig.ident;
25    let body = &input.block;
26
27    let marker_name = quote::format_ident!("{}_test_marker", name);
28    let result = quote::quote! {
29        #[test_case]
30        static #marker_name: crate::tests::Test = crate::tests::Test {
31            test_fn: #name,
32            path: concat!(module_path!(), "::", stringify!(#name))
33        };
34
35        fn #name() {
36            #body
37        }
38    };
39
40    result.into()
41}
